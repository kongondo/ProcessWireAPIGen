 324:      * - You may optionally omit the `$beautify` and/or `$maxLength` arguments and substitute the `$options` array instead.
 325:      * 
 326:      * - When substituted, the beautify and maxLength options can be specified in $options as well.
 327:      * 
 328:      * - If `$config->pageNameCharset` is "UTF8" then non-ASCII page names will be converted to punycode ("xn-") ASCII page names,
 329:      *   rather than converted, regardless of `$beautify` setting. 
 330:      * 
 331:      * ~~~~~
 332:      * $test = "Hello world!";
 333:      * echo $sanitizer->pageName($test, true); // outputs: hello-world
 334:      * ~~~~~
 335:      * 
 336:      * #pw-group-strings
 337:      * #pw-group-pages
 338:      *
 339:      * @param string $value Value to sanitize as a page name
 340:      * @param bool|int|array $beautify This argument accepts a few different possible values (default=false): 
 341:      *  - `true` (boolean): Make it pretty. Use this when using a pageName for the first time. 
 342:      *  - `Sanitizer::translate` (constant): This will make it translate non-ASCII letters based on *InputfieldPageName* module config settings. 
 343:      *  - `$options` (array): You can optionally specify the $options array for this argument instead. 
 344:      * @param int|array $maxLength Maximum number of characters allowed in the name.
 345:      *  You may also specify the $options array for this argument instead. 
 346:      * @param array $options Array of options to modify default behavior. See Sanitizer::name() method for available options. 
 347:      * @return string
 348:      * @see Sanitizer::name()
 349:      *
 350:      */
 351:     public function pageName($value, $beautify = false, $maxLength = 128, array $options = array()) {
 352:     
 353:         if(!strlen($value)) return '';
 354:         
 355:         $defaults = array(
 356:             'charset' => $this->wire('config')->pageNameCharset
 357:         );
 358:         
 359:         if(is_array($beautify)) {
 360:             $options = array_merge($beautify, $options);
 361:             $beautify = isset($options['beautify']) ? $options['beautify'] : false;
 362:             $maxLength = isset($options['maxLength']) ? $options['maxLength'] : 128;
 363:         } else if(is_array($maxLength)) {
 364:             $options = array_merge($maxLength, $options);
 365:             $maxLength = isset($options['maxLength']) ? $options['maxLength'] : 128;
 366:         } else {
 367:             $options = array_merge($defaults, $options);
 368:         }
 369:         
 370:         if($options['charset'] !== 'UTF8' && is_int($beautify) && $beautify > self::translate) {
 371:             // UTF8 beautify modes aren't available if $config->pageNameCharset is not UTF8
 372:             if(in_array($beautify, array(self::toAscii, self::toUTF8, self::okUTF8))) {
 373:                 // if modes aren't supported, disable 
 374:                 $beautify = false;
 375:             }
 376:         }
 377:         
 378:         if($beautify === self::toAscii) {
 379:             // convert UTF8 to ascii (IDN/punycode)
 380:             $beautify = false;
 381:             if(strlen($value) > $maxLength) $value = substr($value, 0, $maxLength);
 382:             $_value = $value;
 383:             
 384:             if(!ctype_alnum($value)
 385:                 && !ctype_alnum(str_replace(array('-', '_', '.'), '', $value)) 
 386:                 && strpos($value, 'xn-') !== 0) {
 387:                 
 388:                 do {
 389:                     // encode value
 390:                     $value = $this->punyEncodeName($_value);
 391:                     // if result stayed within our allowed character limit, then good, we're done
 392:                     if(strlen($value) <= $maxLength) break;
 393:                     // continue loop until encoded value is equal or less than allowed max length
 394:                     $_value = substr($_value, 0, strlen($_value) - 1);
 395:                 } while(true);
 396:                 
 397:                 // if encode was necessary and successful, return with no further processing
 398:                 if(strpos($value, 'xn-') === 0) {
 399:                     return $value;
 400:                 } else {
 401:                     // can't be encoded, send to regular name sanitizer
 402:                     $value = $_value;
 403:                 }
 404:             }
 405:             
 406:         } else if($beautify === self::toUTF8) {
 407:             // convert ascii IDN/punycode to UTF8
 408:             $beautify = self::okUTF8;
 409:             if(strpos($value, 'xn-') === 0) {
 410:                 // found something to convert
 411:                 $value = $this->punyDecodeName($value);
 412:                 // now it will run through okUTF8
 413:             }
 414:         }
 415:         
 416:         if($beautify === self::okUTF8) {
 417:             return $this->pageNameUTF8($value);
 418:         }
 419:         
 420:         return strtolower($this->name($value, $beautify, $maxLength, '-', $options));
 421:     }

 440:     /**
 441:      * Sanitize and allow for UTF-8 characters in page name
 442:      * 
 443:      * - If `$config->pageNameCharset` is not `UTF8` then this function just passes control to the regular page name sanitizer.
 444:      * - Allowed UTF-8 characters are determined from `$config->pageNameWhitelist`.
 445:      * - This method does not convert to or from UTF-8, it only sanitizes it against the whitelist. 
 446:      * - If given a value that has only ASCII characters, this will pass control to the regular page name sanitizer. 
 447:      * 
 448:      * #pw-group-strings
 449:      * #pw-group-pages
 450:      * 
 451:      * @param string $value Value to sanitize
 452:      * @param int $maxLength Maximum number of characters allowed
 453:      * @return string Sanitized value
 454:      *
 455:      */
 456:     public function pageNameUTF8($value, $maxLength = 128) {
 457:         
 458:         if(!strlen($value)) return '';
 459:         
 460:         // if UTF8 module is not enabled then delegate this call to regular pageName sanitizer
 461:         if($this->wire('config')->pageNameCharset != 'UTF8') return $this->pageName($value, false, $maxLength);
 462:     
 463:         // we don't allow UTF8 page names to be prefixed with "xn-"
 464:         if(strpos($value, 'xn-') === 0) $value = substr($value, 3);
 465: 
 466:         // word separators that we always allow 
 467:         $separators = array('.', '-', '_'); 
 468:     
 469:         // we let regular pageName handle chars like these, if they appear without other UTF-8
 470:         $extras = array('.', '-', '_', ' ', ',', ';', ':', '(', ')', '!', '?', '&', '%', '$', '#', '@');
 471: 
 472:         // proceed only if value has some non-ascii characters
 473:         if(ctype_alnum(str_replace($extras, '', $value))) return $this->pageName($value, false, $maxLength);
 474: 
 475:         // validate that all characters are in our whitelist
 476:         $whitelist = $this->wire('config')->pageNameWhitelist;
 477:         if(!strlen($whitelist)) $whitelist = false;
 478:         $blacklist = '/\\%"\'<>?#@:;,+=*^$()[]{}|&';
 479:         $replacements = array();
 480: 
 481:         for($n = 0; $n < mb_strlen($value); $n++) {
 482:             $c = mb_substr($value, $n, 1);
 483:             if(!strlen(trim($c)) || ctype_cntrl($c)) {
 484:                 // character does not resolve to something visible
 485:                 $replacements[] = $c;
 486:             } else if(mb_strpos($blacklist, $c) !== false || strpos($blacklist, $c) !== false) {
 487:                 // character that is in blacklist
 488:                 $replacements[] = $c;
 489:             } else if($whitelist !== false && mb_strpos($whitelist, $c) === false) {
 490:                 // character that is not in whitelist, double check case variants
 491:                 $cLower = mb_strtolower($c);
 492:                 $cUpper = mb_strtoupper($c);
 493:                 if($cLower !== $c && mb_strpos($whitelist, $cLower) !== false) {
 494:                     // allow character and convert to lowercase variant
 495:                     $value = mb_substr($value, 0, $n) . $cLower . mb_substr($value, $n+1);
 496:                 } else if($cUpper !== $c && mb_strpos($whitelist, $cUpper) !== false) {
 497:                     // allow character and convert to uppercase varient
 498:                     $value = mb_substr($value, 0, $n) . $cUpper . mb_substr($value, $n+1);
 499:                 } else {
 500:                     // queue character to be replaced
 501:                     $replacements[] = $c;
 502:                 }
 503:             }
 504:         }
 505: 
 506:         // replace disallowed characters with "-"
 507:         if(count($replacements)) $value = str_replace($replacements, '-', $value);
 508:         
 509:         // replace doubled word separators
 510:         foreach($separators as $c) {
 511:             while(strpos($value, "$c$c") !== false) {
 512:                 $value = str_replace("$c$c", $c, $value);
 513:             }
 514:         }
 515: 
 516:         // trim off any remaining separators/extras
 517:         $value = trim($value, '-_.');
 518:         
 519:         if(mb_strlen($value) > $maxLength) $value = mb_substr($value, 0, $maxLength); 
 520:         
 521:         return $value;
 522:     }

 524:     /**
 525:      * Decode a PW-punycode'd name value
 526:      * 
 527:      * @param string $value
 528:      * @return string
 529:      * 
 530:      */
 531:     protected function punyDecodeName($value) {
 532:         // exclude values that we know can't be converted
 533:         if(strlen($value) < 4 || strpos($value, 'xn-') !== 0) return $value;
 534:         
 535:         if(strpos($value, '__')) {
 536:             $_value = $value;
 537:             $parts = explode('__', $_value);
 538:             foreach($parts as $n => $part) {
 539:                 $parts[$n] = $this->punyDecodeName($part);
 540:             }
 541:             $value = implode('', $parts);
 542:             return $value; 
 543:         }
 544:         
 545:         $_value = $value; 
 546:         // convert "xn-" single hyphen to recognized punycode "xn--" double hyphen
 547:         if(strpos($value, 'xn--') !== 0) $value = 'xn--' . substr($value, 3);
 548:         if(function_exists('idn_to_utf8')) {
 549:             // use native php function if available
 550:             $value = idn_to_utf8($value);
 551:         } else {
 552:             // otherwise use Punycode class
 553:             $pc = new Punycode();
 554:             $value = $pc->decode($value);
 555:         }
 556:         // if utf8 conversion failed, restore original value
 557:         if($value === false || !strlen($value)) $value = $_value;
 558:         return $value;
 559:     }

 561:     /**
 562:      * Encode a name value to PW-punycode
 563:      * 
 564:      * @param string $value
 565:      * @return string
 566:      * 
 567:      */
 568:     protected function punyEncodeName($value) {
 569:         // exclude values that don't need to be converted
 570:         if(strpos($value, 'xn-') === 0) return $value;
 571:         if(ctype_alnum(str_replace(array('.', '-', '_'), '', $value))) return $value;
 572: 
 573:         while(strpos($value, '__') !== false) {
 574:             $value = str_replace('__', '_', $value);
 575:         }
 576:         
 577:         if(strlen($value) >= 50) {
 578:             $_value = $value;
 579:             $parts = array();
 580:             while(strlen($_value)) {
 581:                 $part = mb_substr($_value, 0, 12);
 582:                 $_value = mb_substr($_value, 12);
 583:                 $parts[] = $this->punyEncodeName($part);
 584:             }
 585:             $value = implode('__', $parts);
 586:             return $value; 
 587:         }
 588:         
 589:         $_value = $value;
 590:         
 591:         if(function_exists("idn_to_ascii")) {
 592:             // use native php function if available
 593:             $value = substr(idn_to_ascii($value), 3);
 594:         } else {
 595:             // otherwise use Punycode class
 596:             $pc = new Punycode();
 597:             $value = substr($pc->encode($value), 3);
 598:         }
 599:         if(strlen($value) && $value !== '-') {
 600:             // in PW the xn- prefix has one fewer hyphen than in native Punycode
 601:             // for compatibility with pageName sanitization and beautification
 602:             $value = "xn-$value";
 603:         } else {
 604:             // fallback to regular 'name' sanitization on failure, ensuring that
 605:             // return value is always ascii
 606:             $value = $this->name($_value);
 607:         }
 608:         return $value;
 609:     }
