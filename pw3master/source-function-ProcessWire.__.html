 13: /**
 14:  * Perform a language translation
 15:  * 
 16:  * ~~~~~~
 17:  * echo __('This is translatable text');
 18:  * echo __('Translatable with current file as textdomain', __FILE__); 
 19:  * echo __('Translatable with other file as textdomain', '/site/templates/_init.php');
 20:  * ~~~~~~
 21:  * 
 22:  * @param string $text Text for translation. 
 23:  * @param string $textdomain Textdomain for the text, may be class name, filename, or something made up by you. If omitted, a debug backtrace will attempt to determine it automatically.
 24:  * @param string $context Name of context - DO NOT USE with this function for translation as it won't be parsed for translation. Use only with the _x() function, which will be parsed. 
 25:  * @return string Translated text or original text if translation not available.
 26:  *
 27:  */
 28: function __($text, $textdomain = null, $context = '') {
 29:     static $useLimit = null;
 30:     if(!wire('languages')) return $text; 
 31:     if(!$language = wire('user')->language) return $text; 
 32:     /** @var Language $language */
 33:     if(!$language->id) return $text; 
 34:     if($useLimit === null) {
 35:         $useLimit = version_compare(PHP_VERSION, '5.4.0') >= 0;
 36:     }
 37:     if(is_null($textdomain)) {
 38:         if($useLimit) {
 39:             // PHP 5.4.0 or newer
 40:             $traces = @debug_backtrace(DEBUG_BACKTRACE_IGNORE_ARGS, 2);
 41:         } else if(defined('DEBUG_BACKTRACE_IGNORE_ARGS')) {
 42:             // PHP 5.3.6 or newer
 43:             $traces = @debug_backtrace(DEBUG_BACKTRACE_IGNORE_ARGS);
 44:         } else {
 45:             // older PHP (deprecated)
 46:             $traces = @debug_backtrace();
 47:         }
 48:         if(isset($traces[0]) && $traces[0]['file'] != __FILE__) {
 49:             $textdomain = $traces[0]['file'];
 50:         } else if(isset($traces[1]) && $traces[1]['file'] != __FILE__) {
 51:             $textdomain = $traces[1]['file'];
 52:         }
 53:         if(is_null($textdomain)) $textdomain = 'site';
 54:     } else if($textdomain === 'common') {
 55:         // common translation
 56:         $textdomain = 'wire/modules/LanguageSupport/LanguageTranslator.php';
 57:     }
 58:     $value = $language->translator()->getTranslation($textdomain, $text, $context);
 59:     if($value === "=") {
 60:         $value = $text;
 61:     } else if($value === "+") {
 62:         $v = $language->translator()->commonTranslation($text);
 63:         $value = empty($v) ? $text : $v;
 64:     } else {
 65:         $value = htmlspecialchars($value, ENT_QUOTES, 'UTF-8', false);
 66:     }
 67:     return $value;
 68: }

 70: /**
 71:  * Perform a language translation in a specific context
 72:  * 
 73:  * Used when to text strings might be the same in English, but different in other languages. 
 74:  * 
 75:  * ~~~~~
 76:  * echo _x('Click for more', 'button');
 77:  * echo _x('Click for more', 'text-link'); 
 78:  * ~~~~~
 79:  * 
 80:  * @param string $text Text for translation. 
 81:  * @param string $context Name of context
 82:  * @param string $textdomain Textdomain for the text, may be class name, filename, or something made up by you. If omitted, a debug backtrace will attempt to determine automatically.
 83:  * @return string Translated text or original text if translation not available.
 84:  *
 85:  */
 86: function _x($text, $context, $textdomain = null) {
 87:     return __($text, $textdomain, $context);    
 88: }

 90: /**
 91:  * Perform a language translation with singular and plural versions
 92:  * 
 93:  * ~~~~~
 94:  * $items = array(...);
 95:  * $qty = count($items);
 96:  * echo _n('Found one item', 'Found multiple items', $qty); 
 97:  * echo sprintf(_n('Found one item', 'Found %d items', $qty), $qty);
 98:  * ~~~~~
 99:  * 
100:  * @param string $textSingular Singular version of text (when there is 1 item)
101:  * @param string $textPlural Plural version of text (when there are multiple items or 0 items)
102:  * @param int $count Quantity of items, should be 0 or more.
103:  * @param string $textdomain Textdomain for the text, may be class name, filename, or something made up by you. If omitted, a debug backtrace will attempt to determine automatically.
104:  * @return string Translated text or original text if translation not available.
105:  *
106:  */
107: function _n($textSingular, $textPlural, $count, $textdomain = null) {
108:     return $count == 1 ? __($textSingular, $textdomain) : __($textPlural, $textdomain);     
109: }
