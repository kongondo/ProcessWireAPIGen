118:     public function ___renderRepeaterLabel($label, $cnt, Page $page) {
119: 
120:         // situations where we skip the render of repeater label because it is not needed
121:         $repeaterEditID = (int) $this->wire('input')->get('repeater_edit');
122:         if($repeaterEditID && $repeaterEditID == $page->id) {
123:             // edit of item requested in URL that matches given $page
124:             return $label;
125:         } else if(count($_POST) && !$this->wire('config')->ajax) {
126:             // POST request that is not ajax
127:             return $label;
128:         }

130:         $out = '';

132:         
133:         if($page->id && $repeaterTitle) {
134:             // custom repeater titles specified
135:             $hasCnt = stripos($repeaterTitle, '#n') !== false;
136:             
137:             // update index numbers?
138:             if($hasCnt) {
139:                 // replace "#n" with index number of repeater item
140:                 $repeaterTitle = str_replace("#n", "#$cnt", $repeaterTitle);
141:             }
142:             
143:             if(strpos($repeaterTitle, '{') !== false) {
144:                 // formatted {label}
145:                 $out = $page->getMarkup($repeaterTitle);
146: 
147:             } else if(!$hasCnt && $this->wire('sanitizer')->fieldName($repeaterTitle) === $repeaterTitle) {
148:                 // just a single field name
149:                 $value = $page->getFormatted($repeaterTitle);
150:                 if(is_object($value)) {
151:                     if($value instanceof Page) {
152:                         $out = $value->get('title|name');
153:                     } else if($value instanceof PageArray) {
154:                         $out = $value->implode(', ');
155:                     } else {
156:                         $out = (string) $value;
157:                     }
158:                 }
159:             } else {
160:                 // label, but with no page variables
161:                 $out = $repeaterTitle;
162:             }
163:             
164:             $out = strip_tags(trim($out));
165:         }
166:         
167:         if(!strlen($out)) {
168:             // fallback to default
169:             if(!strlen($label)) $label = $this->field->getLabel();
170:             $out = "$label #" . $cnt;
171:         }
172:     
173:         // note {brackets} surround text that will be visually muted from the JS side
174:         if(!$page->id) {
175:             // non-editable new item
176:             $out .= ' {• ' . $this->_('This item will become editable after you save.') . '}';
177:         } else if($page->isUnpublished() && $page->hasStatus(Page::statusOn)) {
178:             // editable new item

187:         }

189:         return $out;

730:     /**
731:      * Process the input from a submitted repeaters field
732:      * 
733:      * @param WireInputData $input
734:      * @return $this
735:      *
736:      */
737:     public function ___processInput(WireInputData $input) {
738:         
739:         $isSingle = $this->singleMode; 
740:         
741:         /** @var PageArray $value */
742:         $value = $this->attr('value');
743:         $loadedIDs = array();
744:     
745:         // determine which repeater pages have data posted in this request
746:         foreach($value as $key => $page) {
747:             $loadedName = "loaded_repeater$page->id";
748:             if($isSingle || ((int) $input->$loadedName) > 0) $loadedIDs[$page->id] = $page->id;
749:         }
750:     
751:         $this->buildForm(0, $loadedIDs);
752:         
753:         $numChanges = 0;
754:         $sortChanged = false;
755:         $value->setTrackChanges(true);
756:         $pageIDs = array();
757: 
758:         // existing items
759:         foreach($value as $key => $page) {
760:             
761:             /** @var RepeaterPage $page */
762:             $pageIDs[] = $page->id;
763:             
764:             $isHidden = $page->isHidden();
765:             $isUnpublished = $page->isUnpublished();
766:             $isOn = $page->hasStatus(Page::statusOn);
767: 
768:             if($isSingle) {
769:                 $publishName = '';
770:             } else {
771:                 $deleteName = "delete_repeater{$page->id}";
772:                 $sortName = "sort_repeater{$page->id}";
773:                 $publishName = "publish_repeater{$page->id}";
774:                 $depthName = "depth_repeater{$page->id}";
775: 
776:                 if($input->$deleteName == $page->id) {
777:                     // @todo add check to Fieldgroups::isFieldNotRemoveable() before attempting remove
778:                     $value->remove($page);
779:                     $numChanges++;
780:                     continue;
781:                 }
782: 
783:                 $sort = $input->$sortName;
784:                 // skip pages that don't appear in the POST data (most likely ready pages)
785:                 if(is_null($sort)) continue;
786: 
787:                 $page->sort = (int) $sort;
788:                 if($page->isChanged('sort')) {
789:                     // $this->message("Sort changed for field {$this->field} page {$page->id}", Notice::debug); 
790:                     $sortChanged = true;
791:                 }
792: 
793:                 if($this->repeaterDepth > 0) {
794:                     $depth = (int) $input->$depthName;
795:                     if($page->getDepth() != $depth) {
796:                         $page->setDepth($depth);
797:                         $numChanges++;
798:                     }
799:                 }
800:             }
801:             
802:             /** @var InputfieldWrapper $wrapper */
803:             $wrapper = $this->wrappers[$page->id]; 
804:             $wrapper->resetTrackChanges(true); 
805:             $wrapper->processInput($input);
806:             
807:             $numErrors = count($wrapper->getErrors());
808:             $page->setQuietly('_repeater_errors', $numErrors); // signal to FieldtypeRepeater::savePageField() that page has errors
809:             $page->setQuietly('_repeater_processed', true); // signal to FieldtypeRepeater::savePageField() that page had input processed
810:             $this->formToPage($wrapper, $page);
811:             $publish = $isSingle ? 0 : (int) $input->$publishName;
812:             
813:             if($publish > 0 && ($isHidden || $isUnpublished)) {
814:                 // publish requested (publish=1)
815:                 if($isHidden) $page->removeStatus(Page::statusHidden);
816:                 if(!$numErrors && $isUnpublished) $page->removeStatus(Page::statusUnpublished);
817:                 if(!$isOn) $page->addStatus(Page::statusOn);
818:                 
819:             } else if($publish < 0) {
820:                 // unpublish requested (publish=-1)
821:                 if($isOn) $page->removeStatus(Page::statusOn);
822:                 if($isHidden) $page->removeStatus(Page::statusHidden);
823:                 if(!$isUnpublished) $page->addStatus(Page::statusUnpublished);
824:                 
825:             } else if(!$isOn) {
826:                 // no publish change requested, just ensure page is on
827:                 $page->addStatus(Page::statusOn);
828:             }
829: 
830:             if($page->isChanged() && $this->page->id) $numChanges++;
831:         }
832: 
833:         // if the sort changed, then tell the PageArray to sort by _repeater_sort
834:         if($sortChanged) { 
835:             $this->value->sort('sort'); 
836:             $numChanges++;
837:         }
838: 
839:         if(!$isSingle && $this->field->get('repeaterLoading') == FieldtypeRepeater::loadingOff) {   
840:             $numNewItems = (int) $input["_{$this->name}_add_items"];
841:             if($numNewItems) {
842:                 // iterate through each new item added for non-ajax mode
843:                 for($n = 0; $n < $numNewItems; $n++) {
844:                     $page = $this->getNextReadyPage($pageIDs);
845:                     $page->removeStatus(Page::statusHidden);
846:                     $page->sort = count($value)+1;
847:                     $value->add($page);
848:                     $numChanges++;
849:                 }
850:             }
851:         }
852:         
853:         // if changes occurred, then tell $this->page and the PageArray $value
854:         if($numChanges) {
855:             $this->page->trackChange($this->attr('name'));
856:             $this->trackChange('value');
857:         }
858: 
859:         return $this; 
860:     }

903:     }
