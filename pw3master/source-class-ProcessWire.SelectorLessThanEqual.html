  3: /**
  4:  * ProcessWire Selector base type and implementation for various Selector types
  5:  *
  6:  * Selectors hold a field, operator and value and are used in finding things
  7:  *
  8:  * This file provides the base implementation for a Selector, as well as implementation
  9:  * for several actual Selector types under the main Selector class. 
 10:  * 
 11:  * ProcessWire 3.x, Copyright 2016 by Ryan Cramer
 12:  * https://processwire.com
 13:  *
 14:  * #pw-summary Selector maintains a single selector consisting of field name, operator, and value.
 15:  *
 16:  * #pw-body =
 17:  * - Serves as the base class for the different Selector types (`SelectorEqual`, `SelectorNotEqual`, `SelectorLessThan`, etc.)
 18:  * - The constructor requires `$field` and `$value` properties which may either be an array or string. 
 19:  *   An array indicates multiple items in an OR condition. Multiple items may also be specified by
 20:  *   pipe “|” separated strings.  
 21:  * - Operator is determined by the Selector class name, and thus may not be changed without replacing
 22:  *   the entire Selector. 
 23:  * 
 24:  * ~~~~~
 25:  * // very basic usage example
 26:  * // constructor takes ($field, $value) which can be strings or arrays
 27:  * $s = new SelectorEqual('title', 'About Us');
 28:  * // $page can be any kind of Wire-derived object
 29:  * if($s->matches($page)) {
 30:  *   // $page has title "About Us"
 31:  * }
 32:  * ~~~~~
 33:  * ~~~~~
 34:  * // another usage example
 35:  * $s = new SelectorContains('title|body|summary', 'foo|bar'); 
 36:  * if($s->matches($page)) {
 37:  *   // the title, body or summary properties of $page contain either the text "foo" or "bar" 
 38:  * }
 39:  * ~~~~~
 40:  * 
 41:  * ### List of core selector-derived classes
 42:  * 
 43:  * - `SelectorEqual`
 44:  * - `SelectorNotEqual`
 45:  * - `SelectorGreaterThan`
 46:  * - `SelectorLessThan`
 47:  * - `SelectorGreaterThanEqual`
 48:  * - `SelectorLessThanEqual`
 49:  * - `SelectorContains`
 50:  * - `SelectorContainsLike`
 51:  * - `SelectorContainsWords`
 52:  * - `SelectorStarts`
 53:  * - `SelectorStartsLike`
 54:  * - `SelectorEnds`
 55:  * - `SelectorEndsLike`
 56:  * - `SelectorBitwiseAnd`
 57:  * 
 58:  * #pw-body
 59:  * 
 60:  * @property array $fields Fields that were present in selector (same as $field, but always an array).
 61:  * @property string|array $field Field or fields present in the selector (string if single, or array of strings if multiple). Preferable to use $fields property instead.
 62:  * @property-read string $operator Operator used by the selector.
 63:  * @property array $values Values that were present in selector (same as $value, but always array).
 64:  * @property string|array $value Value or values present in the selector (string if single, or array of strings if multiple). Preferable to use $values property instead.
 65:  * @property bool $not Is this a NOT selector? Indicates the selector returns the opposite if what it would otherwise. #pw-group-properties
 66:  * @property string|null $group Group name for this selector (if field was prepended with a "group_name@"). #pw-group-properties
 67:  * @property string $quote Type of quotes value was in, or blank if it was not quoted. One of: '"[{( #pw-group-properties
 68:  * @property-read string $str String value of selector, i.e. “a=b”. #pw-group-properties
 69:  * @property null|bool $forceMatch When boolean, it forces match (true) or non-match (false). (default=null) #pw-group-properties
 70:  * 
 71:  */
 72: abstract class Selector extends WireData {
 73: 
 74:     /**
 75:      * Given a field name and value, construct the Selector. 
 76:      *
 77:      * If the provided $field is an array or pipe "|" separated string, Selector may match any of them (OR field condition)
 78:      * If the provided $value is an array of pipe "|" separated string, Selector may match any one of them (OR value condition).
 79:      * 
 80:      * If only one field is provided as a string, and that field is prepended by an exclamation point, i.e. !field=something
 81:      * then the condition is reversed. 
 82:      *
 83:      * @param string|array $field 
 84:      * @param string|int|array $value 
 85:      *
 86:      */
 87:     public function __construct($field, $value) {
 88: 
 89:         $not = false; 
 90:         if(!is_array($field) && isset($field[0]) && $field[0] == '!') {
 91:             $not = true; 
 92:             $field = ltrim($field, '!'); 
 93:         }
 94: 
 95:         $this->set('field', $field);    
 96:         $this->set('value', $value); 
 97:         $this->set('not', $not); 
 98:         $this->set('group', null); // group name identified with 'group_name@' before a field name
 99:         $this->set('quote', ''); // if $value in quotes, this contains either: ', ", [, {, or (, indicating quote type (set by Selectors class)
100:         $this->set('forceMatch', null); // boolean true to force match, false to force non-match
101:     }
102: 
103:     /**
104:      * Return the operator used by this Selector
105:      * 
106:      * @return string
107:      * @since 3.0.42 Prior versions just supported the 'operator' property.
108:      * 
109:      */
110:     public function operator() {
111:         return static::getOperator();
112:     }
113: 
114:     /**
115:      * Get the field(s) of this Selector
116:      * 
117:      * Note that if calling this as a property (rather than a method) it can return either a string or an array.
118:      * 
119:      * @param bool|int $forceString Specify one of the following:
120:      *  - `true` (bool): to only return a string, where multiple-fields will be split by pipe "|". (default)
121:      *  - `false` (bool): to return string if 1 field, or array of multiple fields (same behavior as field property).
122:      *  - `1` (int): to return only the first value (string).
123:      * @return string|array|null
124:      * @since 3.0.42 Prior versions only supported the 'field' property. 
125:      * @see Selector::fields()
126:      * 
127:      */
128:     public function field($forceString = true) {
129:         $field = parent::get('field');
130:         if($forceString && is_array($field)) {
131:             if($forceString === 1) {
132:                 $field = reset($field);
133:             } else {
134:                 $field = implode('|', $field);
135:             }
136:         } 
137:         return $field;
138:     }
139: 
140:     /**
141:      * Return array of field(s) for this Selector
142:      * 
143:      * @return array
144:      * @see Selector::field()
145:      * @since 3.0.42 Prior versions just supported the 'fields' property. 
146:      * 
147:      */
148:     public function fields() {
149:         $field = parent::get('field');
150:         if(is_array($field)) return $field;
151:         if(!strlen($field)) return array();
152:         return array($field); 
153:     }
154: 
155:     /**
156:      * Get the value(s) of this Selector
157:      *
158:      * Note that if calling this as a property (rather than a method) it can return either a string or an array.
159:      *
160:      * @param bool|int $forceString Specify one of the following:
161:      *  - `true` (bool): to only return a string, where multiple-values will be split by pipe "|". (default)
162:      *  - `false` (bool): to return string if 1 value, or array of multiple values (same behavior as value property).
163:      *  - `1` (int): to return only the first value (string).
164:      * @return string|array|null
165:      * @since 3.0.42 Prior versions only supported the 'value' property.
166:      * @see Selector::values()
167:      *
168:      */
169:     public function value($forceString = true) {
170:         $value = parent::get('value');
171:         if($forceString && is_array($value)) {
172:             if($forceString === 1) {
173:                 $value = reset($value);
174:             } else {
175:                 $value = implode('|', $value);
176:             }
177:         }
178:         return $value;
179:     }
180: 
181:     /**
182:      * Return array of value(s) for this Selector
183:      *
184:      * @param bool $nonEmpty If empty array will be returned, forces it to return array with one blank item instead (default=false). 
185:      * @return array
186:      * @see Selector::value()
187:      * @since 3.0.42 Prior versions just supported the 'values' property. 
188:      *
189:      */
190:     public function values($nonEmpty = false) {
191:         $values = parent::get('value');
192:         if(is_array($values)) {
193:             // ok
194:         } else if(is_string($values)) {
195:             $values = strlen($values) ? array($values) : array();
196:         } else if(is_object($values)) {
197:             $values = $values instanceof WireArray ? $values->getArray() : array($values);
198:         } else if($values) {
199:             $values = array($values);
200:         } else {
201:             $values = array();
202:         }
203:         if($nonEmpty && !count($values)) $values = array('');
204:         return $values; 
205:     }
206: 
207:     /**
208:      * Get a property 
209:      * 
210:      * @param string $key Property name
211:      * @return array|mixed|null|string Property value
212:      * 
213:      */
214:     public function get($key) {
215:         if($key == 'operator') return $this->operator();
216:         if($key == 'str') return $this->__toString();
217:         if($key == 'values') return $this->values();
218:         if($key == 'fields') return $this->fields();
219:         return parent::get($key); 
220:     }
221: 
222:     /**
223:      * Returns the selector field(s), optionally forcing as string or array
224:      * 
225:      * #pw-internal
226:      * 
227:      * @param string $type Omit for automatic, or specify 'string' or 'array' to force return in that type
228:      * @return string|array
229:      * @throws WireException if given invalid type
230:      * 
231:      */
232:     public function getField($type = '') {
233:         $field = $this->field;
234:         if($type == 'string') {
235:             if(is_array($field)) $field = implode('|', $field);
236:         } else if($type == 'array') {
237:             if(!is_array($field)) $field = array($field);
238:         } else if($type) {
239:             throw new WireException("Unknown type '$type' specified to getField()");
240:         }
241:         return $field;
242:     }
243: 
244:     /**
245:      * Returns the selector value(s) with additional processing and forced type options
246:      * 
247:      * When the $type argument is not specified, this method may return a string, array or Selectors object. 
248:      * A Selectors object is only returned if the value happens to contain an embedded selector. 
249:      * 
250:      * #pw-internal
251:      * 
252:      * @param string $type Omit for automatic, or specify 'string' or 'array' to force return in that type
253:      * @return string|array|Selectors
254:      * @throws WireException if given invalid type
255:      * 
256:      */
257:     public function getValue($type = '') {
258:         $value = $this->value; 
259:         if($type == 'string') {
260:             if(is_array($value)) $value = implode('|', $value);
261:         } else if($type == 'array') {
262:             if(!is_array($value)) $value = array($value);
263:         } else if($this->quote == '[') {
264:             if(is_string($value) && Selectors::stringHasSelector($value)) {
265:                 $value = $this->wire(new Selectors($value));
266:             } else if($value instanceof Selectors) {
267:                 // okay
268:             }
269:         } else if($type) {
270:             throw new WireException("Unknown type '$type' specified to getValue()");
271:         }
272:         return $value;
273:     }
274: 
275:     /**
276:      * Set a property of the Selector
277:      * 
278:      * @param string $key
279:      * @param mixed $value
280:      * @return Selector|WireData
281:      * 
282:      */
283:     public function set($key, $value) {
284:         if($key == 'fields') return parent::set('field', $value);
285:         if($key == 'values') return parent::set('value', $value); 
286:         if($key == 'operator') {
287:             $this->error("You cannot set the operator on a Selector: $this");
288:             return $this;
289:         }
290:         return parent::set($key, $value); 
291:     }
292: 
293:     /**
294:      * Return the operator used by this Selector
295:      *
296:      * Strict standards don't let us make static abstract methods, so this one throws an exception if it's not reimplemented.
297:      * 
298:      * #pw-internal
299:      *
300:      * @return string
301:      * @throws WireException
302:      *
303:      */
304:     public static function getOperator() {
305:         throw new WireException("This getOperator method must be implemented"); 
306:     }

308:     /**
309:      * Does $value1 match $value2?
310:      *
311:      * @param mixed $value1 Dynamic comparison value
312:      * @param string $value2 User-supplied value to compare against
313:      * @return bool
314:      *
315:      */
316:     abstract protected function match($value1, $value2);
317: 
318:     /**
319:      * Does this Selector match the given value?
320:      *
321:      * If the value held by this Selector is an array of values, it will check if any one of them matches the value supplied here. 
322:      *
323:      * @param string|int|Wire $value If given a Wire, then matches will also operate on OR field=value type selectors, where present
324:      * @return bool
325:      *
326:      */
327:     public function matches($value) {
328: 
329:         $forceMatch = $this->get('forceMatch');
330:         if(is_bool($forceMatch)) return $forceMatch;
331:         
332:         $matches = false;
333:         $values1 = is_array($this->value) ? $this->value : array($this->value); 
334:         $field = $this->field; 
335:         $operator = $this->operator();
336: 
337:         // prepare the value we are comparing
338:         if(is_object($value)) {
339:             if($this->wire('languages') && $value instanceof LanguagesValueInterface) $value = (string) $value; 
340:                 else if($value instanceof WireData) $value = $value->get($field);
341:                 else if($value instanceof WireArray && is_string($field) && !strpos($field, '.')) $value = (string) $value; // 123|456|789, etc.
342:                 else if($value instanceof Wire) $value = $value->$field; 
343:             $value = (string) $value; 
344:         }
345: 
346:         if(is_string($value) && strpos($value, '|') !== false) $value = explode('|', $value); 
347:         if(!is_array($value)) $value = array($value);
348:         $values2 = $value; 
349:         unset($value);
350: 
351:         // now we're just dealing with 2 arrays: $values1 and $values2
352:         // $values1 is the value stored by the selector
353:         // $values2 is the value passed into the matches() function
