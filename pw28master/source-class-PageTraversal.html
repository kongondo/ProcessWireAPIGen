 16:     /**
 17:      * Return number of children, optionally with conditions
 18:      *
 19:      * Use this over $page->numChildren property when you want to specify a selector or when you want the result to
 20:      * include only visible children. See the options for the $selector argument. 
 21:      *
 22:      * @param Page $page
 23:      * @param bool|string|int|array $selector 
 24:      *  When not specified, result includes all children without conditions, same as $page->numChildren property.
 25:      *  When a string or array, a selector is assumed and quantity will be counted based on selector.
 26:      *  When boolean true, number includes only visible children (excludes unpublished, hidden, no-access, etc.)
 27:      *  When boolean false, number includes all children without conditions, including unpublished, hidden, no-access, etc.
 28:      *  When integer 1 number includes viewable children (as opposed to visible, viewable includes hidden pages + it also includes unpublished pages if user has page-edit permission).
 29:      * @return int Number of children
 30:      *
 31:      */
 32:     public function numChildren(Page $page, $selector = null) {
 33:         if(is_bool($selector)) {
 34:             // onlyVisible takes the place of selector
 35:             $onlyVisible = $selector; 
 36:             if(!$onlyVisible) return $page->get('numChildren');
 37:             return $page->_pages('count', "parent_id=$page->id"); 
 38:             
 39:         } else if($selector === 1) { 
 40:             // viewable pages only
 41:             $numChildren = $page->get('numChildren');
 42:             if(!$numChildren) return 0;
 43:             if($page->wire('user')->isSuperuser()) return $numChildren;
 44:             if($page->wire('user')->hasPermission('page-edit')) {
 45:                 return $page->_pages('count', "parent_id=$page->id, include=unpublished");
 46:             }
 47:             return $page->_pages('count', "parent_id=$page->id, include=hidden"); 
 48: 
 49:         } else if(empty($selector) || (!is_string($selector) && !is_array($selector))) {
 50:             return $page->get('numChildren'); 
 51: 
 52:         } else {
 53:             if(is_string($selector)) {
 54:                 $selector = "parent_id=$page->id, $selector";
 55:             } else if(is_array($selector)) {
 56:                 $selector["parent_id"] = $page->id;
 57:             }
 58:             return $page->_pages('count', $selector);
 59:         }
 60:     }
 61: 
 62:     /**
 63:      * Return this page's children pages, optionally filtered by a selector
 64:      *
 65:      * @param Page $page
 66:      * @param string|array $selector Selector to use, or blank to return all children
 67:      * @param array $options
 68:      * @return PageArray|array
 69:      *
 70:      */
 71:     public function children(Page $page, $selector = '', $options = array()) {
 72:         if(!$page->numChildren) return $page->_pages()->newPageArray();
 73:         $defaults = array('caller' => 'page.children'); 
 74:         $options = array_merge($defaults, $options); 
 75:         $sortfield = $page->sortfield();
 76:         if(is_array($selector)) {
 77:             // selector is array
 78:             $selector["parent_id"] = $page->id;
 79:             if(isset($selector["sort"])) $sortfield = '';
 80:             if($sortfield) $selector[] = array("sort", $sortfield);
 81:         } else {
 82:             // selector is string
 83:             $selector = trim("parent_id=$page->id, $selector", ", ");
 84:             if(strpos($selector, 'sort=') === false) $selector .= ", sort=$sortfield";
 85:         }
 86:         return $page->_pages('find', $selector, $options); 
 87:     }
 88: 
 89:     /**
 90:      * Return the page's first single child that matches the given selector. 
 91:      *
 92:      * Same as children() but returns a Page object or NullPage (with id=0) rather than a PageArray
 93:      *
 94:      * @param Page $page
 95:      * @param string|array $selector Selector to use, or blank to return the first child. 
 96:      * @param array $options
 97:      * @return Page|NullPage
 98:      *
 99:      */
100:     public function child(Page $page, $selector = '', $options = array()) {
101:         if(!$page->numChildren) return $page->_pages()->newNullPage();
102:         $defaults = array('getTotal' => false, 'caller' => 'page.child'); 
103:         $options = array_merge($defaults, $options); 
104:         if(is_array($selector)) {
105:             $selector["limit"] = 1;
106:             $selector[] = array("start", "0");
107:         } else {
108:             $selector .= ($selector ? ', ' : '') . "limit=1";
109:             if(strpos($selector, 'start=') === false) $selector .= ", start=0"; // prevent pagination
110:         }
111:         $children = $this->children($page, $selector, $options); 
112:         return count($children) ? $children->first() : $page->_pages()->newNullPage();
113:     }
114: 
115:     /**
116:      * Return this page's parent pages, or the parent pages matching the given selector.
117:      *
118:      * @param Page $page
119:      * @param string|array $selector Optional selector string to filter parents by
120:      * @return PageArray
121:      *
122:      */
123:     public function parents(Page $page, $selector = '') {
124:         $parents = $page->wire('pages')->newPageArray();
125:         $parent = $page->parent();

279:     /**
280:      * Provides the core logic for next, prev, nextAll, prevAll, nextUntil, prevUntil
281:      *
282:      * @param Page $page
283:      * @param string|array $selector Optional selector. When specified, will find nearest sibling(s) that match.
284:      * @param array $options Options to modify behavior
285:      *  - `prev` (bool): When true, previous siblings will be returned rather than next siblings.
286:      *  - `all` (bool): If true, returns all nextAll or prevAll rather than just single sibling (default=false).
287:      *  - `until` (string): If specified, returns siblings until another is found matching the given selector (default=false).
288:      *  - `qty` (bool): If true, makes it return just the quantity that would match (default=false). 
289:      * @return Page|NullPage|PageArray|int Returns one of the following: 
290:      *  - `PageArray` if the "all" or "until" option is specified. 
291:      *  - `Page|NullPage` in other cases. 
292:      *
293:      */ 
294:     protected function _next(Page $page, $selector = '', array $options = array()) {
295:         
296:         $defaults = array(
297:             'prev' => false, // get previous rather than next
298:             'all' => false, // get multiple/all
299:             'until' => '', // until selector string ('all' option assumed)
300:             'qty' => false, // when true, returns just the quantity that would match ('all' option assumed)
301:         );
302: 
303:         $options = array_merge($defaults, $options);
304:         $pages = $page->wire('pages');
305:         $parent = $page->parent();
306:         
307:         if($options['until'] || $options['qty']) $options['all'] = true;
308:         if(!$parent || !$parent->id) return $options['all'] ? $pages->newPageArray() : $pages->newNullPage();
309:         
310:         if(is_array($selector)) {
311:             $selector['parent_id'] = $parent->id; 
312:         } else {
313:             $selector = trim("parent_id=$parent->id, $selector", ", ");
314:         }
315:         
316:         $pageFinder = $pages->getPageFinder();
317:         $pageFinderOptions = $this->_nextFinderOptions($page, $selector, $options);
318:         $rows = $pageFinder->find($selector, $pageFinderOptions);
319:         
320:         if($options['qty']) {
321:             $result = count($rows);
322: 
323:         } else if(!count($rows)) {
324:             $result = $options['all'] ? $pages->newPageArray() : $pages->newNullPage();
325: 
326:         } else if($options['all']) {
327:             $result = $pages->getById($rows, array(
328:                 'parent_id' => $parent->id,
329:                 'cache' => $page->loaderCache
330:             ));
331:             if($options['all'] && $options['prev']) $result = $result->reverse();
332:             
333:         } else {
334:             $row = reset($rows);
335:             $result = $pages->getById(array($row['id']), array(
336:                 'template' => $page->wire('templates')->get($row['templates_id']),
337:                 'parent_id' => $row['parent_id'],
338:                 'getOne' => true,
339:                 'cache' => $page->loaderCache
340:             ));
341:         }
342:         
343:         return $result;
344:     }

455:     /**
456:      * Returns the URL to the page with $options
457:      *
458:      * You can specify an `$options` argument to this method with any of the following:
459:      *
460:      * - `pageNum` (int|string): Specify pagination number, or "+" for next pagination, or "-" for previous pagination.
461:      * - `urlSegmentStr` (string): Specify a URL segment string to append.
462:      * - `urlSegments` (array): Specify array of URL segments to append (may be used instead of urlSegmentStr).
463:      * - `data` (array): Array of key=value variables to form a query string.
464:      * - `http` (bool): Specify true to make URL include scheme and hostname (default=false).
465:      * - `language` (Language): Specify Language object to return URL in that Language.
466:      *
467:      * You can also specify any of the following for `$options` as shortcuts:
468:      *
469:      * - If you specify an `int` for options it is assumed to be the `pageNum` option.
470:      * - If you specify `+` or `-` for options it is assumed to be the `pageNum` next/previous pagination option.
471:      * - If you specify any other `string` for options it is assumed to be the `urlSegmentStr` option.
472:      * - If you specify a `boolean` (true) for options it is assumed to be the `http` option.
473:      *
474:      * Please also note regarding `$options`:
475:      *
476:      * - This method honors template slash settings for page, URL segments and page numbers.
477:      * - Any passed in URL segments are automatically sanitized with `Sanitizer::pageNameUTF8()`.
478:      * - If using the `pageNum` or URL segment options please also make sure these are enabled on the pages template.
479:      * - The query string generated by any `data` variables is entity encoded when output formatting is on.
480:      * - The `language` option requires that the `LanguageSupportPageNames` module is installed.
481:      * - The prefix for page numbers honors `$config->pageNumUrlPrefix` and multi-language prefixes as well.
482:      *
483:      * @param Page $page
484:      * @param array|int|string|bool|Language $options Optionally specify options to modify default behavior (see method description).
485:      * @return string Returns page URL, for example: `/my-site/about/contact/`
486:      * @see Page::path(), Page::httpUrl(), Page::editUrl(), Page::localUrl()
487:      *
488:      */
489:     public function urlOptions(Page $page, $options = array()) {
490: 
491:         $config = $page->wire('config');
492:         $template = $page->template;
493: 
494:         $defaults = array(
495:             'http' => is_bool($options) ? $options : false,
496:             'pageNum' => is_int($options) || (is_string($options) && in_array($options, array('+', '-'))) ? $options : 1,
497:             'data' => array(),
498:             'urlSegmentStr' => is_string($options) ? $options : '',
499:             'urlSegments' => array(),
500:             'language' => is_object($options) && $options instanceof Page && $options->className() === 'Language' ? $options : null,
501:         );
502: 
503:         if(empty($options)) {
504:             $url = rtrim($config->urls->root, '/') . $page->path();
505:             if($template->slashUrls === 0 && $page->id > 1) $url = rtrim($url, '/');
506:             return $url;
507:         }
508: 
509:         $options = is_array($options) ? array_merge($defaults, $options) : $defaults;
510:         $sanitizer = $page->wire('sanitizer');
511:         $language = null;
512:         $url = null;
513: 
514:         if(count($options['urlSegments'])) {
515:             $options['urlSegmentStr'] = implode('/', $options['urlSegments']);
516:         }
517: 
518:         if($options['language'] && $page->wire('modules')->isInstalled('LanguageSupportPageNames')) {
519:             if(!is_object($options['language'])) {
520:                 $options['language'] = null;
521:             } else if(!$options['language'] instanceof Page) {
522:                 $options['language'] = null;
523:             } else if(strpos($options['language']->className(), 'Language') === false) {
524:                 $options['language'] = null;
525:             }
526:             if($options['language']) {
527:                 /** @var Language $language */
528:                 $language = $options['language'];
529:                 // localUrl method provided as hook by LanguageSupportPageNames
530:                 $url = $page->localUrl($language);
531:             }
532:         }
533: 
534:         if(is_null($url)) {
535:             $url = rtrim($config->urls->root, '/') . $page->path();
536:             if($template->slashUrls === 0 && $page->id > 1) $url = rtrim($url, '/');
537:         }
538: 
539:         if(is_string($options['urlSegmentStr']) && strlen($options['urlSegmentStr'])) {
540:             $url = rtrim($url, '/') . '/' . $sanitizer->pagePathNameUTF8(trim($options['urlSegmentStr'], '/'));
541:             if($template->slashUrlSegments === '' || $template->slashUrlSegments) $url .= '/';
542:         }
543: 
544:         if($options['pageNum']) {
545:             if($options['pageNum'] === '+') {
546:                 $options['pageNum'] = $page->wire('input')->pageNum + 1;
547:             } else if($options['pageNum'] === '-' || $options['pageNum'] === -1) {
548:                 $options['pageNum'] = $page->wire('input')->pageNum - 1;
549:             }
550:             if((int) $options['pageNum'] > 1) {
551:                 $prefix = '';
552:                 if($language) {
553:                     $lsp = $page->wire('modules')->get('LanguageSupportPageNames');
554:                     $prefix = $lsp ? $lsp->get("pageNumUrlPrefix$language") : '';
555:                 }
556:                 if(!strlen($prefix)) $prefix = $config->pageNumUrlPrefix;

559:             }
560:         }
561: 
562:         if(count($options['data'])) {

571:                 case 1: $scheme = 'https'; break;
572:                 default: $scheme = $config->https ? 'https' : 'http';
573:             }
574:             $url = "$scheme://" . $page->wire('config')->httpHost . $url;
575:         }
576: 
577:         return $url;
578:     }
