403:     public function validateFilename($value, $extensions = array()) {
404:         $value = basename($value);
405:         if($value[0] == '.') return false; // no hidden files
406:         if($this->lowercase) $value = function_exists('mb_strtolower') ? mb_strtolower($value) : strtolower($value);
407:         $value = $this->wire('sanitizer')->filename($value, Sanitizer::translate); 
408:         $value = trim($value, "_");
409: 
410:         $p = pathinfo($value);
411:         if(!isset($p['extension'])) return false;
412:         $extension = strtolower($p['extension']);
413:         $basename = basename($p['basename'], ".$extension"); 
414:         // replace any dots in the basename with underscores
415:         $basename = trim(str_replace(".", "_", $basename), "_");

432:      * 
433:      */

435: 
436:         if(!$this->checkDestinationPath()) return false; 
437:         $filename = $this->getTargetFilename($filename); 
438:         $filename = $this->validateFilename($filename);
439:         if($this->lowercase) $filename = strtolower($filename); 
440:         $destination = $this->destinationPath . $filename;
441:         $p = pathinfo($destination); 
442:         $exists = file_exists($destination);

444:         if(!$this->overwrite && $filename != $this->overwriteFilename) {
445:             // overwrite not allowed, so find a new name for it
446:             $destination = $this->getUniqueFilename($destination); 
447:             $filename = basename($destination); 
448:             
449:         } else if($exists && $this->overwrite) {
450:             // file already exists in destination and will be overwritten
451:             // here we back it up temporarily, and we don't remove the backup till __destruct()
452:             $bakName = $filename;

458:             $this->overwrittenFiles[$bakDestination] = $destination;
459:         }
460: 
461:         if($ajax) $success = @rename($tmp_name, $destination);
462:             else $success = move_uploaded_file($tmp_name, $destination);
463: 
464:         if(!$success) {
465:             $this->error("Unable to move uploaded file to: $destination");
466:             if(is_file($tmp_name)) @unlink($tmp_name); 
467:             return false;
468:         }
