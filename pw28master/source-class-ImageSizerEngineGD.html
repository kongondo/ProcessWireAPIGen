 55:         // first we check parts that are mandatory for all $actions
 56:         if(!function_exists('gd_info')) return false;
 57:         // and if it passes the mandatory requirements, we check particularly aspects here
 58:         
 59:         switch($action) {
 60:             case 'imageformat':
 61:                 // compare current imagefile infos fetched from ImageInspector
 62:                 $requested = $this->getImageInfo(false);
 63:                 switch($requested) {
 64:                     case 'gif-anim':
 65:                     case 'gif-trans-anim':
 66:                         // Animated GIF images are not supported, but GD renders the first image of the animation
 67:                         #return false;
 68:                     default:
 69:                         return true;
 70:                 }
 71:                 break;
 72: 
 73:             case 'install':
 74:                 /*
 75:                 $gd  = gd_info();
 76:                 $jpg = isset($gd['JPEG Support']) ? $gd['JPEG Support'] : false;
 77:                 $png = isset($gd['PNG Support']) ? $gd['PNG Support'] : false;
 78:                 $gif = isset($gd['GIF Read Support']) && isset($gd['GIF Create Support']) ? $gd['GIF Create Support'] : false;
 79:                 $freetype = isset($gd['FreeType Support']) ? $gd['FreeType Support'] : false;
 80:                 $this->config->gdReady = true;
 81:                 */
 82:                 return true;
 83: 
 84:             default:
 85:                 return false;
 86:         }

 89:     /**

100:      *

103:         
104:         $this->modified = false;
105:         if(isset($this->info['bits'])) $this->imageDepth = $this->info['bits'];
106:         $this->imageFormat = strtoupper(str_replace('image/', '', $this->info['mime']));
107: 
108:         if(!in_array($this->imageFormat, $this->validSourceImageFormats())) {
109:             throw new WireException(sprintf($this->_("loaded file '%s' is not in the list of valid images", basename($dstFilename))));
110:         }
111: 
112:         $image = null;
113:         $orientations = null; // @horst
114:         $needRotation = $this->autoRotation !== true ? false : ($this->checkOrientation($orientations) &&
115:         (!empty($orientations[0]) || !empty($orientations[1])) ? true : false);
116: 
117:         // check if we can load the sourceimage into ram
118:         if(self::checkMemoryForImage(array($this->info['width'], $this->info['height'], $this->info['channels'])) === false) {
119:             throw new WireException(basename($srcFilename) . " - not enough memory to load");
120:         }
121: 
122:         switch($this->imageType) { // @teppo
123:             case \IMAGETYPE_GIF:
124:                 $image = @imagecreatefromgif($srcFilename);
125:                 break;
126:             case \IMAGETYPE_PNG:
127:                 $image = @imagecreatefrompng($srcFilename);
128:                 break;
129:             case \IMAGETYPE_JPEG:
130:                 $image = @imagecreatefromjpeg($srcFilename);
131:                 break;
132:         }
133: 
134:         if(!$image) return false;
135: 
136:         if($this->imageType != \IMAGETYPE_PNG || !$this->hasAlphaChannel()) {
137:             // @horst: linearize gamma to 1.0 - we do not use gamma correction with pngs containing alphachannel, because GD-lib doesn't respect transparency here (is buggy)
138:             $this->gammaCorrection($image, true);
139:         }
140: 
141:         if($this->rotate || $needRotation) { // @horst
142:             $degrees = $this->rotate ? $this->rotate : $orientations[0];
143:             $image = $this->imRotate($image, $degrees);
144:             if(abs($degrees) == 90 || abs($degrees) == 270) {
145:                 // we have to swap width & height now!
146:                 $tmp = array($this->getWidth(), $this->getHeight());
147:                 $this->setImageInfo($tmp[1], $tmp[0]);
148:             }
149:         }
150: 
151:         if($this->flip || $needRotation) {
152:             $vertical = null;
153:             if($this->flip) {
154:                 $vertical = $this->flip == 'v';
155:             } else if($orientations[1] > 0) {
156:                 $vertical = $orientations[1] == 2;
157:             }
158:             if(!is_null($vertical)) $image = $this->imFlip($image, $vertical);
159:         }
160: 
161:         // if there is requested to crop _before_ resize, we do it here @horst
162:         if(is_array($this->cropExtra)) {
163:             // check if we can load a second copy from sourceimage into ram
164:             if(self::checkMemoryForImage(array($this->info['width'], $this->info['height'], 3)) === false) {
165:                 throw new WireException(basename($srcFilename) . " - not enough memory to load a copy for cropExtra");
166:             }
167: 
168:             $imageTemp = imagecreatetruecolor(imagesx($image), imagesy($image));  // create an intermediate memory image
169:             $this->prepareImageLayer($imageTemp, $image);
170:             imagecopy($imageTemp, $image, 0, 0, 0, 0, imagesx($image), imagesy($image)); // copy our initial image into the intermediate one
171:             imagedestroy($image); // release the initial image
172: 
173:             // get crop values and create a new initial image
174:             list($x, $y, $w, $h) = $this->cropExtra;
175: 
176:             // check if we can load a cropped version into ram
177:             if(self::checkMemoryForImage(array($w, $h, 3)) === false) {
178:                 throw new WireException(basename($srcFilename) . " - not enough memory to load a cropped version for cropExtra");
179:             }

182:             $this->prepareImageLayer($image, $imageTemp);
183:             imagecopy($image, $imageTemp, 0, 0, $x, $y, $w, $h);
184:             unset($x, $y, $w, $h);
185: 
186:             // now release the intermediate image and update settings
187:             imagedestroy($imageTemp);

189:             // $this->cropping = false; // ?? set this to prevent overhead with the following manipulation ??
190:         }

192:         // here we check for cropping, upscaling, sharpening
193:         // we get all dimensions at first, before any image operation !

195:         $bgWidth = $fullWidth;
196:         $bgHeight = $fullHeight;

198:         $thumb = null;

204: 
205:             // the current version is allready the desired result, we only may have to apply compression where possible
206:             $this->sharpening = 'none'; // we set sharpening to none
207: 
208:             if(self::checkMemoryForImage(array(imagesx($image), imagesy($image), 3)) === false) {
209:                 throw new WireException(basename($srcFilename) . " - not enough memory to copy the final image");
210:             }
211:             $thumb = imagecreatetruecolor(imagesx($image), imagesy($image));          // create the final memory image
212:             $this->prepareImageLayer($thumb, $image);
213:             imagecopy($thumb, $image, 0, 0, 0, 0, imagesx($image), imagesy($image));  // copy our intermediate image into the final one
214: 
215:         } else if(2 == $resizeMethod) { // 2 = resize with aspect ratio
216: 
217:             // this is the case if we scale up or down _without_ cropping
218: 
219:             if(self::checkMemoryForImage(array($finalWidth, $finalHeight, 3)) === false) {
220:                 throw new WireException(basename($srcFilename) . " - not enough memory to resize to the final image");
221:             }
222: 
223:             $thumb = imagecreatetruecolor($finalWidth, $finalHeight);
224:             $this->prepareImageLayer($thumb, $image);
225:             imagecopyresampled($thumb, $image, 0, 0, 0, 0, $finalWidth, $finalHeight, $this->image['width'], $this->image['height']);

227:         } else if(4 == $resizeMethod) { // 4 = resize and crop with aspect ratio

232:                 throw new WireException(basename($srcFilename) . " - not enough memory to resize to the intermediate image");
233:             }
234: 
235:             $thumb2 = imagecreatetruecolor($bgWidth, $bgHeight);
236:             $this->prepareImageLayer($thumb2, $image);
237:             imagecopyresampled($thumb2, $image, 0, 0, 0, 0, $bgWidth, $bgHeight, $this->image['width'], $this->image['height']);
238: 
239:             if(self::checkMemoryForImage(array($finalWidth, $finalHeight, 3)) === false) {
240:                 throw new WireException(basename($srcFilename) . " - not enough memory to crop to the final image");
241:             }
242: 
243:             $thumb = imagecreatetruecolor($finalWidth, $finalHeight);
244:             $this->prepareImageLayer($thumb, $image);
245:             imagecopyresampled($thumb, $thumb2, 0, 0, $bgX, $bgY, $finalWidth, $finalHeight, $finalWidth, $finalHeight);
246:             imagedestroy($thumb2);
247:         }
248: 
249:         // early release of obsolete GD image object(s) to free memory before processing sharpening
250:         if(isset($image) && is_resource($image)) @imagedestroy($image); // @horst
251:         if(isset($thumb2) && is_resource($thumb2)) @imagedestroy($thumb2);
252:         if(isset($image)) $image = null;
253:         if(isset($thumb2)) $thumb2 = null;
254: 
255:         // optionally apply sharpening to the final thumb
256:         if($this->sharpening && $this->sharpening != 'none') { // @horst
257:             if(\IMAGETYPE_PNG != $this->imageType || !$this->hasAlphaChannel()) {
258:                 $w = imagesx($thumb);
259:                 $h = imagesy($thumb);
260:                 if($this->useUSM) {
261:                     // calculate if there is enough memory available to apply the USM algorithm, if enabled
262:                     if(true === ($this->useUSM = self::checkMemoryForImage(array($w, $h, 3), array($w, $h, 3)))) {
263:                         // is needed for the USM sharpening function to calculate the best sharpening params
264:                         $this->usmValue = $this->calculateUSMfactor($finalWidth, $finalHeight);
265:                         $thumb = $this->imSharpen($thumb, $this->sharpening);
266:                     }
267:                 }
268:                 if(!$this->useUSM) {
269:                     if(false !== self::checkMemoryForImage(array($w, $h, 3))) {
270:                         $thumb = $this->imSharpen($thumb, $this->sharpening);
271:                     }
272:                 }
273:             }
274:         }
275: 
276:         // write to file
277:         $result = false;
278:         switch($this->imageType) {
279:             case \IMAGETYPE_GIF:
280:                 // correct gamma from linearized 1.0 back to 2.0
281:                 $this->gammaCorrection($thumb, false);
282:                 $result = imagegif($thumb, $dstFilename);
283:                 break;
284:             case \IMAGETYPE_PNG:
285:                 if(!$this->hasAlphaChannel()) $this->gammaCorrection($thumb, false);
286:                 // always use highest compression level for PNG (9) per @horst
287:                 $result = imagepng($thumb, $dstFilename, 9);
288:                 break;
289:             case \IMAGETYPE_JPEG:
290:                 // correct gamma from linearized 1.0 back to 2.0
291:                 $this->gammaCorrection($thumb, false);
292:                 $result = imagejpeg($thumb, $dstFilename, $this->quality);
293:                 break;
294:         }
295: 
296:         // release the last GD image object
297:         if(isset($thumb) && is_resource($thumb)) @imagedestroy($thumb);
298:         if(isset($thumb)) $thumb = null;
299: 
300:         return $result;
301:     }

340:     /**
341:      * Sharpen image (@horst)
342:      *
343:      * @param resource $im
344:      * @param string $mode May be: none | soft | medium | strong
345:      *
346:      * @return resource|bool
347:      *
348:      */
349:     protected function imSharpen($im, $mode) {
350: 
351:         // due to a bug in PHP's bundled GD-Lib with the function imageconvolution in some PHP versions
352:         // we have to bypass this for those who have to run on this PHP versions
353:         // see: https://bugs.php.net/bug.php?id=66714
354:         // and here under GD: http://php.net/ChangeLog-5.php#5.5.11
355:         $buggyPHP = (version_compare(phpversion(), '5.5.8', '>') && version_compare(phpversion(), '5.5.11', '<')) ? true : false;
356:         if($buggyPHP && !$this->useUSM
357:             && self::checkMemoryForImage(array(imagesx($im), imagesy($im), 3), array(imagesx($im), imagesy($im), 3)) !== true
358:         ) {
359:             // we have not enough memory available for USM and cannot use the other algorithm because of the buggy PHP version
360:             return $im;
361:         }
362: 
363:         // USM method is used for buggy PHP versions
364:         // for regular versions it can be omitted per: useUSM = false passes as pageimage option
365:         // or set in the site/config.php under $config->imageSizerOptions: 'useUSM' => false | true
366:         if($buggyPHP || $this->useUSM) {
367: 
368:             switch($mode) {
369: 
370:                 case 'none':
371:                     return $im;
372:                     break;
373: 
374:                 case 'strong':
375:                     $amount = 160;
376:                     $radius = 1.0;
377:                     $threshold = 7;
378:                     break;
379: 
380:                 case 'medium':
381:                     $amount = 130;
382:                     $radius = 0.75;
383:                     $threshold = 7;
384:                     break;
385: 
386:                 case 'soft':
387: 
388:                 default:
389:                     $amount = 100;
390:                     $radius = 0.5;
391:                     $threshold = 7;
392:             }
393: 
394:             // calculate the final amount according to the usmValue
395:             $this->usmValue = $this->usmValue < 0 ? 0 : ($this->usmValue > 100 ? 100 : $this->usmValue);
396:             if(0 == $this->usmValue) return $im;
397:             $amount = intval($amount / 100 * $this->usmValue);
398: 
399:             // apply unsharp mask filter
400:             return $this->UnsharpMask($im, $amount, $radius, $threshold);
401:         }
402: 
403:         // if we do not use USM, we use our default sharpening method,
404:         // entirely based on GDs imageconvolution
405:         switch($mode) {
406: 
407:             case 'none':
408:                 return $im;
409:                 break;
410: 
411:             case 'strong':
412:                 $sharpenMatrix = array(
413:                     array(-1.2, -1, -1.2),
414:                     array(-1, 16, -1),
415:                     array(-1.2, -1, -1.2)
416:                 );
417:                 break;
418: 
419:             case 'medium':
420:                 $sharpenMatrix = array(
421:                     array(-1.1, -1, -1.1),
422:                     array(-1, 20, -1),
423:                     array(-1.1, -1, -1.1)
424:                 );
425:                 break;
426: 
427:             case 'soft':
428: 
429:             default:
430:                 $sharpenMatrix = array(
431:                     array(-1, -1, -1),
432:                     array(-1, 24, -1),
433:                     array(-1, -1, -1)
434:                 );
435:         }
436: 
437:         // calculate the sharpen divisor
438:         $divisor = array_sum(array_map('array_sum', $sharpenMatrix));
439:         $offset = 0;
440: 
441:         // TODO 4 -c errorhandling: Throw WireException?
442:         if(!imageconvolution($im, $sharpenMatrix, $divisor, $offset)) return false;
443: 
444:         return $im;
445:     }

450:      *
451:      * with mode = true it linearizes an image to 1

453:      *
454:      * @param GD -image-resource $image

470:     /**
471:      * Unsharp Mask for PHP - version 2.1.1
472:      *
473:      * Unsharp mask algorithm by Torstein Hønsi 2003-07.
474:      * thoensi_at_netcom_dot_no.
475:      * Please leave this notice.
476:      *
477:      * http://vikjavev.no/computing/ump.php
478:      * 
479:      * @param resource $img
480:      * @param int $amount
481:      * @param int $radius
482:      * @param int $threshold
483:      * @return resource
484:      *
485:      */
486:     protected function unsharpMask($img, $amount, $radius, $threshold) {
487:         // Attempt to calibrate the parameters to Photoshop:
488:         if($amount > 500) $amount = 500;
489:         $amount = $amount * 0.016;
490:         if($radius > 50) $radius = 50;
491:         $radius = $radius * 2;
492:         if($threshold > 255) $threshold = 255;
493: 
494:         $radius = abs(round($radius));     // Only integers make sense.
495:         if($radius == 0) {
496:             return $img;
497:         }
498:         $w = imagesx($img);
499:         $h = imagesy($img);
500:         $imgCanvas = imagecreatetruecolor($w, $h);
501:         $imgBlur = imagecreatetruecolor($w, $h);
502: 
503:         // due to a bug in PHP's bundled GD-Lib with the function imageconvolution in some PHP versions
504:         // we have to bypass this for those who have to run on this PHP versions
505:         // see: https://bugs.php.net/bug.php?id=66714
506:         // and here under GD: http://php.net/ChangeLog-5.php#5.5.11
507:         $buggyPHP = (version_compare(phpversion(), '5.5.8', '>') && version_compare(phpversion(), '5.5.11', '<')) ? true : false;
508: 
509:         // Gaussian blur matrix:
510:         //
511:         //    1    2    1
512:         //    2    4    2
513:         //    1    2    1
514:         //
515:         //////////////////////////////////////////////////
516:         if(function_exists('imageconvolution') && !$buggyPHP) {
517:             $matrix = array(
518:                 array(1, 2, 1),
519:                 array(2, 4, 2),
520:                 array(1, 2, 1)
521:             );
522:             imagecopy($imgBlur, $img, 0, 0, 0, 0, $w, $h);
523:             imageconvolution($imgBlur, $matrix, 16, 0);
524:         } else {
525:             // Move copies of the image around one pixel at the time and merge them with weight
526:             // according to the matrix. The same matrix is simply repeated for higher radii.
527:             for($i = 0; $i < $radius; $i++) {
528:                 imagecopy($imgBlur, $img, 0, 0, 1, 0, $w - 1, $h); // left
529:                 imagecopymerge($imgBlur, $img, 1, 0, 0, 0, $w, $h, 50); // right
530:                 imagecopymerge($imgBlur, $img, 0, 0, 0, 0, $w, $h, 50); // center
531:                 imagecopy($imgCanvas, $imgBlur, 0, 0, 0, 0, $w, $h);
532: 
533:                 imagecopymerge($imgBlur, $imgCanvas, 0, 0, 0, 1, $w, $h - 1, 33.33333); // up
534:                 imagecopymerge($imgBlur, $imgCanvas, 0, 1, 0, 0, $w, $h, 25); // down
535:             }
536:         }
537: 
538:         if($threshold > 0) {
539:             // Calculate the difference between the blurred pixels and the original
540:             // and set the pixels
541:             for($x = 0; $x < $w - 1; $x++) { // each row
542:                 for($y = 0; $y < $h; $y++) { // each pixel
543: 
544:                     $rgbOrig = ImageColorAt($img, $x, $y);
545:                     $rOrig = (($rgbOrig >> 16) & 0xFF);
546:                     $gOrig = (($rgbOrig >> 8) & 0xFF);
547:                     $bOrig = ($rgbOrig & 0xFF);
548: 
549:                     $rgbBlur = ImageColorAt($imgBlur, $x, $y);
550: 
551:                     $rBlur = (($rgbBlur >> 16) & 0xFF);
552:                     $gBlur = (($rgbBlur >> 8) & 0xFF);
553:                     $bBlur = ($rgbBlur & 0xFF);
554: 
555:                     // When the masked pixels differ less from the original
556:                     // than the threshold specifies, they are set to their original value.
557:                     $rNew = (abs($rOrig - $rBlur) >= $threshold)
558:                         ? max(0, min(255, ($amount * ($rOrig - $rBlur)) + $rOrig))
559:                         : $rOrig;
560:                     $gNew = (abs($gOrig - $gBlur) >= $threshold)
561:                         ? max(0, min(255, ($amount * ($gOrig - $gBlur)) + $gOrig))
562:                         : $gOrig;
563:                     $bNew = (abs($bOrig - $bBlur) >= $threshold)
564:                         ? max(0, min(255, ($amount * ($bOrig - $bBlur)) + $bOrig))
565:                         : $bOrig;
566: 
567:                     if(($rOrig != $rNew) || ($gOrig != $gNew) || ($bOrig != $bNew)) {
568:                         $pixCol = ImageColorAllocate($img, $rNew, $gNew, $bNew);
569:                         ImageSetPixel($img, $x, $y, $pixCol);
570:                     }
571:                 }
572:             }
573:         } else {
574:             for($x = 0; $x < $w; $x++) { // each row
575:                 for($y = 0; $y < $h; $y++) { // each pixel
576:                     $rgbOrig = ImageColorAt($img, $x, $y);
577:                     $rOrig = (($rgbOrig >> 16) & 0xFF);
578:                     $gOrig = (($rgbOrig >> 8) & 0xFF);
579:                     $bOrig = ($rgbOrig & 0xFF);
580: 
581:                     $rgbBlur = ImageColorAt($imgBlur, $x, $y);
582: 
583:                     $rBlur = (($rgbBlur >> 16) & 0xFF);
584:                     $gBlur = (($rgbBlur >> 8) & 0xFF);
585:                     $bBlur = ($rgbBlur & 0xFF);
586: 
587:                     $rNew = ($amount * ($rOrig - $rBlur)) + $rOrig;
588:                     if($rNew > 255) {
589:                         $rNew = 255;
590:                     } else if($rNew < 0) {
591:                         $rNew = 0;
592:                     }
593:                     $gNew = ($amount * ($gOrig - $gBlur)) + $gOrig;
594:                     if($gNew > 255) {
595:                         $gNew = 255;
596:                     } else if($gNew < 0) {
597:                         $gNew = 0;
598:                     }
599:                     $bNew = ($amount * ($bOrig - $bBlur)) + $bOrig;
600:                     if($bNew > 255) {
601:                         $bNew = 255;
602:                     } else if($bNew < 0) {
603:                         $bNew = 0;
604:                     }
605:                     $rgbNew = ($rNew << 16) + ($gNew << 8) + $bNew;
606:                     ImageSetPixel($img, $x, $y, $rgbNew);
607:                 }
608:             }
609:         }
610:         imagedestroy($imgCanvas);
611:         imagedestroy($imgBlur);
612: 
613:         return $img;
614:     }

617:     /**
618:      * Calculate USM factor
619:      *
620:      * Return an integer value indicating how much an image should be sharpened
621:      * according to resizing scalevalue and absolute target dimensions
622:      *
623:      * @param mixed $targetWidth width of the targetimage
624:      * @param mixed $targetHeight height of the targetimage
625:      * @param mixed $origWidth
626:      * @param mixed $origHeight
627:      *
628:      * @return int
629:      *
630:      */
631:     protected function calculateUSMfactor($targetWidth, $targetHeight, $origWidth = null, $origHeight = null) {
632: 
633:         if(null === $origWidth) $origWidth = $this->getWidth();
634:         if(null === $origHeight) $origHeight = $this->getHeight();
635:         
636:         $w = ceil($targetWidth / $origWidth * 100);
637:         $h = ceil($targetHeight / $origHeight * 100);
638:         
639:         $resizingScalevalue = null;
640:         $target = null;
641:         $res = null;
642: 
643:         // select the resizing scalevalue with check for crop images
644:         if($w == $h || ($w - 1) == $h || ($w + 1) == $h) {  // equal, no crop
645:             $resizingScalevalue = $w;
646:             $target = $targetWidth;
647:         } else { // crop
648:             if(($w < $h && $w < 100) || ($w > $h && $h >= 100)) {
649:                 $resizingScalevalue = $w;
650:                 $target = $targetWidth;
651:             } elseif(($w < $h && $w >= 100) || ($w > $h && $h < 100)) {
652:                 $resizingScalevalue = $h;
653:                 $target = $targetHeight;
654:             }
655:         }
656: 
657:         // adjusting with respect to the scalefactor
658:         $resizingScalevalue = ($resizingScalevalue * -1) + 100;
659:         $resizingScalevalue = $resizingScalevalue < 0 ? $resizingScalevalue * -1 : $resizingScalevalue;
660: 
661:         if($resizingScalevalue > 0 && $resizingScalevalue < 10) $resizingScalevalue += 15;
662:             else if($resizingScalevalue > 9 && $resizingScalevalue < 25) $resizingScalevalue += 20;
663:             else if($resizingScalevalue > 24 && $resizingScalevalue < 40) $resizingScalevalue += 35;
664:             else if($resizingScalevalue > 39 && $resizingScalevalue < 55) $resizingScalevalue += 20;
665:             else if($resizingScalevalue > 54 && $resizingScalevalue < 70) $resizingScalevalue += 5;
666:             else if($resizingScalevalue > 69 && $resizingScalevalue < 80) $resizingScalevalue -= 10;
667: 
668:         // adjusting with respect to absolute dimensions
669:         if($target < 50) $res = intval($resizingScalevalue / 18 * 3);
670:             else if($target < 100) $res = intval($resizingScalevalue / 18 * 4);
671:             else if($target < 200) $res = intval($resizingScalevalue / 18 * 6);
672:             else if($target < 300) $res = intval($resizingScalevalue / 18 * 8);
673:             else if($target < 400) $res = intval($resizingScalevalue / 18 * 10);
674:             else if($target < 500) $res = intval($resizingScalevalue / 18 * 12);
675:             else if($target < 600) $res = intval($resizingScalevalue / 18 * 15);
676:             else if($target > 599) $res = $resizingScalevalue;
677: 
678:         $res = $res < 0 ? $res * -1 : $res; // avoid negative numbers
679: 
680:         return $res;
681:     }

718:     /**
719:      * calculation if there is enough memory available at runtime for loading and resizing an given imagefile
720:      *
721:      * @param array $sourceDimensions - array with three values: width, height, number of channels
722:      * @param array|bool $targetDimensions - optional - mixed: bool true | false or array with three values:
723:      *  width, height, number of channels
724:      *
725:      * @return bool|null if a calculation was possible (true|false), or null if the calculation could not be done
726:      *
727:      */
728:     static public function checkMemoryForImage($sourceDimensions, $targetDimensions = false) {
729: 
730:         // with this static we only once need to read from php.ini and calculate phpMaxMem,
731:         // regardless how often this function is called in a request
732:         static $phpMaxMem = null;
733: 
734:         if(null === $phpMaxMem) {
735:             $sMem = trim(strtoupper(ini_get('memory_limit')), ' B'); // trim B just in case it has Mb rather than M
736:             switch(substr($sMem, -1)) {
737:                 case 'M':
738:                     $phpMaxMem = ((int) $sMem) * 1048576;
739:                     break;
740:                 case 'K':
741:                     $phpMaxMem = ((int) $sMem) * 1024;
742:                     break;
743:                 case 'G':
744:                     $phpMaxMem = ((int) $sMem) * 1073741824;
745:                     break;
746:                 default:
747:                     $phpMaxMem = (int) $sMem;
748:             }
749:         }
750: 
751:         if($phpMaxMem <= 0) {
752:             // we couldn't read the MaxMemorySetting or there isn't one set,
753:             // so in both cases we do not know if there is enough or not
754:             return null;
755:         }
756: 
757:         // calculate $sourceDimensions
758:         if(!isset($sourceDimensions[0]) || !isset($sourceDimensions[1])
759:             || !isset($sourceDimensions[2]) || !is_int($sourceDimensions[0])
760:             || !is_int($sourceDimensions[1]) || !is_int($sourceDimensions[2])
761:         ) {
762:             return null;
763:         }
764: 
765:         //            width             *        height        *       channels
766:         $imgMem = ($sourceDimensions[0] * $sourceDimensions[1] * $sourceDimensions[2]);
767: 
768:         if(true === $targetDimensions) {
769:             // we have to add ram for a copy of the sourceimage
770:             $imgMem += $imgMem;
771: 
772:         } else if(is_array($targetDimensions)) {
773:             // we have to add ram for a targetimage
774:             if(!isset($targetDimensions[0]) || !isset($targetDimensions[1])
775:                 || !isset($targetDimensions[2]) || !is_int($targetDimensions[0])
776:                 || !is_int($targetDimensions[1]) || !is_int($targetDimensions[2])
777:             ) {
778:                 return null;
779:             }
780: 
781:             $imgMem += ($targetDimensions[0] * $targetDimensions[1] * $targetDimensions[2]);
782:         }
783: 
784:         // read current allocated memory
785:         $curMem = memory_get_usage(true);  // memory_get_usage() is always available with PHP since 5.2.1
786: 
787:         // check if there is enough RAM loading the image(s), plus 3 MB for GD to use for calculations/transforms
788:         return ($phpMaxMem - $curMem >= $imgMem + (3 * 1048576)) ? true : false;
789:     }
