 66:     public function __construct($attr = SmartyPants::ATTR_DEFAULT) {
 67:     #
 68:     # Initialize a SmartyPantsTypographer_Parser with certain attributes.
 69:     #
 70:     # Parser attributes:
 71:     # 0 : do nothing
 72:     # 1 : set all, except dash spacing
 73:     # 2 : set all, except dash spacing, using old school en- and em- dash shortcuts
 74:     # 3 : set all, except dash spacing, using inverted old school en and em- dash shortcuts
 75:     # 
 76:     # Punctuation:
 77:     # q -> quotes
 78:     # b -> backtick quotes (``double'' only)
 79:     # B -> backtick quotes (``double'' and `single')
 80:     # c -> comma quotes (,,double`` only)
 81:     # g -> guillemets (<<double>> only)
 82:     # d -> dashes
 83:     # D -> old school dashes
 84:     # i -> inverted old school dashes
 85:     # e -> ellipses
 86:     # w -> convert &quot; entities to " for Dreamweaver users
 87:     #
 88:     # Spacing:
 89:     # : -> colon spacing +-
 90:     # ; -> semicolon spacing +-
 91:     # m -> question and exclamation marks spacing +-
 92:     # h -> em-dash spacing +-
 93:     # H -> en-dash spacing +-
 94:     # f -> french quote spacing +-
 95:     # t -> thousand separator spacing -
 96:     # u -> unit spacing +-
 97:     #   (you can add a plus sign after some of these options denoted by + to 
 98:     #    add the space when it is not already present, or you can add a minus 
 99:     #    sign to completly remove any space present)
100:     #
101:         # Initialize inherited SmartyPants parser.
102:         parent::__construct($attr);
103:                 
104:         if ($attr == "1" || $attr == "2" || $attr == "3") {
105:             # Do everything, turn all options on.
106:             $this->do_comma_quotes      = 1;
107:             $this->do_guillemets  = 1;
108:             $this->do_space_emdash      = 1;
109:             $this->do_space_endash      = 1;
110:             $this->do_space_colon       = 1;
111:             $this->do_space_semicolon   = 1;
112:             $this->do_space_marks       = 1;
113:             $this->do_space_frenchquote = 1;
114:             $this->do_space_thousand    = 1;
115:             $this->do_space_unit        = 1;
116:         }
117:         else if ($attr == "-1") {
118:             # Special "stupefy" mode.
119:             $this->do_stupefy   = 1;
120:         }
121:         else {
122:             $chars = preg_split('//', $attr);
123:             foreach ($chars as $c){
124:                 if      ($c == "c") { $current =& $this->do_comma_quotes; }
125:                 else if ($c == "g") { $current =& $this->do_guillemets; }
126:                 else if ($c == ":") { $current =& $this->do_space_colon; }
127:                 else if ($c == ";") { $current =& $this->do_space_semicolon; }
128:                 else if ($c == "m") { $current =& $this->do_space_marks; }
129:                 else if ($c == "h") { $current =& $this->do_space_emdash; }
130:                 else if ($c == "H") { $current =& $this->do_space_endash; }
131:                 else if ($c == "f") { $current =& $this->do_space_frenchquote; }
132:                 else if ($c == "t") { $current =& $this->do_space_thousand; }
133:                 else if ($c == "u") { $current =& $this->do_space_unit; }
134:                 else if ($c == "+") {
135:                     $current = 2;
136:                     unset($current);
137:                 }
138:                 else if ($c == "-") {
139:                     $current = -1;
140:                     unset($current);
141:                 }
142:                 else {
143:                     # Unknown attribute option, ignore.
144:                 }
145:                 $current = 1;
146:             }
147:         }
148:     }

171:     function educate($t, $prev_token_last_char) {
172:         $t = parent::educate($t, $prev_token_last_char);
173:         
174:         if ($this->do_comma_quotes)      $t = $this->educateCommaQuotes($t);
175:         if ($this->do_guillemets)        $t = $this->educateGuillemets($t);
176:         
177:         if ($this->do_space_emdash)      $t = $this->spaceEmDash($t);
178:         if ($this->do_space_endash)      $t = $this->spaceEnDash($t);
179:         if ($this->do_space_colon)       $t = $this->spaceColon($t);
180:         if ($this->do_space_semicolon)   $t = $this->spaceSemicolon($t);
181:         if ($this->do_space_marks)       $t = $this->spaceMarks($t);
182:         if ($this->do_space_frenchquote) $t = $this->spaceFrenchQuotes($t);
183:         if ($this->do_space_thousand)    $t = $this->spaceThousandSeparator($t);
184:         if ($this->do_space_unit)        $t = $this->spaceUnit($t);
185:         
186:         return $t;
187:     }

190:     protected function educateQuotes($_) {
191:     #
192:     #   Parameter:  String.
193:     #
194:     #   Returns:    The string, with "educated" curly quote HTML entities.
195:     #
196:     #   Example input:  "Isn't this fun?"
197:     #   Example output: &#8220;Isn&#8217;t this fun?&#8221;
198:     #
199:         $dq_open  = $this->smart_doublequote_open;
200:         $dq_close = $this->smart_doublequote_close;
201:         $sq_open  = $this->smart_singlequote_open;
202:         $sq_close = $this->smart_singlequote_close;
203:     
204:         # Make our own "punctuation" character class, because the POSIX-style
205:         # [:PUNCT:] is only available in Perl 5.6 or later:
206:         $punct_class = "[!\"#\\$\\%'()*+,-.\\/:;<=>?\\@\\[\\\\\]\\^_`{|}~]";
207: 
208:         # Special case if the very first character is a quote
209:         # followed by punctuation at a non-word-break. Close the quotes by brute force:
210:         $_ = preg_replace(
211:             array("/^'(?=$punct_class\\B)/", "/^\"(?=$punct_class\\B)/"),
212:             array($sq_close,                 $dq_close), $_);
213: 
214:         # Special case for double sets of quotes, e.g.:
215:         #   <p>He said, "'Quoted' words in a larger quote."</p>
216:         $_ = preg_replace(
217:             array("/\"'(?=\w)/",     "/'\"(?=\w)/"),
218:             array($dq_open.$sq_open, $sq_open.$dq_open), $_);
219: 
220:         # Special case for decade abbreviations (the '80s):
221:         $_ = preg_replace("/'(?=\\d{2}s)/", $sq_close, $_);
222: 
223:         $close_class = '[^\ \t\r\n\[\{\(\-]';
224:         $dec_dashes = '&\#8211;|&\#8212;';
225: 
226:         # Get most opening single quotes:
227:         $_ = preg_replace("{
228:             (
229:                 \\s          |   # a whitespace char, or
230:                 &nbsp;      |   # a non-breaking space entity, or
231:                 --          |   # dashes, or
232:                 &[mn]dash;  |   # named dash entities
233:                 $dec_dashes |   # or decimal entities
234:                 &\\#x201[34];    # or hex
235:             )
236:             '                   # the quote
237:             (?=\\w)              # followed by a word character
238:             }x", '\1'.$sq_open, $_);
239:         # Single closing quotes:
240:         $_ = preg_replace("{
241:             ($close_class)?
242:             '
243:             (?(1)|          # If $1 captured, then do nothing;
244:               (?=\\s | s\\b)  # otherwise, positive lookahead for a whitespace
245:             )               # char or an 's' at a word ending position. This
246:                             # is a special case to handle something like:
247:                             # \"<i>Custer</i>'s Last Stand.\"
248:             }xi", '\1'.$sq_close, $_);
249: 
250:         # Any remaining single quotes should be opening ones:
251:         $_ = str_replace("'", $sq_open, $_);
252: 
253: 
254:         # Get most opening double quotes:
255:         $_ = preg_replace("{
256:             (
257:                 \\s          |   # a whitespace char, or
258:                 &nbsp;      |   # a non-breaking space entity, or
259:                 --          |   # dashes, or
260:                 &[mn]dash;  |   # named dash entities
261:                 $dec_dashes |   # or decimal entities
262:                 &\\#x201[34];    # or hex
263:             )
264:             \"                   # the quote
265:             (?=\\w)              # followed by a word character
266:             }x", '\1'.$dq_open, $_);
267: 
268:         # Double closing quotes:
269:         $_ = preg_replace("{
270:             ($close_class)?
271:             \"
272:             (?(1)|(?=\\s))   # If $1 captured, then do nothing;
273:                                # if not, then make sure the next char is whitespace.
274:             }x", '\1'.$dq_close, $_);
275: 
276:         # Any remaining quotes should be opening ones.
277:         $_ = str_replace('"', $dq_open, $_);
278: 
279:         return $_;
280:     }

315:     protected function spaceFrenchQuotes($_) {
316:     #
317:     #   Parameters: String, replacement character, and forcing flag.
318:     #   Returns:    The string, with appropriates spaces replaced 
319:     #               inside french-style quotes, only french quotes.
320:     #
321:     #   Example input:  Quotes in « French », »German« and »Finnish» style.
322:     #   Example output: Quotes in «_French_», »German« and »Finnish» style.
323:     #
324:         $opt = ( $this->do_space_frenchquote ==  2 ? '?' : '' );
325:         $chr = ( $this->do_space_frenchquote != -1 ? $this->space_frenchquote : '' );
326:         
327:         # Characters allowed immediatly outside quotes.
328:         $outside_char = $this->space . '|\s|[.,:;!?\[\](){}|@*~=+-]|¡|¿';
329:         
330:         $_ = preg_replace(
331:             "/(^|$outside_char)(&#171;|«|&#8250;|‹)$this->space$opt/",
332:             "\\1\\2$chr", $_);
333:         $_ = preg_replace(
334:             "/$this->space$opt(&#187;|»|&#8249;|›)($outside_char|$)/", 
335:             "$chr\\1\\2", $_);
336:         return $_;
337:     }

536:     #
537:     #   Adding a few more escapes to SmartyPants's escapes:
