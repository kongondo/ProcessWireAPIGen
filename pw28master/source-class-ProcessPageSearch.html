 592:             return $cnt == 0; 
 593: 
 594:         } else if(strpos($name, '.')) {
 595:             list($name, $subname) = explode('.', $name); 
 596:             if(!$this->isSelectableFieldName($subname, $level)) return false;   
 597:         }
 598:         
 599:         if($name == 'path' && (!$this->wire('modules')->isInstalled('PagePaths') || $this->wire('languages'))) {
 600:             $name = 'name';
 601:         }
 602: 
 603:         if(in_array($name, $this->nativeSorts)) $is = true; 
 604:             else if(in_array($name, $selectable)) $is = true;

 607: 
 608:         if($name == 'pass' || $name == 'config' || $name == 'it' || $name == 'display') $is = false;
 609:         
 610:         return $is;
