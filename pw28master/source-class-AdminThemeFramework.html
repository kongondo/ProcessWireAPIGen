306:         $numChildren = count($children);

309:             // ProcessPageAdd: avoid showing this menu item if there are no predefined family settings to use

324:             // no permission specified
325:             
326:             if(!$p->process) {
327:                 // no process module present, so we delegate to just the page viewable state if no children to check
328:                 if($pageViewable && !$numChildren) return true;
