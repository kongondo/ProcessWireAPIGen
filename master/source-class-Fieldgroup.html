388:     /**
389:      * Get all of the Inputfields associated with the provided Page and populate them
390:      *
391:      * @param Page $page
392:      * @param string $contextStr Optional context string to append to all the Inputfield's names (helper for things like repeaters)
393:      * @param string $fieldName Limit to a particular fieldName (may be a Fieldset too, which will include all fields in fieldset)
394:      * @return Inputfield acting as a container for multiple Inputfields
395:      *
396:      */
397:     public function getPageInputfields(Page $page, $contextStr = '', $fieldName = '') {
398: 
399:         $container = new InputfieldWrapper();
400:         $inFieldset = false;
401:         $inModalGroup = '';
402: 
403:         foreach($this as $field) {
404:             
405:             // get a clone in the context of this fieldgroup, if it has contextual settings
406:             if(isset($this->fieldContexts[$field->id])) $field = $this->getField($field->id, true); 
407:             
408:             if($inModalGroup) {
409:                 // we are in a modal group that should be skipped since all the inputs require the modal
410:                 if($field->name == $inModalGroup . "_END") {
411:                     // exit modal group
412:                     $inModalGroup = false; 
413:                 } else {
414:                     // skip field
415:                     continue; 
416:                 }
417:             }
418:             
419:             if($fieldName) {
420:                 // limit to specific field name
421:                 if($inFieldset) {
422:                     // allow the field
423:                     if($field->type instanceof FieldtypeFieldsetClose && $field->name == $fieldName . "_END") {
424:                         // stop, as we've got all the fields we need
425:                         break;
426:                     }
427:                     // allow
428:                     
429:                 } else if($field->name == $fieldName) {
430:                     // start allow fields
431:                     if($field->type instanceof FieldtypeFieldsetOpen) {
432:                         $container = $field->getInputfield($page, $contextStr);
433:                         $inFieldset = true;
434:                         continue; 
435:                     } else {
436:                         // allow 1 field
437:                     }
438:                 } else {
439:                     // disallow
440:                     continue; 
441:                 }
442:                 
443:             } else if($field->modal && $field->type instanceof FieldtypeFieldsetOpen) {
444:                 // field requires modal
445:                 $inModalGroup = $field->name; 
446:             }
447: 
448:             $inputfield = $field->getInputfield($page, $contextStr);
449:             if(!$inputfield) continue; 
450:             if($inputfield->collapsed == Inputfield::collapsedHidden) continue;
451: 
452:             $inputfield->value = $page->get($field->name); 
453:             $container->add($inputfield); 
454:         }       
455: 
456:         return $container; 
457:     }
