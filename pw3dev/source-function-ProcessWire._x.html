  6:  * #pw-summary-translation Provides GetText-like language translation functions to ProcessWire.

 13: /**
 14:  * Perform a language translation
 15:  * 
 16:  * This function enables you to specify static text as the argument, and that text becomes translatable (for each language) 
 17:  * with ProcessWire’s built-in language translation tools. This function also works just fine if you do not have multi-language 
 18:  * support installed, though in that case it just returns the text that it is given. 
 19:  * 
 20:  * For full documentation, please see the [Code Internationalization (i18n)](https://processwire.com/docs/multi-language-support/code-i18n/)
 21:  * documentation page at ProcessWire.com.
 22:  *
 23:  * This function is very similar to the GNU gettext `_('text');` function and essentially does the same thing, except that it is
 24:  * native to ProcessWire and not using GNU gettext. 
 25:  * 
 26:  * Use this `__('text')` function for common translations, use the `_x('text', 'context')` function for translations
 27:  * that also require additional context, and use the `_n('singular', 'plural', $n)` function for translations that should
 28:  * changed based on whether a value `$n` would require a singular or plural tense. 
 29:  * 
 30:  * ### Additional behaviors 
 31:  * 
 32:  * - You can optionally specify a “textdomain” as the second argument. The textdomain represents the source file of the 
 33:  *   translation. Most often it would be the current file, so the argument can be omitted, or you can specify `__FILE__`. 
 34:  *   But in some cases you may want to use a translation from another file, and the textdomain argument enables you to. 
 35:  * 
 36:  * - When in a class (or module) it is preferable to use `$this->_('text');` rather than `__('text');`, as it is slightly
 37:  *   more efficient to do so. 
 38:  * 
 39:  * - A PHP comment, i.e. `// comment` that appears somewhere after a translation call (on the same line) is used as an
 40:  *   additional description for the person translating text. Another comment after that, i.e. `// comment1 // comment2`
 41:  *   is used as an additional secondary note for the person translating the text. 
 42:  *   
 43:  * ### Limitations
 44:  * 
 45:  * - The function call (and translatable text within it) cannot span more than one line. If your translatable text is long 
 46:  *   enough to require multiple lines, split them into multiple calls (like one per sentence). 
 47:  * 
 48:  * - There cannot be more than one `__('text')` function call per line in the PHP code. 
 49:  * 
 50:  * - The provided text argument must be one string of static text. It cannot contain PHP variables or concatenation. To populate
 51:  *   dynamic values you should use PHP’s `sprintf()` (see examples below).
 52:  * 
 53:  * ### Advanced features 
 54:  * 
 55:  * *The following features are for specific cases and supported in ProcessWire 3.0.125+ only.*
 56:  *
 57:  * You can use this function to get or set specific options that affect future calls by specifying boolean true
 58:  * for the first argument, option name for the second argument, and option value for the third argument.
 59:  * Currently there are just two options, “entityEncode” and “translations”. The “entityEncode” option enables
 60:  * you to control whether future calls return entity encoded text or not (see advanced examples).
 61:  *
 62:  * If given an array for the second argument then it is assumed that it is a list of predefined fallback
 63:  * translations you want to define. These translations will be used if the text is not translated in the
 64:  * admin. The translations are not specific to any textdomain and thus can serve as a fallback for any file.
 65:  * These are intended for front-end site usage and not recommended for use in modules. As such, they are not
 66:  * supported by the `$this->_('text');` function either. The array you provide should be associative, where the
 67:  * keys contain the text to translate, and the values contain the translation (see advanced examples).
 68:  * 
 69:  * 
 70:  * ~~~~~~
 71:  * // COMMON USAGE EXAMPLES --------------------------------------------------------------
 72:  * 
 73:  * echo __('This is translatable text');
 74:  * echo __('Translatable with current file as textdomain (optional)', __FILE__);  
 75:  * echo __('Translatable with other file as textdomain', '/site/templates/_init.php');
 76:  * 
 77:  * // using placeholders to populate dynamic values in translatable text:
 78:  * echo sprintf(__('You are reading the %s page'), $page->title); 
 79:  * echo sprintf(__('%d is the current page ID'), $page->id); 
 80:  * echo sprintf(__('Today is %1$s and the time is %2$s'), date('l'), date('g:i a'));
 81:  * 
 82:  * // providing a note via PHP comment to the person doing translation
 83:  * echo __('Welcome friend!'); // A friendly welcome message for new users
 84:  * 
 85:  * // ADVANCED EXAMPLES (3.0.125+) -------------------------------------------------------
 86:  * 
 87:  * // using the entityEncode option
 88:  * // true=always encode, 1=encode only if not already, false=never encode, null=undefined
 89:  * __(true, 'entityEncode', true);
 90:  * 
 91:  * // get current entityEncode option value
 92:  * $val = __(true, 'entityEncode');
 93:  * 
 94:  * // Setting predefined translations
 95:  * if($user->language->name == 'es') {
 96:  *   __(true, [
 97:  *     'Hello' => 'Hola',
 98:  *     'World' => 'Mundo'
 99:  *   ]);
100:  * }
101:  * 
102:  * // Setting predefined translations with context
103:  * __(true, [
104:  *   // would apply only to a _x('Search', 'nav'); call (context)
105:  *   'Search' => [ 'Buscar', 'nav' ]
106:  * ]); 
107:  * ~~~~~~
108:  * 
109:  * #pw-group-translation
110:  * 
111:  * @param string|bool $text Text for translation.
112:  * @param string|array $textdomain Textdomain for the text, may be class name, filename, or something made up by you. 
113:  *   If omitted, a debug backtrace will attempt to determine it automatically.
114:  * @param string|bool|array $context Name of context - DO NOT USE with this function for translation as it will not be parsed for translation. 
115:  *   Use only with the `_x()` function, which will be parsed. 
116:  * @return string|array Translated text or original text if translation not available. Returns array only if getting/setting options.
117:  * @see _x(), _n()  
118:  * @link https://processwire.com/docs/multi-language-support/code-i18n/
119:  *
120:  */
121: function __($text, $textdomain = null, $context = '') {
122:     static $options = array(
123:         'entityEncode' => null, // true=always, false=never, 1=only if not already encoded, null=undefined (backwards compatible behavior)
124:         'translations' => array(), // fallback translations to use when live translation not available ['Original text' => 'Translated text']
125:         '_useLimit' => null, // internal use: use limit argument for debug_backtrace call
126:     );
127:     if($text === true) {
128:         // set and get options
129:         if(is_array($textdomain)) {
130:             // translations specified as array in $textdomain argument
131:             $context = $textdomain;
132:             $textdomain = 'translations';
133:         }
134:         // merge existing translations if specified
135:         if($textdomain == 'translations' && is_array($context)) $context = array_merge($options['translations'], $context);
136:         if($context !== '') $options[$textdomain] = $context;
137:         return $options[$textdomain];
138:     }
139:     if(!wire('languages') || (!$language = wire('user')->language) || !$language->id) {
140:         // multi-language not installed or not available
141:         return $options['entityEncode'] ? htmlspecialchars($text, ENT_QUOTES, 'UTF-8', $options['entityEncode'] === true) : $text;
142:     }
143:     /** @var Language $language */
144:     if($options['_useLimit'] === null) {
145:         $options['_useLimit'] = version_compare(PHP_VERSION, '5.4.0') >= 0;
146:     }
147:     if($textdomain === null) {
148:         if($options['_useLimit']) {
149:             // PHP 5.4.0 or newer
150:             $traces = @debug_backtrace(DEBUG_BACKTRACE_IGNORE_ARGS, 2);
151:         } else if(defined('DEBUG_BACKTRACE_IGNORE_ARGS')) {
152:             // PHP 5.3.6 or newer
153:             $traces = @debug_backtrace(DEBUG_BACKTRACE_IGNORE_ARGS);
154:         } else {
155:             // older PHP (deprecated)
156:             $traces = @debug_backtrace();
157:         }
158:         if(isset($traces[0]) && $traces[0]['file'] != __FILE__) {
159:             $textdomain = $traces[0]['file'];
160:         } else if(isset($traces[1]) && $traces[1]['file'] != __FILE__) {
161:             $textdomain = $traces[1]['file'];
162:         }
163:         if(is_null($textdomain)) $textdomain = 'site';
164:     } else if($textdomain === 'common') {
165:         // common translation
166:         $textdomain = 'wire/modules/LanguageSupport/LanguageTranslator.php';
167:     }
168:     $value = $language->translator()->getTranslation($textdomain, $text, $context);
169:     $encode = $options['entityEncode'];
170:     if($value === "=") {
171:         // translated value should be same as source value
172:         $value = $text;
173:     } else if($value === "+") {
174:         // use common translation value if available
175:         $v = $language->translator()->commonTranslation($text);
176:         $value = empty($v) ? $text : $v;
177:     } else {
178:         // regular translation 
179:         // if translated value same as original check if alternate available in pre-defined translations
180:         if($value === $text && isset($options['translations']["$text"])) {
181:             $value = $options['translations']["$text"];
182:             // array for translation means only apply to named context, ie. 'old text' => [ 'new text', 'context' ]
183:             if(is_array($value)) $value = isset($value[1]) && $value[1] === $context ? $value[0] : $text;
184:         }
185:         // force original behavior fallback if encode mode not set (i.e. encode when translation available)
186:         if($encode === null) $encode = 1; 
187:     }
188:     if($encode) $value = htmlspecialchars($value, ENT_QUOTES, 'UTF-8', $encode === true);
189:     return $value;
190: }

193:  * Perform a language translation in a specific context
194:  * 
195:  * Used when two or more text strings might be the same in default language, but different in other languages. 
196:  * This enables you to limit the context of the translation to a named context, like "button" or "headline" or 
197:  * whatever name you decide to use. 
198:  * 
199:  * ~~~~~
200:  * echo _x('Click for more', 'button');
201:  * echo _x('Click for more', 'text-link'); 
202:  * ~~~~~
203:  * 
204:  * #pw-group-translation
205:  * 
206:  * @param string $text Text for translation. 
207:  * @param string $context Name of context
208:  * @param string $textdomain Textdomain for the text, may be class name, filename, or something made up by you. 
209:  *   If omitted, a debug backtrace will attempt to determine automatically.
210:  * @return string Translated text or original text if translation not available.
211:  * @see __(), _n()
212:  * @link https://processwire.com/docs/multi-language-support/code-i18n/
213:  *
214:  */

219: /**
220:  * Perform a language translation with singular and plural versions
221:  * 
222:  * ~~~~~
223:  * $items = array(...);
224:  * $qty = count($items);
225:  * echo _n('Found one item', 'Found multiple items', $qty); 
226:  * echo sprintf(_n('Found one item', 'Found %d items', $qty), $qty);
227:  * ~~~~~
228:  * 
229:  * #pw-group-translation
230:  * 
231:  * @param string $textSingular Singular version of text (when there is 1 item)
232:  * @param string $textPlural Plural version of text (when there are multiple items or 0 items)
233:  * @param int $count Quantity of items, should be 0 or more.
234:  * @param string $textdomain Textdomain for the text, may be class name, filename, or something made up by you. 
235:  *   If omitted, a debug backtrace will attempt to determine automatically.
236:  * @return string Translated text or original text if translation not available.
237:  * @see __(), _x()
238:  * @link https://processwire.com/docs/multi-language-support/code-i18n/
239:  *
240:  */
241: function _n($textSingular, $textPlural, $count, $textdomain = null) {
242:     return $count == 1 ? __($textSingular, $textdomain) : __($textPlural, $textdomain);     
243: }
