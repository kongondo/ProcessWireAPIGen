 122:     /**
 123:      * Install this Pagefile
 124:      *
 125:      * Implies copying the file to the correct location (if not already there), and populating its name.
 126:      * The given $filename may be local (path) or external (URL). 
 127:      * 
 128:      * #pw-hooker

 131:      * @throws WireException
 132:      *
 133:      */
 134:     protected function ___install($filename) {
 135:     
 136:         $basename = $filename;
 137:         
 138:         if(strpos($basename, '?') !== false) {
 139:             list($basename, $queryString) = explode('?', $basename);    
 140:             if($queryString) {} // do not use in basename
 141:         } 
 142:     
 143:         if(empty($basename)) throw new WireException("Empty filename");
 144: 
 145:         $basename = $this->pagefiles->cleanBasename($basename, true, false, true); 
 146:         $pathInfo = pathinfo($basename); 
 147:         $basename = basename($basename, ".$pathInfo[extension]"); 
 148: 
 149:         $basenameNoExt = $basename; 
 150:         $basename .= ".$pathInfo[extension]"; 
 151: 
 152:         // ensure filename is unique
 153:         $cnt = 0; 
 154:         while(file_exists($this->pagefiles->path() . $basename)) {
 155:             $cnt++;
 156:             $basename = "$basenameNoExt-$cnt.$pathInfo[extension]";
 157:         }
 158:         
 159:         $destination = $this->pagefiles->path() . $basename; 
 160:         
 161:         if(strpos($filename, '://') === false) {
 162:             if(!is_readable($filename)) throw new WireException("Unable to read: $filename");
 163:             if(!copy($filename, $destination)) throw new WireException("Unable to copy: $filename => $destination");
 164:         } else {
 165:             $http = $this->wire(new WireHttp());
 166:             // note: download() method throws excepton on failure
 167:             $http->download($filename, $destination);
 168:             // download was successful
 169:         }
 170:         
 171:         $this->wire('files')->chmod($destination);
 172:         $this->changed('file');
 173:         parent::set('basename', $basename);

 223:      *

 225:      * - To get all values, omit all arguments. An associative array will be returned. 
 226:      * - To set a value, specify the $key and the $value to set.

 230:      * 
 231:      * #pw-internal
 232:      * 
 233:      * @param string|array|false|null $key Specify array to set all file data, or key (string) to set or get a property,
 234:      *  Or specify boolean false to remove key specified by $value argument.
 235:      * @param null|string|array|int|float $value Specify a value to set for given property
 236:      * @return Pagefile|Pageimage|array|string|int|float|bool|null
 237:      *

 239:     public function filedata($key = '', $value = null) {
 240:         $filedata = $this->filedata;
 241:         $changed = false;
 242:         if($key === false || $key === null) {
 243:             // unset property named in $value
 244:             if(!empty($value) && isset($filedata[$value])) {
 245:                 unset($this->filedata[$value]);
 246:                 $changed = true; 
 247:             }
 248:         } else if(empty($key)) {
 249:             // return all
 250:             return $filedata;
 251:         } else if(is_array($key)) {
 252:             // set all
 253:             if($key != $filedata) {
 254:                 $this->filedata = $key;
 255:                 $changed = true; 
 256:             }
 257:         } else if($value === null) {
 258:             // return value for key
 259:             return isset($this->filedata[$key]) ? $this->filedata[$key] : null;
 260:         } else {
 261:             // set value for key
 262:             if(!isset($filedata[$key]) || $filedata[$key] != $value) {
 263:                 $this->filedata[$key] = $value;
 264:                 $changed = true;
 265:             }
 266:         }
 267:         if($changed) {
 268:             $this->trackChange('filedata', $filedata, $this->filedata);
 269:             if($this->page && $this->field) $this->page->trackChange($this->field->name);
 270:         }
 271:         return $this;
 272:     }

 286:         
 287:         /** @var Language|null $language */
 288:         
 289:         $field = $this->field; 
 290:         $noLang = $field && $field->get('noLang'); // noLang setting to disable multi-language from InputfieldFile
 291: 
 292:         if(!is_null($language) && $language->id) {
 293:             $name = "description";
 294:             if(!$language->isDefault() && !$noLang) {
 295:                 $name .= $language->id;
 296:             }
 297:             parent::set($name, $value); 
 298:             if($name != 'description' && $this->isChanged($name)) $this->trackChange('description');
 299:             return $this; 
 300:         }
 301: 
 302:         if(is_array($value)) {
 303:             $values = $value;
 304:         } else {
 305:             // check if it contains JSON?
 306:             $first = substr($value, 0, 1);
 307:             $last = substr($value, -1);
 308:             if(($first == '{' && $last == '}') || ($first == '[' && $last == ']')) {
 309:                 $values = json_decode($value, true);
 310:             } else {
 311:                 $values = array();
 312:             }
 313:         }
 314:         
 315:         $numChanges = 0;
 316: 
 317:         if($values && count($values)) {
 318:             $n = 0; 
 319:             foreach($values as $id => $v) { 
 320:                 // first item is always default language. this ensures description will still
 321:                 // work even if language support is later uninstalled. 
 322:                 $name = 'description';
 323:                 if($noLang && $n > 0) break;
 324:                 $n++; 
 325:                 if(ctype_digit("$id")) {
 326:                     $id = (int) $id;
 327:                     if(!$id) $id = '';
 328:                     $name = $n > 0 ? "description$id" : "description";
 329:                 } else if($id === 'default') {
 330:                     $name = 'description';
 331:                 } else if($languages) {
 332:                     $language = $languages->get($id); // i.e. "default" or "es"
 333:                     if(!$language->id) continue;
 334:                     $name = $language->isDefault() ? "description" : "description$language->id";
 335:                 }
 336:                 parent::set($name, $v);
 337:                 if($this->isChanged($name)) $numChanges++;
 338:             }
 339:         } else {
 340:             // no JSON values so assume regular language description
 341:             $languages = $this->wire('languages');
 342:             $language = $languages ? $this->wire('user')->language : null; 
 343: 
 344:             if($languages && $language && !$noLang && !$language->isDefault()) {
 345:                 $name = "description$language->id";
 346:             } else {
 347:                 $name = "description";
 348:             }
 349:             parent::set($name, $value);
 350:             if($this->isChanged($name)) $numChanges++;
 351:         }
 352:         
 353:         if($numChanges && !$this->isChanged('description')) $this->trackChange('description');
 354: 
 355:         return $this;
 356:     }

 358:     /**
 359:      * Get or set the file’s description (with multi-language support). 
 360:      * 
 361:      * When not in a multi-language environment, you can still use this method but we recommend using the simpler method of just
 362:      * getting/seting the `Pagefile::$description` property directly instead. 
 363:      * 
 364:      * ~~~~~
 365:      * // Get a Pagefile to work with
 366:      * $pagefile = $page->files->first();
 367:      * 
 368:      * // Setting description
 369:      * $pagefile->description('en', 'Setting English description');
 370:      * $pagefile->description('de', 'Setting German description');
 371:      * 
 372:      * // Getting description for current language (whatever it happens to be)
 373:      * echo $pagefile->description();
 374:      * 
 375:      * // Getting description for language "de"
 376:      * echo $pagefile->description('de');
 377:      * ~~~~~
 378:      * 
 379:      * #pw-group-common
 380:      * #pw-group-manipulation
 381:      * 
 382:      * @param null|bool|Language|array
 383:      * - To GET in current user language: Omit arguments or specify null.
 384:      * - To GET in another language: Specify a Language name, id or object.
 385:      * - To GET in all languages as a JSON string: Specify boolean true (if LanguageSupport not installed, regular string returned).
 386:      * - To GET in all languages as an array indexed by language name: Specify boolean true for both arguments.
 387:      * - To SET for a language: Specify a language name, id or object, plus the $value as the 2nd argument.
 388:      * - To SET in all languages as a JSON string: Specify boolean true, plus the JSON string $value as the 2nd argument (internal use only).
 389:      * - To SET in all languages as an array: Specify the array here, indexed by language ID or name, and omit 2nd argument. 
 390:      * @param null|string $value Specify only when you are setting (single language) rather than getting a value.
 391:      * @return string
 392:      *
 393:      */
 394:     public function description($language = null, $value = null) {
 395:         
 396:         if($language === true && $value === true) {
 397:             // return all in array indexed by language name
 398:             /** @var Languages $languages */
 399:             $languages = $this->wire('languages');
 400:             if(!$languages) return array('default' => parent::get('description'));
 401:             $value = array();
 402:             foreach($languages as $language) {
 403:                 $value[$language->name] = (string) parent::get("description" . ($language->isDefault() ? '' : $language->id));
 404:             }
 405:             return $value;  
 406:         }
 407: 
 408:         if(!is_null($value)) {
 409:             // set description mode
 410:             if($language === true) {
 411:                 // set all language descriptions
 412:                 $this->setDescription($value); 
 413:             } else {
 414:                 // set specific language description
 415:                 $this->setDescription($value, $language); 
 416:             }
 417:             return $value; 
 418:         }
 419:         
 420:         if(is_array($language)) {
 421:             // set all from array, then return description in current language
 422:             $this->setDescription($language);   
 423:             $language = null;
 424:             $value = null;
 425:         }
 426: 
 427:         if((is_string($language) || is_int($language)) && $this->wire('languages')) {
 428:             // convert named or ID'd languages to Language object
 429:             $language = $this->wire('languages')->get($language); 
 430:         }
 431: 
 432:         if(is_null($language)) {    
 433:             // return description for current user language, or inherit from default if not available
 434:             $user = $this->wire('user'); 
 435:             $value = null;
 436:             if($user->language && $user->language->id) $value = parent::get("description{$user->language}"); 
 437:             if(empty($value)) {
 438:                 // inherit default language value
 439:                 $value = parent::get("description"); 
 440:             }
 441: 
 442:         } else if($language === true) {
 443:             // return JSON string of all languages if applicable
 444:             $languages = $this->wire('languages'); 
 445:             if($languages && $languages->count() > 1) {
 446:                 $values = array(0 => parent::get("description"));
 447:                 foreach($languages as $lang) {
 448:                     if($lang->isDefault()) continue; 
 449:                     $v = parent::get("description$lang"); 
 450:                     if(empty($v)) continue; 
 451:                     $values[$lang->id] = $v; 
 452:                 }
 453:                 $flags = defined("JSON_UNESCAPED_UNICODE") ? JSON_UNESCAPED_UNICODE | JSON_UNESCAPED_SLASHES : 0; // more fulltext friendly
 454:                 $value = json_encode($values, $flags); 
 455:                 
 456:             } else {
 457:                 // no languages present so just return string with description
 458:                 $value = parent::get("description"); 
 459:             }
 460:             
 461:         } else if(is_object($language) && $language->id) {
 462:             // return description for specific language or blank if not available
 463:             if($language->isDefault()) $value = parent::get("description"); 
 464:                 else $value = parent::get("description$language"); 
 465:         }
 466: 
 467:         // we only return strings, so return blank rather than null
 468:         if(is_null($value)) $value = '';
 469: 
 470:         return $value; 
 471:     }

 473:     /**
 474:      * Get a value from this Pagefile
 475:      * 
 476:      * #pw-internal
 477:      *
 478:      * @param string $key
 479:      * @return mixed Returns null if value does not exist
 480:      *
 481:      */
 482:     public function get($key) {
 483:         $value = null; 
 484: 
 485:         if($key == 'name') $key = 'basename';
 486:         if($key == 'pathname') $key = 'filename';
 487: 
 488:         switch($key) {
 489:             case 'url':
 490:             case 'httpUrl':
 491:             case 'filename':
 492:             case 'description':
 493:             case 'tags':
 494:             case 'ext':
 495:             case 'hash': 
 496:             case 'filesize':
 497:             case 'filesizeStr':
 498:                 // 'basename' property intentionally excluded 
 499:                 $value = $this->$key();
 500:                 break;
 501:             case 'tagsArray': 
 502:                 $value = $this->tags(true);
 503:                 break;
 504:             case 'URL':
 505:                 // nocache url
 506:                 $value = $this->noCacheURL();
 507:                 break;
 508:             case 'HTTPURL':
 509:                 $value = $this->noCacheURL(true);
 510:                 break;
 511:             case 'pagefiles': 
 512:                 $value = $this->pagefiles; 
 513:                 break;
 514:             case 'page': 
 515:                 $value = $this->pagefiles->getPage(); 
 516:                 break;
 517:             case 'field': 
 518:                 $value = $this->pagefiles->getField(); 
 519:                 break;
 520:             case 'modified':
 521:             case 'created':
 522:                 $value = parent::get($key); 
 523:                 if(empty($value)) {
 524:                     $value = filemtime($this->filename()); 
 525:                     parent::set($key, $value); 
 526:                 }
 527:                 break;
 528:             case 'modifiedStr':
 529:             case 'createdStr':
 530:                 $value = parent::get(str_replace('Str', '', $key));
 531:                 $value = wireDate($this->wire('config')->dateFormat, $value);
 532:                 break;
 533:             case 'fileData':
 534:             case 'filedata':
 535:                 $value = $this->filedata();
 536:                 break;
 537:             case 'mtime':
 538:             case 'mtimeStr':
 539:             case 'filemtime':   
 540:             case 'filemtimeStr':    
 541:                 $value = filemtime($this->filename()); 
 542:                 if(strpos($key, 'Str')) $value = wireDate($this->wire('config')->dateFormat, $value);
 543:                 break;
 544:         }
 545:         if(is_null($value)) return parent::get($key); 
 546:         return $value; 
 547:     }

 664:     /**
 665:      * Get or set the "tags" property, when in use. 
 666:      * 
 667:      * ~~~~~
 668:      * $file = $page->files->first();
 669:      * $tags = $file->tags(); // Get tags string
 670:      * $tags = $file->tags(true); // Get tags array
 671:      * $file->tags("foo bar baz"); // Set tags to be these 3 tags
 672:      * $tags->tags(["foo", "bar", "baz"]); // Same as above, using array
 673:      * ~~~~~
 674:      * 
 675:      * #pw-group-tags
 676:      * #pw-group-manipulation
 677:      *
 678:      * @param bool|string|array $value Specify one of the following:
 679:      *   - Omit to simply return the tags as a string. 
 680:      *   - Boolean true if you want to return tags as an array (rather than string). 
 681:      *   - Boolean false to return tags as an array, with lowercase enforced. 
 682:      *   - String or array if you are setting the tags.
 683:      * @return string|array Returns the current tags as a string or an array. 
 684:      *   When an array is returned, it is an associative array where the key and value are both the tag (keys are always lowercase).
 685:      * @see Pagefile::addTag(), Pagefile::hasTag(), Pagefile::removeTag()
 686:      *
 687:      */
 688:     public function tags($value = null) {
 689:         if(is_bool($value)) {
 690:             // return array of tags
 691:             $tags = parent::get('tags');
 692:             $tags = str_replace(array(',', '|'), ' ', $tags);
 693:             $_tags = explode(' ', $tags);
 694:             $tags = array();
 695:             foreach($_tags as $key => $tag) {
 696:                 $tag = trim($tag);
 697:                 if($value === false) $tag = strtolower($tag); // force lowercase
 698:                 if(!strlen($tag)) continue;
 699:                 $tags[strtolower($tag)] = $tag;
 700:             }
 701:         } else if($value !== null) {
 702:             // set tags
 703:             if(is_array($value)) $value = implode(' ', $value); // convert to string
 704:             $value = $this->wire('sanitizer')->text($value);
 705:             if(strpos($value, "\t") !== false) $value = str_replace("\t", " ", $value);
 706:             // collapse extra whitespace
 707:             while(strpos($value, "  ") !== false) $value = str_replace("  ", " ", $value);
 708:             parent::set('tags', $value);    
 709:             $tags = $value; 
 710:         } else {
 711:             // just get tags string
 712:             $tags = parent::get('tags');
 713:         }
 714:         
 715:         return $tags;
 716:     }

 718:     /**
 719:      * Does this file have the given tag(s)?
 720:      *
 721:      * ~~~~~
 722:      * $file = $page->files->first();
 723:      * 
 724:      * if($file->hasTag('foobar')) {
 725:      *   // file has the "foobar" tag
 726:      * }
 727:      *
 728:      * if($file->hasTag("foo|baz")) {
 729:      *   // file has either the foo OR baz tag
 730:      * }
 731:      * 
 732:      * if($file->hasTag("foo,baz")) {
 733:      *  // file has both the foo AND baz tags (since 3.0.17)
 734:      * }
 735:      * ~~~~~
 736:      *
 737:      * #pw-changelog 3.0.17 Added support for AND mode, where multiple tags can be specified and all must be present to return true.
 738:      * #pw-changelog 3.0.17 OR mode now returns found tag rather than boolean true.
 739:      * #pw-group-tags
 740:      *
 741:      * @param string $tag Specify one of the following:
 742:      *  - Single tag without whitespace.
 743:      *  - Multiple tags separated by a "|" to determine if Pagefile has at least one of the tags.
 744:      *  - Multiple tags separated by a comma to determine if Pagefile has all of the tags. (since 3.0.17)
 745:      * @return bool|string True if it has the given tag(s), false if not.
 746:      * - If multiple tags were specified separated by a "|", then if at least one was present, this method returns the found tag.
 747:      * - If multiple tags were specified separated by a space or comma, and all tags are present, it returns true. (since 3.0.17)
 748:      * @see Pagefile::tags(), Pagefile::addTag(), Pagefile::removeTag()
 749:      *
 750:      */
 751:     public function hasTag($tag) {
 752: 
 753:         $tags = $this->tags(false); // all tags in array, lowercase
 754:         if(empty($tags)) return false;
 755:         $modeAND = null;
 756:         $tag = trim(strtolower($tag));
 757:         
 758:         if(strpos($tag, '|') !== false) {
 759:             $findTags = explode('|', $tag);
 760:             $modeAND = false;
 761:         } else if(strpos($tag, ',') !== false) {
 762:             $findTags = explode(',', $tag);
 763:             $modeAND = true;
 764:         } else {
 765:             $findTags = array($tag);
 766:         }
 767: 
 768:         $numTags = 0;
 769:         $numFound = 0;
 770:         $tagFound = '';
 771: 
 772:         foreach($findTags as $tag) {
 773:             $tag = trim($tag);
 774:             if(!strlen($tag)) continue;
 775:             $tag = str_replace(' ', '_', $tag);
 776:             $numTags++;
 777:             if(isset($tags[$tag])) {
 778:                 $numFound++;
 779:                 if($modeAND === false) {
 780:                     $tagFound = $tag;
 781:                     break;
 782:                 }
 783:             }
 784:         }
 785: 
 786:         if($modeAND === false) {
 787:             // OR mode: must have at least one of given tags, and we return the found tag
 788:             return $numFound > 0 ? $tagFound : false;
 789: 
 790:         } else if($modeAND === true) {
 791:             // AND mode: must have all of the given tags
 792:             return $numFound == $numTags;
 793:         }
 794: 
 795:         // single tag
 796:         return $numFound > 0;
 797:     }

 799:     /**
 800:      * Add the given tag to this file’s tags (if not already present)
 801:      * 
 802:      * ~~~~~
 803:      * $file = $page->files->first();
 804:      * $file->addTag('foo'); // add single tag 
 805:      * $file->addTag('foo,bar,baz'); // add multiple tags
 806:      * $file->addTag(['foo', 'bar', 'baz']); // same as above, using array
 807:      * ~~~~~
 808:      * 
 809:      * #pw-group-tags
 810:      * #pw-group-manipulation
 811:      * 
 812:      * @param string|array $tag Tag to add, or array of tags to add, or CSV string of tags to add.
 813:      * @return $this
 814:      * @since 3.0.17
 815:      * @see Pagefile::tags(), Pagefile::hasTag(), Pagefile::removeTag()
 816:      * 
 817:      */
 818:     public function addTag($tag) {
 819:         if(is_array($tag)) {
 820:             $addTags = $tag;
 821:         } else if(strpos($tag, ',') !== false) {
 822:             $addTags = explode(',', $tag);
 823:         } else {
 824:             $addTags = array($tag);
 825:         }
 826:         $tags = $this->tags(true);  
 827:         $numAdded = 0;
 828:         foreach($addTags as $tag) {
 829:             if($this->hasTag($tag)) continue; 
 830:             $tag = $this->wire('sanitizer')->text(trim($tag));
 831:             $tag = str_replace(' ', '_', $tag);
 832:             $tags[strtolower($tag)] = $tag;
 833:             $numAdded++;
 834:         }
 835:         if($numAdded) $this->tags($tags); 
 836:         return $this;
 837:     }

 839:     /**
 840:      * Remove the given tag from this file’s tags (if present)
 841:      * 
 842:      * ~~~~~
 843:      * $file = $page->files->first();
 844:      * $file->removeTag('foo'); // remove single tag
 845:      * $file->removeTag('foo,bar,baz'); // remove multiple tags
 846:      * $file->removeTag(['foo', 'bar', 'baz']); // same as above, using array
 847:      * ~~~~~
 848:      * 
 849:      * #pw-group-tags
 850:      * #pw-group-manipulation
 851:      *
 852:      * @param string $tag Tag to remove, or array of tags to remove, or CSV string of tags to remove. 
 853:      * @return $this
 854:      * @since 3.0.17
 855:      * @see Pagefile::tags(), Pagefile::hasTag(), Pagefile::addTag()
 856:      *
 857:      */
 858:     public function removeTag($tag) {
 859:         $tags = $this->tags(true);
 860:         if(!count($tags)) return $this; // no tags to remove
 861:         if(is_array($tag)) {
 862:             $removeTags = $tag;
 863:         } else if(strpos($tag, ',') !== false) {
 864:             $removeTags = explode(',', $tag); 
 865:         } else {
 866:             $removeTags = array($tag);
 867:         }
 868:         $numRemoved = 0;
 869:         foreach($removeTags as $tag) {
 870:             $tag = strtolower(trim($tag));
 871:             $tag = str_replace(' ', '_', $tag);
 872:             if(!isset($tags[$tag])) continue;
 873:             unset($tags[strtolower($tag)]);
 874:             $numRemoved++;
 875:         }
 876:         if($numRemoved) $this->tags($tags); 
 877:         return $this;       
 878:     }
