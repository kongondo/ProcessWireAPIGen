 34:  * $field = $modules->get('InputfieldEmail');
 35:  * $field->attr('name', 'your_email');
 36:  * $field->label = 'Your Email Address';
 37:  * $field->required = true;
 38:  * $form->add($field);
 39:  *
 40:  * $submit = $modules->get('InputfieldSubmit');

 46:  *   $form->processInput($input->post);

 49:  *     // unsuccessful submit, re-display form

 53:  *     // successful submit (save $name and $email somewhere)
 54:  *     $name = $form->get('your_name')->attr('value');
 55:  *     $email = $form->get('your_email')->attr('value');
