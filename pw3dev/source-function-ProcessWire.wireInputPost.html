  3: /**
  4:  * ProcessWire functions API maps function names to common API variables
  5:  *
  6:  * Provides an alternative to the API variables by providing functions of the same
  7:  * name, with these benefits:
  8:  *
  9:  * - They are always in scope
 10:  * - Makes life simpler in an IDE that recognizes phpdoc, as it can more easily
 11:  *   recognize the types an return values.
 12:  * - In some cases it makes for shorter API calls.
 13:  *
 14:  * The primary drawback is that the function calls are not mapped to a specific
 15:  * instance, so in a multi-instance environment it's possible these function calls
 16:  * may not be referring to the correct ProcessWire instance. For this reason, we
 17:  * think these functions are primarily useful for front-end/website usages, and
 18:  * not as useful for back-end and module development.
 19:  * 
 20:  * Shorter versions of these functions (without the leading "wire") can be found in 
 21:  * FunctionsAPI.php file, which is used only if $config->useFunctionsAPI is true. 
 22:  * The functions in this file are always available regardless of that setting. 
 23:  *
 24:  *
 25:  */
 26: 
 27: /**
 28:  * Common helper for API functions dealing with pages
 29:  * 
 30:  * @param $_apiVar
 31:  * @param $selector
 32:  * @return null|NullPage|Page|PageArray|Pages|PagesType
 33:  * 
 34:  */
 35: function _wirePagesAPI($_apiVar, $selector) {
 36:     
 37:     /** @var Pages|PagesType $pages */
 38:     $pages = is_object($_apiVar) ? $_apiVar : wire($_apiVar);
 39:     if(!$pages) return null;
 40:     if(!$selector) return $pages;
 41: 
 42:     if(is_array($selector) || is_object($selector)) {
 43:         return $pages->find($selector);
 44: 
 45:     } else if(ctype_digit("$selector")) {
 46:         // i.e. "123"
 47:         return $pages->get((int) $selector);
 48: 
 49:     } else if(wireSanitizer('pageName', $selector) === $selector) {
 50:         // i.e. "contact"
 51:         return $pages->get("name=$selector");
 52: 
 53:     } else if(strpos($selector, '/') !== false && wireSanitizer('pagePathName', $selector) === $selector) {
 54:         // i.e. "/path/to/page/"
 55:         return $pages->get($selector);
 56: 
 57:     } else {
 58:         return $pages->find($selector);
 59:     }
 60: }

 62: /**
 63:  * Common helper for API functions dealing with WireData objects
 64:  * 
 65:  * @param $_apiVar
 66:  * @param $key
 67:  * @param $value
 68:  * @return mixed|null|WireData|Page
 69:  * 
 70:  */
 71: function _wireDataAPI($_apiVar, $key, $value) {
 72:     /** @var WireData $item */
 73:     $item = wire($_apiVar);
 74:     if(!$item) return null;
 75:     if(strlen($key)) {
 76:         if(is_null($value)) {
 77:             return $item->get($key);

 87:  *
 88:  * ~~~~
 89:  * // A call with no arguments returns the $pages API variable
 90:  * $pages = pages();

367: /**
368:  * Access the $input API variable as a function
369:  *
370:  * - Default behavior is to return the $input API var.
371:  * - If given just a $type (like "get" or "post"), it will return a WireInputData object for that type.
372:  * - If given a $type and $key it will return the input variable.
373:  * - If all arguments given, the returned value will also be run through the given sanitizer.
374:  *
375:  * ~~~~~
376:  * // Examples
377:  * $input = wireInput(); // Returns $input API var (WireInput)
378:  * $post = wireInput('post'); // Returns $input->post (WireInputData)
379:  * $post = wireInput()->post(); // Same as above
380:  * $value = wireInput('get', 'sort'); // Returns $input->get('sort');
381:  * $value = wireInput('get', 'sort', 'fieldName'); // Returns $input->get('sort') run through $sanitizer->fieldName().
382:  * $value = wireInput('get', 'sort', 'fieldName', 'title'); // Same as above but fallback to 'title' if no sort is present (3.0.125)
383:  * $value = wireInput('get', 'sort', ['title', 'created', 'likes'], 'title'); // Require value to be one given or fallback to 'title' (3.0.125+)
384:  * $value = wireInput()->get('sort', ['title', 'created', 'likes'], 'title'); // Same as above (3.0.125)

514: /**
515:  * Get or set a runtime site setting
516:  *
517:  * This is a simple helper function for maintaining runtime settings in a site profile.
518:  * It simply sets and gets settings that you define. It is preferable to using ProcessWire’s
519:  * `$config` or `config()` API var/function because it is not used to store anything else for
520:  * ProcessWire. It is also preferable to using a variable (or variables) because it is always
521:  * in scope and accessible anywhere in your template files, even within existing functions.
522:  *
523:  * ~~~~~
524:  * // set a setting named “foo” to value “bar”
525:  * setting('foo', 'bar');
526:  *
527:  * // get a setting named “foo”
528:  * $value = setting('foo');
529:  *
530:  * // set or replace multiple settings
531:  * setting([
532:  *   'foo' => 'value',
533:  *   'bar' => 123,
534:  *   'baz' => [ 'foo', 'bar', 'baz' ]
535:  * ]);
536:  *
537:  * // get all settings in associative array
538:  * $a = setting();
539:  *
540:  * // to unset a setting
541:  * setting(false, 'foo');
542:  * ~~~~~
543:  *
544:  * @param string|array $name Setting name, or array to set multiple
545:  * @param string|int|array|float|mixed $value Value to set, or omit if getting value of $name (default=null)
546:  * @return array|string|int|bool|mixed|null
547:  *
548:  */
549: function wireSetting($name = '', $value = null) {
550:     static $settings = array();
551:     if($name === '') return $settings;
552:     if(is_array($name)) return $settings = array_merge($settings, $name);
553:     if($name === false) { unset($settings[(string) $value]); return null; }
554:     if($value !== null) $settings[$name] = $value;
555:     return isset($settings[$name]) ? $settings[$name] : null;
556: }
