 766:      * Clone this page instance
 767:      *
 768:      */
 769:     public function __clone() {
 770:         $track = $this->trackChanges();
 771:         $this->setTrackChanges(false); 
 772:         if($this->filesManager) {
 773:             $this->filesManager = clone $this->filesManager; 
 774:             $this->filesManager->setPage($this);
 775:         }
 776:         foreach($this->template->fieldgroup as $field) {
 777:             $name = $field->name; 
 778:             if(!$field->type->isAutoload() && !isset($this->data[$name])) continue; // important for draft loading
 779:             $value = $this->get($name); 
 780:             // no need to clone non-objects, as they've already been cloned
 781:             // no need to clone Page objects as we still want to reference the original page
 782:             if(!is_object($value) || $value instanceof Page) continue;
 783:             $value2 = clone $value; 
 784:             $this->set($name, $value2); // commit cloned value
 785:             // if value is Pagefiles, then tell it the new page
 786:             if($value2 instanceof PageFieldValueInterface) $value2->setPage($this);
 787:         }
 788:         $this->instanceID .= ".clone";
 789:         if($track) $this->setTrackChanges(true); 
 790:         parent::__clone();
 791:     }

 793:     /**
 794:      * Set the value of a page property
 795:      * 
 796:      * You can set properties to a page using either `$page->set('property', $value);` or `$page->property = $value;`. 
 797:      * 
 798:      * ~~~~~
 799:      * // Set the page title using set() method
 800:      * $page->set('title', 'About Us'); 
 801:      * 
 802:      * // Set the page title directly (equivalent to the above)
 803:      * $page->title = 'About Us';
 804:      * ~~~~~
 805:      * 
 806:      * #pw-group-common
 807:      * #pw-group-manipulation
 808:      *
 809:      * @param string $key Name of property to set
 810:      * @param mixed $value Value to set
 811:      * @return Page|WireData Reference to this Page
 812:      * @see __set
 813:      * @throws WireException
 814:      *
 815:      */
 816:     public function set($key, $value) {
 817:         
 818:         if(isset(self::$basePropertiesAlternates[$key])) $key = self::$basePropertiesAlternates[$key];
 819: 
 820:         if(($key == 'id' || $key == 'name') && $this->settings[$key] && $value != $this->settings[$key]) 
 821:             if( ($key == 'id' && (($this->settings['status'] & Page::statusSystem) || ($this->settings['status'] & Page::statusSystemID))) ||
 822:                 ($key == 'name' && (($this->settings['status'] & Page::statusSystem)))) {
 823:                     throw new WireException("You may not modify '$key' on page '{$this->path}' because it is a system page"); 
 824:         }
 825: 
 826:         switch($key) {
 827:             /** @noinspection PhpMissingBreakStatementInspection */
 828:             case 'id':
 829:                 if(!$this->isLoaded) Page::$loadingStack[(int) $value] = $this;
 830:                 // no break is intentional
 831:             case 'sort': 
 832:             case 'numChildren': 
 833:                 $value = (int) $value; 
 834:                 if($this->settings[$key] !== $value) $this->trackChange($key, $this->settings[$key], $value); 
 835:                 $this->settings[$key] = $value; 
 836:                 break;
 837:             case 'status':
 838:                 $this->setStatus($value); 
 839:                 break;
 840:             case 'statusPrevious':
 841:                 $this->statusPrevious = is_null($value) ? null : (int) $value; 
 842:                 break;
 843:             case 'name':
 844:                 $this->setName($value);
 845:                 break;
 846:             case 'parent': 
 847:             case 'parent_id':
 848:                 if(is_object($value) && $value instanceof Page) {
 849:                     // ok
 850:                     $this->setParent($value);
 851:                 } else if($value && !$this->_parent && 
 852:                     ($key == 'parent_id' || is_int($value) || (is_string($value) && ctype_digit("$value")))) {
 853:                     // store only parent ID so that parent is lazy loaded,
 854:                     // but only if parent hasn't already been previously loaded
 855:                     $this->_parent_id = (int) $value;   
 856:                 } else if($value && (is_string($value) || is_int($value))) {
 857:                     $value = $this->_pages('get', $value);
 858:                     $this->setParent($value);
 859:                 }
 860:                 break;
 861:             case 'parentPrevious':
 862:                 if(is_null($value) || $value instanceof Page) $this->parentPrevious = $value;
