 46:     public static function defaultTransform($text, $attr = SmartyPants::ATTR_DEFAULT) {
 47:     #
 48:     # Initialize the parser and return the result of its transform method.
 49:     # This will work fine for derived classes too.
 50:     #
 51:         # Take parser class on which this function was called.
 52:         $parser_class = \get_called_class();
 53: 
 54:         # try to take parser from the static parser list
 55:         static $parser_list;
 56:         $parser =& $parser_list[$parser_class][$attr];
 57: 
 58:         # create the parser if not already set
 59:         if (!$parser)
 60:             $parser = new $parser_class($attr);
 61: 
 62:         # Transform text using parser.
 63:         return $parser->transform($text);
 64:     }

 84:     public function __construct($attr = SmartyPants::ATTR_DEFAULT) {
 85:     #
 86:     # Initialize a parser with certain attributes.
 87:     #
 88:     # Parser attributes:
 89:     # 0 : do nothing
 90:     # 1 : set all
 91:     # 2 : set all, using old school en- and em- dash shortcuts
 92:     # 3 : set all, using inverted old school en and em- dash shortcuts
 93:     # 
 94:     # q : quotes
 95:     # b : backtick quotes (``double'' only)
 96:     # B : backtick quotes (``double'' and `single')
 97:     # d : dashes
 98:     # D : old school dashes
 99:     # i : inverted old school dashes
100:     # e : ellipses
101:     # w : convert &quot; entities to " for Dreamweaver users
102:     #
103:         if ($attr == "0") {
104:             $this->do_nothing   = 1;
105:         }
106:         else if ($attr == "1") {
107:             # Do everything, turn all options on.
108:             $this->do_quotes    = 1;
109:             $this->do_backticks = 1;
110:             $this->do_dashes    = 1;
111:             $this->do_ellipses  = 1;
112:         }
113:         else if ($attr == "2") {
114:             # Do everything, turn all options on, use old school dash shorthand.
115:             $this->do_quotes    = 1;
116:             $this->do_backticks = 1;
117:             $this->do_dashes    = 2;
118:             $this->do_ellipses  = 1;
119:         }
120:         else if ($attr == "3") {
121:             # Do everything, turn all options on, use inverted old school dash shorthand.
122:             $this->do_quotes    = 1;
123:             $this->do_backticks = 1;
124:             $this->do_dashes    = 3;
125:             $this->do_ellipses  = 1;
126:         }
127:         else if ($attr == "-1") {
128:             # Special "stupefy" mode.
129:             $this->do_stupefy   = 1;
130:         }
131:         else {
132:             $chars = preg_split('//', $attr);
133:             foreach ($chars as $c){
134:                 if      ($c == "q") { $this->do_quotes    = 1; }
135:                 else if ($c == "b") { $this->do_backticks = 1; }
136:                 else if ($c == "B") { $this->do_backticks = 2; }
137:                 else if ($c == "d") { $this->do_dashes    = 1; }
138:                 else if ($c == "D") { $this->do_dashes    = 2; }
139:                 else if ($c == "i") { $this->do_dashes    = 3; }
140:                 else if ($c == "e") { $this->do_ellipses  = 1; }
141:                 else if ($c == "w") { $this->convert_quot = 1; }
142:                 else {
143:                     # Unknown attribute option, ignore.
144:                 }
145:             }
146:         }
147:     }

149:     public function transform($text) {
150: 
151:         if ($this->do_nothing) {
152:             return $text;
153:         }
154: 
155:         $tokens = $this->tokenizeHTML($text);
156:         $result = '';
157:         $in_pre = 0;  # Keep track of when we're inside <pre> or <code> tags.
158: 
159:         $prev_token_last_char = ""; # This is a cheat, used to get some context
160:                                     # for one-character tokens that consist of 
161:                                     # just a quote char. What we do is remember
162:                                     # the last character of the previous text
163:                                     # token, to use as context to curl single-
164:                                     # character quote tokens correctly.
165: 
166:         foreach ($tokens as $cur_token) {
167:             if ($cur_token[0] == "tag") {
168:                 # Don't mess with quotes inside tags.
169:                 $result .= $cur_token[1];
170:                 if (preg_match('@<(/?)(?:'.$this->tags_to_skip.')[\s>]@', $cur_token[1], $matches)) {
171:                     $in_pre = isset($matches[1]) && $matches[1] == '/' ? 0 : 1;
172:                 }
173:             } else {
174:                 $t = $cur_token[1];
175:                 $last_char = substr($t, -1); # Remember last char of this token before processing.
176:                 if (! $in_pre) {
177:                     $t = $this->educate($t, $prev_token_last_char);
178:                 }
179:                 $prev_token_last_char = $last_char;
180:                 $result .= $t;
181:             }
182:         }
183: 
184:         return $result;
185:     }

188:     protected function educate($t, $prev_token_last_char) {
189:         $t = $this->processEscapes($t);
190: 
191:         if ($this->convert_quot) {
192:             $t = preg_replace('/&quot;/', '"', $t);
193:         }
194: 
195:         if ($this->do_dashes) {
196:             if ($this->do_dashes == 1) $t = $this->educateDashes($t);
197:             if ($this->do_dashes == 2) $t = $this->educateDashesOldSchool($t);
198:             if ($this->do_dashes == 3) $t = $this->educateDashesOldSchoolInverted($t);
199:         }
200: 
201:         if ($this->do_ellipses) $t = $this->educateEllipses($t);
202: 
203:         # Note: backticks need to be processed before quotes.
204:         if ($this->do_backticks) {
205:             $t = $this->educateBackticks($t);
206:             if ($this->do_backticks == 2) $t = $this->educateSingleBackticks($t);
207:         }
208: 
209:         if ($this->do_quotes) {
210:             if ($t == "'") {
211:                 # Special case: single-character ' token
212:                 if (preg_match('/\S/', $prev_token_last_char)) {
213:                     $t = "&#8217;";
214:                 }
215:                 else {
216:                     $t = "&#8216;";
217:                 }
218:             }
219:             else if ($t == '"') {
220:                 # Special case: single-character " token
221:                 if (preg_match('/\S/', $prev_token_last_char)) {
222:                     $t = "&#8221;";
223:                 }
224:                 else {
225:                     $t = "&#8220;";
226:                 }
227:             }
228:             else {
229:                 # Normal case:
230:                 $t = $this->educateQuotes($t);
231:             }
232:         }
233: 
234:         if ($this->do_stupefy) $t = $this->stupefyEntities($t);
235:         
236:         return $t;
237:     }

240:     protected function educateQuotes($_) {
241:     #
242:     #   Parameter:  String.
243:     #
244:     #   Returns:    The string, with "educated" curly quote HTML entities.
245:     #
246:     #   Example input:  "Isn't this fun?"
247:     #   Example output: &#8220;Isn&#8217;t this fun?&#8221;
248:     #
249:         # Make our own "punctuation" character class, because the POSIX-style
250:         # [:PUNCT:] is only available in Perl 5.6 or later:
251:         $punct_class = "[!\"#\\$\\%'()*+,-.\\/:;<=>?\\@\\[\\\\\]\\^_`{|}~]";
252: 
253:         # Special case if the very first character is a quote
254:         # followed by punctuation at a non-word-break. Close the quotes by brute force:
255:         $_ = preg_replace(
256:             array("/^'(?=$punct_class\\B)/", "/^\"(?=$punct_class\\B)/"),
257:             array('&#8217;',                 '&#8221;'), $_);
258: 
259: 
260:         # Special case for double sets of quotes, e.g.:
261:         #   <p>He said, "'Quoted' words in a larger quote."</p>
262:         $_ = preg_replace(
263:             array("/\"'(?=\w)/",    "/'\"(?=\w)/"),
264:             array('&#8220;&#8216;', '&#8216;&#8220;'), $_);
265: 
266:         # Special case for decade abbreviations (the '80s):
267:         $_ = preg_replace("/'(?=\\d{2}s)/", '&#8217;', $_);
268: 
269:         $close_class = '[^\ \t\r\n\[\{\(\-]';
270:         $dec_dashes = '&\#8211;|&\#8212;';
271: 
272:         # Get most opening single quotes:
273:         $_ = preg_replace("{
274:             (
275:                 \\s          |   # a whitespace char, or
276:                 &nbsp;      |   # a non-breaking space entity, or
277:                 --          |   # dashes, or
278:                 &[mn]dash;  |   # named dash entities
279:                 $dec_dashes |   # or decimal entities
280:                 &\\#x201[34];    # or hex
281:             )
282:             '                   # the quote
283:             (?=\\w)              # followed by a word character
284:             }x", '\1&#8216;', $_);
285:         # Single closing quotes:
286:         $_ = preg_replace("{
287:             ($close_class)?
288:             '
289:             (?(1)|          # If $1 captured, then do nothing;
290:               (?=\\s | s\\b)  # otherwise, positive lookahead for a whitespace
291:             )               # char or an 's' at a word ending position. This
292:                             # is a special case to handle something like:
293:                             # \"<i>Custer</i>'s Last Stand.\"
294:             }xi", '\1&#8217;', $_);
295: 
296:         # Any remaining single quotes should be opening ones:
297:         $_ = str_replace("'", '&#8216;', $_);
298: 
299: 
300:         # Get most opening double quotes:
301:         $_ = preg_replace("{
302:             (
303:                 \\s          |   # a whitespace char, or
304:                 &nbsp;      |   # a non-breaking space entity, or
305:                 --          |   # dashes, or
306:                 &[mn]dash;  |   # named dash entities
307:                 $dec_dashes |   # or decimal entities
308:                 &\\#x201[34];    # or hex
309:             )
310:             \"                   # the quote
311:             (?=\\w)              # followed by a word character
312:             }x", '\1&#8220;', $_);
313: 
314:         # Double closing quotes:
315:         $_ = preg_replace("{
316:             ($close_class)?
317:             \"
318:             (?(1)|(?=\\s))   # If $1 captured, then do nothing;
319:                                # if not, then make sure the next char is whitespace.
320:             }x", '\1&#8221;', $_);
321: 
322:         # Any remaining quotes should be opening ones.
323:         $_ = str_replace('"', '&#8220;', $_);
324: 
325:         return $_;
326:     }

391:     #
392:     #   Parameter:  String.
393:     #
394:     #   Returns:    The string, with each instance of "--" translated to
395:     #               an em-dash HTML entity, and each "---" translated to
396:     #               an en-dash HTML entity. Two reasons why: First, unlike the
397:     #               en- and em-dash syntax supported by
398:     #               EducateDashesOldSchool(), it's compatible with existing
399:     #               entries written before SmartyPants 1.1, back when "--" was
400:     #               only used for em-dashes.  Second, em-dashes are more
401:     #               common than en-dashes, and so it sort of makes sense that
402:     #               the shortcut should be shorter to type. (Thanks to Aaron
403:     #               Swartz for the idea.)
404:     #
405: 
406:         #                      en         em
407:         $_ = str_replace(array("---",     "--",),
408:                          array('&#8211;', '&#8212;'), $_);
409:         return $_;

430:     #
431:     #   Parameter:  String.
432:     #   Returns:    The string, with each SmartyPants HTML entity translated to
433:     #               its ASCII counterpart.

456:     #
457:     #   Parameter:  String.
458:     #   Returns:    The string, with after processing the following backslash
459:     #               escape sequences. This is useful if you want to force a "dumb"

479:     protected function tokenizeHTML($str) {
480:     #
481:     #   Parameter:  String containing HTML markup.
482:     #   Returns:    An array of the tokens comprising the input
483:     #               string. Each token is either a tag (possibly with nested,
484:     #               tags contained therein, such as <a href="<MTFoo>">, or a
485:     #               run of text between tags. Each element of the array is a
486:     #               two-element array; the first is either 'tag' or 'text';
487:     #               the second is the actual value.
488:     #
489:     #
490:     #   Regular expression derived from the _tokenize() subroutine in 
491:     #   Brad Choate's MTRegex plugin.
492:     #   <http://www.bradchoate.com/past/mtregex.php>
493:     #
494:         $index = 0;
495:         $tokens = array();
496: 
497:         $match = '(?s:<!--.*?-->)|'.    # comment
498:                  '(?s:<\?.*?\?>)|'.             # processing instruction
499:                                                 # regular tags
500:                  '(?:<[/!$]?[-a-zA-Z0-9:]+\b(?>[^"\'>]+|"[^"]*"|\'[^\']*\')*>)'; 
501: 
502:         $parts = preg_split("{($match)}", $str, -1, PREG_SPLIT_DELIM_CAPTURE);
503: 
504:         foreach ($parts as $part) {
505:             if (++$index % 2 && $part != '') 
506:                 $tokens[] = array('text', $part);
507:             else
508:                 $tokens[] = array('tag', $part);
509:         }
510:         return $tokens;
511:     }
