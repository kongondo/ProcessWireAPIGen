 571:     /**
 572:      * Compile include(), require() (and variations) to refer to compiled files where possible
 573:      * 
 574:      * @param string $data
 575:      * @param string $sourceFile
 576:      * 
 577:      */

 579:         
 580:         // other related to includes
 581:         $rawPHP = $this->rawPHP;
 582:         if(strpos($rawPHP, '__DIR__') !== false) {
 583:             $data = str_replace('__DIR__', "'" . dirname($sourceFile) . "'", $data);
 584:             $rawPHP = str_replace('__DIR__', "'" . dirname($sourceFile) . "'", $rawPHP);
 585:         }
 586:         if(strpos($rawPHP, '__FILE__') !== false) {
 587:             $data = str_replace('__FILE__', "'" . $sourceFile . "'", $data);
 588:             $rawPHP = str_replace('__FILE__', "'" . $sourceFile . "'", $rawPHP);
 589:         }
 590:         
 591:         $optionsStr = $this->optionsToString($this->options);
 592:         
 593:         $funcs = array(
 594:             'include_once',
 595:             'include', 
 596:             'require_once',
 597:             'require',
 598:             'wireIncludeFile',
 599:             'wireRenderFile',
 600:             'TemplateFile',
 601:         );
 602: 
 603:         // main include regex
 604:         $re = '/^' . 
 605:             '(.*?)' . // 1: open
 606:             '(' . implode('|', $funcs) . ')' . // 2:function
 607:             '([\( ]+)' . // 3: argOpen: open parenthesis and/or space
 608:             '(["\']?[^;\r\n]+)' . // 4:filename, and rest of the statement (file may be quoted or end with closing parens)
 609:             '([;\r\n])' . // 5:close, whatever the last character is on the line
 610:             '/im';
 611:         
 612:         if(!preg_match_all($re, $rawPHP, $matches)) return;
 613:     
 614:         foreach($matches[0] as $key => $fullMatch) {
 615:     
 616:             // if the include statement looks like one of these below then skip compilation for included file
 617:             // include(/*NoCompile*/__DIR__ . '/file.php');
 618:             // include(__DIR__ . '/file.php'/*NoCompile*/);

 621:             $open = $matches[1][$key];

 646:                 $close = ")$close";
 647:             } 
 648:             
 649:             if(empty($fileMatch)) continue;
 650:             
 651:             if(empty($argOpen)) {
 652:                 // if there was no opening "(", compiler will be adding one, so we'll need an additional corresponding ")"
 653:                 $close = ")$close";
 654:             }
 655:             
 656:             $commaPos = strpos($fileMatch, ',');
 657:             if($commaPos) {
 658:                 // fileMatch contains additional function arguments
 659:                 $argsMatch = substr($fileMatch, $commaPos);
 660:                 $fileMatch = substr($fileMatch, 0, $commaPos);
 661:             }
 662:         
 663:             if(strpos($fileMatch, '"') === 0 || strpos($fileMatch, "'") === 0) {

 665:                 if(strpos($fileMatch, './') === 1) {

 669:                     && strpos($fileMatch, '$') === false
 670:                     && strpos($fileMatch, '(') === false
 671:                     && strpos($fileMatch, '\\') === false) {
 672:                     // i.e. include("file.php")
 673:                     $fileMatch = $fileMatch[0] . dirname($sourceFile) . '/' . substr($fileMatch, 1);
 674:                 }
 675:             }
 676:             
 677:             $fileMatch = str_replace("\t", '', $fileMatch);
 678:             if(strlen($open)) $open .= ' ';
 679:             $ns = __NAMESPACE__ ? "\\ProcessWire" : "";

 681:             $newFullMatch = "$open$funcMatch($ns\\wire('files')->compile($fileMatch,$optionsStr)$argsMatch$close";

 690:                 $ns . '\\wire("config")->paths->root . $1$2',

 692:         }

 696:     /**
 697:      * Test the given line $open preceding an include statement for validity
 698:      * 
 699:      * @param string $open
 700:      * @return bool Returns true if valid, false if not
 701:      * 
 702:      */
 703:     protected function compileIncludesValidLineOpen($open) {
 704:         if(!strlen($open)) return true;
 705:         $skipMatch = false;
 706:         $test = $open;
 707:         foreach(array('"', "'") as $quote) {
 708:             // skip when words like "require" are in a string
 709:             if(strpos($test, $quote) === false) continue;
 710:             $test = str_replace('\\' . $quote, '', $test); // ignore quotes that are escaped
 711:             if(strpos($test, $quote) === false) continue;
 712:             if(substr_count($test, $quote) % 2 > 0) {
 713:                 // there are an uneven number of quotes, indicating that
 714:                 // our $funcMatch is likely part of a quoted string
 715:                 $skipMatch = true;
 716:                 break;
 717:             }
 718:             if($quote == '"' && strpos($test, "'") !== false) {
 719:                 // remove quoted apostrophes so they don't confuse the next iteration
 720:                 $test = preg_replace('/"[^"\']*\'[^"]*"/', '', $test);
 721:             }
 722:         }
 723:         if(!$skipMatch && preg_match('/^[$_a-zA-Z0-9]+$/', substr($open, -1))) {
 724:             // skip things like: something_include(... and $include
 725:             $skipMatch = true;
 726:         }
 727:         return $skipMatch ? false : true;
 728:     }

 730:     /**
 731:      * Returns fileMatch type of 'var', 'file', 'func' or boolean false if not valid
 732:      * 
 733:      * @param string $fileMatch The $fileMatch var from compileIncludes() method
 734:      * @param string $funcMatch include function name
 735:      * @return string|bool 
 736:      * 
 737:      */
 738:     protected function compileIncludesFileMatchType($fileMatch, $funcMatch) {
 739: 
 740:         $fileMatch = trim($fileMatch);
 741:         $isValid = false;
 742: 
 743:         $phpVarSign = strpos($fileMatch, '$');
 744:         $doubleQuote1 = strpos($fileMatch, '"');
 745:         $doubleQuote2 = strrpos($fileMatch, '"');
 746:         $singleQuote1 = strpos($fileMatch, "'");
 747:         $singleQuote2 = strrpos($fileMatch, "'");
 748:         $parenthesis1 = strpos($fileMatch, '(');
 749:         $parenthesis2 = strrpos($fileMatch, ')');
 750:         $testFile = '';
 751: 
 752:         if($phpVarSign === 0) {
 753:             // fileMatch starts with a var name, make sure it at least starts in PHP var format
 754:             if(preg_match('/^\$[_a-zA-Z]/', $fileMatch)) $isValid = 'var';
 755:             
 756:         } else if($doubleQuote1 !== false && $doubleQuote2 > $doubleQuote1) {
 757:             // fileMatch has both open and close double quotes with possibly a filename, so validate extension
 758:             $testFile = substr($fileMatch, $doubleQuote1 + 1, $doubleQuote2 - $doubleQuote1 - 1);
 759: 
 760:         } else if($singleQuote1 !== false && $singleQuote2 > $singleQuote1) {
 761:             // fileMatch has both open and close single quotes with possibly a filename, so validate extension
 762:             $testFile = substr($fileMatch, $singleQuote1 + 1, $singleQuote2 - $singleQuote1 - 1);
 763: 
 764:         } else if($parenthesis1 > 0 && $parenthesis2 > $parenthesis1) {
 765:             // likely a function call, make sure open parenthesis is preceded by PHP name format
 766:             if(preg_match('/[_a-zA-Z][_a-zA-Z0-9]+\(/', $fileMatch)) $isValid = 'func';
 767: 
 768:         } else {
 769:             // likely NOT a valid file match, as it doesn't have any of the expected characters
 770:             $isValid = false;
 771:         }
 772: 
 773:         if($testFile) {
 774:             if(strrpos($testFile, '.')) {
 775:                 // test contains a filename that needs extension validated
 776:                 $parts = explode('.', $testFile);
 777:                 $testExt = array_pop($parts);
 778:                 if($testExt && in_array(strtolower($testExt), $this->extensions)) $isValid = 'file';
 779:             } else if($funcMatch == 'wireRenderFile' || $funcMatch == 'wireIncludeFile') {
 780:                 // these methods don't require a file extension
 781:                 $isValid = 'file';
 782:             }
 783:         }
 784:         
 785:         return $isValid;
 786:     }

 939:         
 940:         $source = rtrim($source, '/') . '/';
 941:         $target = rtrim($target, '/') . '/';
 942:     
 943:         // don't perform full copies of some directories
 944:         // @todo convert this to use the user definable exclusions list
 945:         if($source === $this->wire('config')->paths->site) return 0;
 946:         if($source === $this->wire('config')->paths->siteModules) return 0;
