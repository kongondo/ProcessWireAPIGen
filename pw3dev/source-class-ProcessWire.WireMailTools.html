117:      *
118:      * - Note that the order of arguments is slightly different from PHP's `mail()` function.
119:      * - If no arguments are specified it simply returns a `WireMail` object (see #5 in examples).
120:      * - This function will attempt to use an installed module that extends `WireMail`.

125:      * $mail->send($to, $from, $subject, $body);

134:      * $mail->send($to, $from, $subject, $body, $bodyHTML);
135:      *  
136:      * // 5. Specify a blank call to wireMail() to get the WireMail sending module:
137:      * $wireMail = $mail->send();

141:      * @param string $from Email address FROM. This may be an email address, or a combined name and email address.

143:      * @param string $subject Email subject
144:      * @param string|array $body Email body or omit to move straight to $options
145:      * @param array|string $options Array of options OR the $bodyHTML string. Array $options are:
146:      *  - `body` (string): Email body (text)
147:      *  - `bodyHTML` (string): Email body (HTML)
148:      *  - `replyTo` (string): Reply-to email address
149:      *  - `headers` (array): Associative array of header name => header value
150:      *  - Any additional options will be sent along to the WireMail module or class, in tact.
151:      * @return int|WireMail Returns number of messages sent or WireMail object if no arguments specified.
152:      *

156:         $mail = $this->new();
157:     
158:         // if no $to address specified, return WireMail object
159:         if(empty($to)) return $mail;
160: 
161:         $defaults = array(
162:             'body' => is_string($body) ? $body : '',
163:             'bodyHTML' => '',
164:             'replyTo' => '', // email address
165:             'headers' => array(),
166:         );

168:         if(is_array($body)) {
169:             // use case #2: body is provided in $options
170:             $options = $body;
171:         } else if(is_string($options)) {
172:             // use case #3: body and bodyHTML are provided, but no $options
173:             $options = array('bodyHTML' => $options);

235:      * 
236:      * This is an alternative to using the `$mail->send()` method, and may be simpler for those converting
237:      * existing PHP `mail()` calls to WireMail calls. 
238:      * 
239:      * This function duplicates the same arguments as PHPs mail function, enabling you to replace an existing 
240:      * PHP `mail()` call with `$mail->mail()`.

243:      * an `$options` array for the `$message` argument (rather than a string). See the options array for 
244:      * the `$mail->send()` method for details. 
245:      * ~~~~~

247:      * $mail->mail('ryan@processwire.com', 'Subject', 'Message body');

252:      * // 3. Alternate usage with html and text body
253:      * $mail->mail('ryan@processwire.com', 'Subject', [
254:      *   'bodyHTML' => '<html><body><h1>Message HTML body</h1></body</html>',
255:      *   'body' => 'Message text body',
256:      *   'from' => 'hello@world.com',
257:      * ]);
258:      * ~~~~~
259:      * 
260:      * @param string|array $to Email address TO. For multiple, specify CSV string or array. 
261:      * @param string $subject Email subject
262:      * @param string|array $message Email body (PHP mail style), OR specify $options array with any of the following:
263:      *  - `bodyHTML` (string): Email body (HTML)
264:      *  - `body` (string): Email body (text). If not specified, and bodyHTML is, then text body will be auto-generated.
265:      *  - `from` (string): From email address
266:      *  - `replyTo` (string): Reply-to email address
267:      *  - `headers` (array): Associative array of header name => header value
268:      * @param array $headers Optional additional headers as [name=value] array or "Name: Value" newline-separated string. 
269:      *   Use this argument to duplicate PHP mail() style arguments. No need to use if you used $options array for the $message argument.
270:      * @return bool True on success, false on fail.

274:         $from = '';
275:         
276:         if(is_string($headers)) {
277:             $_headers = explode("\n", $headers); 
278:             $headers = array();
279:             foreach($_headers as $header) {
280:                 if(!strpos($header, ':')) continue;
281:                 list($key, $val) = explode(':', $header, 2);
282:                 $headers[trim($key)] = trim($val);
283:             }
284:         }
285:         
286:         foreach($headers as $key => $val) {
287:             if(strtolower($key) !== 'from') continue;
288:             $from = $val;
289:             unset($headers[$key]); 
290:             break;

294:             // message is $options array
295:             $options = $message;
296:             if(!empty($options['headers'])) $headers = array_merge($headers, $options['headers']);
297:             $options['headers'] = $headers;
298:             if(isset($options['from'])) {
299:                 if(empty($from)) $from = $options['from'];
300:                 unset($options['from']);
301:             }
302:             $qty = $this->send($to, $from, $subject, $options); 
303:             
304:         } else {
305:             // regular PHP style mail() call converted to $mail->send() call
306:             $qty = $this->send($to, $from, $subject, $message, $headers);
