229:      *

231:      * @return string

234:     protected function renderSuccess(Comment $comment = null) {
235: 
236:         $pageID = (int) $this->wire('input')->post->page_id; 
237:         
238:         if($pageID && $this->options['redirectAfterPost']) {
239:             // redirectAfterPost option
240:             $page = $this->wire('pages')->get($pageID); 
241:             if(!$page->viewable() || !$page->id) $page = $this->wire('page');
242:             $url = $page->id ? $page->url : './';
243:             $url .= "?comment_success=1";
244:             if($comment && $comment->id && $comment->status > Comment::statusPending) {
245:                 $url .= "&comment_approved=1#Comment$comment->id";
246:             } else {
247:                 $url .= "#CommentPostNote";
248:             }
249:             $this->wire('session')->set('PageRenderNoCachePage', $page->id); // tell PageRender not to use cache if it exists for this page
250:             $this->wire('session')->redirect($url);
251:             return '';
252:         }
253:         
254:         if(!$this->commentsField || $this->commentsField->moderate == FieldtypeComments::moderateAll) {
255:             // all comments are moderated
256:             $message = $this->options['pendingMessage'];
257:             
258:         } else if($this->commentsField->moderate == FieldtypeComments::moderateNone) {
259:             // no moderation in service
260:             $message = $this->options['successMessage'];
261:             
262:         } else if($comment && $comment->status > Comment::statusPending) {
263:             // comment is approved
264:             $message = $this->options['successMessage'];
265: 
266:         } else if($this->wire('input')->get('comment_approved') == 1) {
267:             // comment was approved in previous request
268:             $message = $this->options['successMessage'];
269:             
270:         } else {
271:             // other/comment still pending
272:             $message = $this->options['pendingMessage'];
273:         }
274:         
275:         return "<div id='CommentPostNote'>$message</div>"; 
276:     }

300:             $inputValues['email'] = $user->email;
301:         }
302:

304:         $divClass = 'new';
305:         $class = trim("CommentForm " . $attrs['class']); 
306:         $note = '';
307: 
308:         /*
309:          * Removed because this is not cache safe! Converted to JS cookie. 
310:          * 
311:         if(is_array($this->session->CommentForm)) {
312:             // submission data available in the session
313:             $sessionValues = $this->session->CommentForm;
314:             foreach($inputValues as $key => $value) {
315:                 if($key == 'text') continue;

323:         foreach($options['presets'] as $key => $value) {
324:             if(!is_null($value)) $inputValues[$key] = $value;

327:         $out = '';
328:         $showForm = true;

345:         }

347:         $form = '';
348:         if($showForm) {

350:                 $form = $this->renderFormThread($id, $class, $attrs, $labels, $inputValues);
351:             } else {

353:             }
