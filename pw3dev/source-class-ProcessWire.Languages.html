274:      * 
275:      * If given no arguments, it sets the current `$user` to have the default language temporarily. It is
276:      * expected you will follow it up with a later call to `$languages->unsetDefault()` to restore the 
277:      * previous language the user had. 
278:      * 
279:      * If given a Language object, it sets that as the default language (for internal use only). 
280:      * 
281:      * ~~~~~
282:      * // set current user to have default language
283:      * $languages->setDefault();
284:      * // perform some operation that has a default language dependency ...
285:      * // then restore the user's previous language with unsetDefault()
286:      * $languages->unsetDefault();
287:      * ~~~~~

291:      *

293:      * 
294:      */
295:     public function setDefault(Language $language = null) {
296:         if(is_null($language)) {
297:             // save current user language setting and make current language default
298:             if(!$this->defaultLanguage) return;
299:             /** @var User $user */
300:             $user = $this->wire('user');
301:             if($user->language->id == $this->defaultLanguage->id) return; // already default
302:             $this->savedLanguage = $user->language;
303:             $previouslyChanged = $user->isChanged('language');  
304:             $user->language = $this->defaultLanguage; 
305:             if(!$previouslyChanged) $user->untrackChange('language');
306:         } else {
307:             // set what language is the default
308:             $this->defaultLanguage = $language; 
309:         }
310:     }

399:      *
400:      * - If the $locale argument is omitted, it uses the locale setting translated for the current user language.
401:      * - You can optionally specify a CSV string of locales to try for the $locale argument. 
402:      * - You can optionally or a “category=locale;category=locale;category=locale” string for the $locale argument.
403:      *   When this type of string is used, the $category argument is ignored. 
404:      * - This method does not accept more than the 3 indicated arguments. 
405:      * - Any of the arguments may be swapped.
406:      * 
407:      * See the PHP setlocale link above for a list of constants that can be used for the `$category` argument. 
408:      * 
409:      * Note that the locale is set once at bootup by ProcessWire, and does not change after that unless you call this
410:      * method. Meaning, a change to `$user->language` does not automatically change the locale. If you want to change
411:      * the locale, you would have to call this method after changing the user’s language from the API side.
412:      * 
413:      * ~~~~~
414:      * // Set locale to whatever settings defined for current $user language
415:      * $languages->setLocale(); 
416:      * 
417:      * // Set all locale categories 
418:      * $languages->setLocale(LC_ALL, 'en_US.UTF-8'); 
419:      * 
420:      * // Set locale for specific category (CTYPE)
421:      * $langauges->setLocale(LC_CTYPE, 'en_US.UTF-8'); 
422:      * 
423:      * // Try multiple locales till one works (in order) using array
424:      * $languages->setLocale(LC_ALL, [ 'en_US.UTF-8', 'en_US', 'en' ]);
425:      * 
426:      * // Same as above, except using CSV string
427:      * $languages->setLocale(LC_ALL, 'en_US.UTF-8, en_US, en'); 
428:      * 
429:      * // Set multiple categories and locales (first argument ignored)
430:      * $languages->setLocale(null, 'LC_CTYPE=en_US;LC_NUMERIC=de_DE;LC_TIME=es_ES'); 
431:      * ~~~~~
432:      * 
433:      * @param int|string|array|null|Language $category Specify a PHP “LC_” constant (int) or omit (or null) for default (LC_ALL).
434:      * @param int|string|array|null|Language $locale Specify string, array or CSV string of locale name(s), 
435:      *   omit (null) for current language locale, or specify Language object to pull locale from that language. 
436:      * @return string|bool Returns the locale that was set or boolean false if requested locale cannot be set.
437:      * @see Languages::getLocale()
438:      *
439:      */
440:     public function setLocale($category = LC_ALL, $locale = null) {
441:         
442:         $setLocale = ''; // return value
443:         
444:         if(!is_int($category)) {
445:             list($category, $locale) = array($locale, $category); // swap arguments
446:         }
447:         
448:         if($category === null) $category = LC_ALL;  
449: 
450:         if($locale === null || is_object($locale)) {
451:             // argument omitted means set according to language settings
452:             $language = $locale instanceof Language ? $locale : $this->wire('user')->language;
453:             $textdomain = 'wire--modules--languagesupport--languagesupport-module';
454:             $locale = $language->translator()->getTranslation($textdomain, 'C');
455:         }
456: 
457:         if(is_string($locale)) {
458:             
459:             if(strpos($locale, ',') !== false) {
460:                 // convert CSV string to array of locales
461:                 $locale = explode(',', $locale);
462:                 foreach($locale as $key => $value) {
463:                     $locale[$key] = trim($value);
464:                 }
465:                 
466:             } else if(strpos($locale, ';') !== false) {
467:                 // multi-category and locale string, i.e. LC_CTYPE=en_US.UTF-8;LC_NUMERIC=C;LC_TIME=C
468:                 foreach(explode(';', $locale) as $s) {
469:                     // call setLocale() for each locale item present in the string
470:                     if(strpos($s, '=') === false) continue;
471:                     list($cats, $loc) = explode('=', $s); 
472:                     $cat = constant($cats);
473:                     if($cat !== null) {
474:                         $loc = $this->setLocale($cat, $loc);
475:                         if($loc !== false) $setLocale .= trim($cats) . '=' . trim($loc) . ";";
476:                     }
477:                 }
478:                 $setLocale = rtrim($setLocale, ';');
479:                 if(empty($setLocale)) $setLocale = false;
480:             }
481:         }
482: 
483:         if($setLocale === '') {
484:             if($locale === '0' || $locale === 0) {
485:                 // get locale (to be consistent with behavior of PHP setlocale)
486:                 $setLocale = $this->getLocale($category);
487:             } else {
488:                 // set the locale
489:                 $setLocale = setlocale($category, $locale);
490:             }
491:         }
492: 
493:         return $setLocale;
494:     }

496:     /**
497:      * Return the current locale setting
498:      * 
499:      * If using LC_ALL category and locales change by category, the returned string will be in 
500:      * the format: “category=locale;category=locale”, and so on. 
501:      * 
502:      * The first and second arguments may optionally be swapped and either can be omitted. 
503:      * 
504:      * @param int|Language|string|null $category Optionally specify a PHP LC constant (default=LC_ALL)
505:      * @param Language|string|int|null $language Optionally return locale for specific language (default=current locale, regardless of language)
506:      * @return string|bool Locale(s) string or boolean false if not supported by the system. 
507:      * @see Languages::setLocale()
508:      * @throws WireException if given a $language argument that is invalid
509:      *
510:      */
511:     public function getLocale($category = LC_ALL, $language = null) {
512:         if(is_int($language)) list($category, $language) = array($language, $category); // argument swap
513:         if($category === null) $category = LC_ALL;
514:         if($language) {
515:             if(!$language instanceof Language) {
516:                 $language = $this->get($language);

518:             }
519:             $locale = $language->translator()->getTranslation('wire--modules--languagesupport--languagesupport-module', 'C');
520:         } else {
521:             $locale = setlocale($category, '0');
522:         }
523:         return $locale;
524:     }

604:      * Get all language specific page-edit permissions, or individually one of them
605:      * 
606:      * #pw-internal
607:      *
608:      * @param string $name Optionally specify a permission or language name to change return value.
609:      * @return array|string|bool Array of Permission names indexed by language name, or:
610:      *  - If given a language name, it will return permission name (if exists) or false if not. 
611:      *  - If given a permission name, it will return the language name (if exists) or false if not. 
612:      *
613:      */
614:     public function getPageEditPermissions($name = '') {
615:         
616:         $prefix = "page-edit-lang-";
617:         
618:         if(!is_array($this->pageEditPermissions)) {
619:             $this->pageEditPermissions = array();
620:             $langNames = array();
621:             foreach($this->wire('languages') as $language) {
622:                 $langNames[$language->name] = $language->name;
623:             }
624:             foreach($this->wire('permissions') as $permission) {
625:                 if(strpos($permission->name, $prefix) !== 0) continue;
626:                 if($permission->name === $prefix . 'none') {
627:                     $this->pageEditPermissions['none'] = $permission->name;
628:                     continue;
629:                 }
630:                 foreach($langNames as $langName) {
631:                     $permissionName = $prefix . $langName;
632:                     if($permission->name === $permissionName) {
633:                         $this->pageEditPermissions[$langName] = $permissionName;
634:                         break;
635:                     }
636:                 }
637:             }
638:         }
639:         
640:         if($name) {
641:             if(strpos($name, $prefix) === 0) {
642:                 // permission name specified: will return language name or false
643:                 return array_search($name, $this->pageEditPermissions);
644:             } else {
645:                 // language name specified: will return permission name or false
646:                 return isset($this->pageEditPermissions[$name]) ? $this->pageEditPermissions[$name] : false;
647:             }
648:             
649:         } else {
650:             return $this->pageEditPermissions;
651:         }
652:     }

675:     /**
676:      * Does current user have edit access for page fields in given language?
677:      * 
678:      * @param Language|int|string $language Language id, name or object, or string "none" to refer to non-multi-language fields
679:      * @return bool True if editable, false if not
680:      * 
681:      */
682:     public function editable($language) {
683:         
684:         $user = $this->wire('user');
685:         if($user->isSuperuser()) return true; 
686:         if(empty($language)) return false;
687:         $cacheKey = "$user->id.$language";
688:         
689:         if(array_key_exists($cacheKey, $this->editableCache)) {
690:             // accounts for 'none', or language ID
691:             return $this->editableCache[$cacheKey];
692:         }
693:         
694:         if($language === 'none') {
695:             // page-edit-lang-none permission applies to non-multilanguage fields, if present
696:             $permissions = $this->getPageEditPermissions();
697:             if(isset($permissions['none'])) {
698:                 // if the 'none' permission exists, then the user must have it in order to edit non-multilanguage fields
699:                 $has = $user->hasPermission('page-edit') && $user->hasPermission($permissions['none']); 
700:             } else {
701:                 // if the page-edit-lang-none permission doesn't exist, then it's not applicable
702:                 $has = $user->hasPermission('page-edit');
703:             }
704:             $this->editableCache[$cacheKey] = $has;
705:             
706:         } else {
707:             
708:             if(!$language instanceof Language) $language = $this->get($this->wire('sanitizer')->pageNameUTF8($language));
709:             if(!$language || !$language->id) return false;
710:         
711:             $cacheKey = "$user->id.$language->id";
712:             
713:             if(array_key_exists($cacheKey, $this->editableCache)) {
714:                 return $this->editableCache[$cacheKey];
715:             }
716:             
717:             if(!$user->hasPermission('page-edit')) {
718:                 // page-edit is a pre-requisite permission
719:                 $has = false;
720:             } else {
721:                 $permissionName = $this->getPageEditPermission($language);
722:                 // if a language-specific page-edit permission doesn't exist, then fallback to regular page-edit permission
723:                 if(!$permissionName) {
724:                     $has = true;
725:                 } else {
726:                     $has = $user->hasPermission($permissionName);
727:                 }
728:             }
729:             
730:             $this->editableCache[$cacheKey] = $has; 
731:         }
732:         
733:         return $has; 
734:     }
