 58:      * @return array of module classname => full pathname to module file

 61:     public function findModuleFiles($path = '', $maxLevel = 4) {
 62:         
 63:         static $level = 0;
 64:         $level++;
 65:         $files = array();
 66:         if(!$path) $path = $this->wire('config')->paths->siteModules;
 67:         
 68:         // find the names of all existing module files, so we can defer to their dirs
 69:         // if a module is being installed that already exists
 70:         
 71:         foreach(new \DirectoryIterator($path) as $file) {
 72:             
 73:             if($file->isDot()) continue; 
 74:             if(substr($file->getBasename(), 0, 1) == '.') continue;
 75:             
 76:             if($file->isDir() && $level < $maxLevel) {
 77:                 $_files = $this->findModuleFiles($file->getPathname());
 78:                 $files = array_merge($_files, $files);

 81:                 $basename = $file->getBasename();
 82:                 if(!strpos($basename, '.module')) continue; 
 83:                 if(!preg_match('/^([A-Z][a-zA-Z0-9_]+)\.module(?:\.php)?$/', $basename, $matches)) continue; 
 84:                 $className = $matches[1]; 
 85:                 $files[$className] = $file->getPathname();
 86:                 
 87:             }
 88:         }
 89:         
 90:         $level--;
 91:         return $files; 
 92:     }

116:                 $dir = dirname($f);
117:                 if($dir != '.') $moduleDirs[$dir] = $dir;
118:             }
119:             if(preg_match('{^(.*?/|)(([A-Z][a-zA-Z0-9_]+)\.module(?:\.php)?)$}', $f, $matches)) {
120:                 $path = $matches[1];
121:                 $name = $matches[3];
122:                 $file = $matches[2];
123:                 $moduleFiles[$name] = $path . $file;
124:             }
125:         }

127:         if(!count($moduleFiles)) {
128:             return false;
129:         }
130: 
131:         // determine which module files are at lowest level in dir tree
132:         $numSlashes = 0;
133:         while(!count($moduleFiles1) && $numSlashes < 5) {

135:                 if(substr_count($file, '/') == $numSlashes) {

137:                 }
138:             }
139:             $numSlashes++;
140:         }

142:         if(count($moduleFiles1) == 1) {
143:             // if only 1 module file, use that as the dir name
144:             reset($moduleFiles1);
145:             $moduleDir = key($moduleFiles1);

148:             return $dir;
149:         }

152:         // in which case we'll defer to that module for the destination dir
153:         $moduleFilesAll = $this->findModuleFiles($modulePath); 
154:         foreach($moduleFiles1 as $name) {
155:             if(isset($moduleFilesAll[$name])) {
156:                 $moduleDir = dirname($moduleFilesAll[$name]);

158:                 if($moduleDir == 'modules') $moduleDir = '';

164:             return $dir;
165:         }

172:             while(isset($sorted[$len])) $len++;
173:             $sorted[$len] = $name;
174:         }
175:         krsort($sorted); // sort by longest to shortest
176:         $moduleFiles1 = array();

241:         } catch(\Exception $e) {

244:             return false;
245:         }

248:             $destinationDir = $this->determineDestinationDir($files); 
249:             if(!$destinationDir) throw new WireException($this->_('Unable to find any module files'));
250:         }
251:         $this->message("Destination directory: $destinationDir", Notice::debug); 
252:         
253:         $files0 = trim($files[0], '/');
254:         $extractedDir = is_dir("$tempDir/$files0") && substr($files0, 0, 1) != '.' ? "$files0/" : "";
255: 
256:         // now create module directory and copy files over
257:         if(is_dir($destinationDir)) {
258:             // destination dir already there, perhaps an older version of same module?
259:             // create a backup of it
260:             $hasBackup = $this->backupDir($destinationDir);
261:             if($hasBackup) $this->wire('files')->mkdir($destinationDir, true); 
262:         } else {
263:             if($this->wire('files')->mkdir($destinationDir, true)) $mkdirDestination = true;
264:             $hasBackup = false; 
265:         }
266: 
267:         // label to identify destinationDir in messages and errors
268:         $dirLabel = str_replace($this->config->paths->root, '/', $destinationDir);
269: 
270:         if(is_dir($destinationDir)) {
271:             $from = $tempDir . $extractedDir;
272:             if($this->wire('files')->copy($from, $destinationDir)) {
273:                 $this->message($this->_('Successfully copied files to new directory:') . ' ' . $dirLabel);
274:                 $this->wire('files')->chmod($destinationDir, true);
275:                 $success = true;
276:             } else {
277:                 $this->error($this->_('Unable to copy files to new directory:') . ' ' . $dirLabel);
278:                 if($hasBackup) $this->restoreDir($destinationDir); 
279:             }
280:         } else {
281:             $this->error($this->_('Could not create directory:') . ' ' . $dirLabel);
282:         }
283: 
284:         if(!$success) {
285:             $this->error($this->_('Unable to copy module files:') . ' ' . $dirLabel);
286:             if($mkdirDestination && !$this->wire('files')->rmdir($destinationDir, true)) {
287:                 $this->error($this->_('Could not delete failed module dir:') . ' ' . $destinationDir, Notice::log);
288:             }
289:         }
290: 
291:         return $success ? $destinationDir : false;
292:     }
