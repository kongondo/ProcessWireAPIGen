284:         $selector = $this->findPagesSelector;

286:         if($this->parent_id) {

288:                 // if a selector was specified, AND a parent, then we'll use the parent as a root

290:             } else {

292:                 $selector = "parent_id={$this->parent_id}"; 
293:             }
294:         }
295: 
296:         if(count($this->template_ids)) {
297:             $selector .= ",templates_id=" . implode('|', $this->template_ids);
298:         } else if($this->template_id) {
299:             $selector .= ",templates_id={$this->template_id}";
300:         }

304:         }

307:         if(!is_null($allowUnpub) && !$allowUnpub) {
308:             $selector .= ",status<" . Page::statusUnpublished;
309:         }
310: 
311:         // allow for full site matches
312:         if(!strlen($selector)) $selector = "id>0";
313: 
314:         // match no more than 50, unless selector specifies it's own limit
315:         if(strpos($selector, 'limit=') === false) $selector .= ",limit=50";
316: 
317:         // replace non-escaped commas with ampersands
318:         $selector = preg_replace('/(?<!\\\\),\s*/', '&', $selector); 
319:
