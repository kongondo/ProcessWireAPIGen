 413: /**
 414:  * Given a string ($str) and values ($vars), replace {tags} in the string with the values
 415:  *
 416:  * - The `$vars` should be an associative array of `[ 'tag' => 'value' ]`.
 417:  * - The `$vars` may also be an object, in which case values will be pulled as properties of the object. 
 418:  *
 419:  * By default, tags are specified in the format: {first_name} where first_name is the name of the
 420:  * variable to pull from $vars, `{` is the opening tag character, and `}` is the closing tag char.
 421:  *
 422:  * The tag parser can also handle subfields and OR tags, if `$vars` is an object that supports that.
 423:  * For instance `{products.title}` is a subfield, and `{first_name|title|name}` is an OR tag. 
 424:  * 
 425:  * ~~~~~
 426:  * $vars = [ 'foo' => 'FOO!', 'bar' => 'BAR!' ];
 427:  * $str = 'This is a test: {foo}, and this is another test: {bar}';
 428:  * echo wirePopulateStringTags($str, $vars); 
 429:  * // outputs: This is a test: FOO!, and this is another test: BAR!
 430:  * ~~~~~
 431:  * 
 432:  * #pw-group-strings
 433:  *
 434:  * @param string $str The string to operate on (where the {tags} might be found)
 435:  * @param WireData|object|array $vars Object or associative array to pull replacement values from. 
 436:  * @param array $options Array of optional changes to default behavior, including: 
 437:  *  - `tagOpen` (string): The required opening tag character(s), default is '{'
 438:  *  - `tagClose` (string): The optional closing tag character(s), default is '}'
 439:  *  - `recursive` (bool): If replacement value contains tags, populate those too? (default=false)
 440:  *  - `removeNullTags` (bool): If a tag resolves to a NULL, remove it? If false, tag will remain. (default=true)
 441:  *  - `entityEncode` (bool): Entity encode the values pulled from $vars? (default=false)
 442:  *  - `entityDecode` (bool): Entity decode the values pulled from $vars? (default=false)
 443:  * @return string String with tags populated. 
 444:  *
 445:  */
 446: function wirePopulateStringTags($str, $vars, array $options = array()) {
 447: 
 448:     $defaults = array(
 449:         // opening tag (required)
 450:         'tagOpen' => '{', 
 451:         // closing tag (optional)
 452:         'tagClose' => '}', 
 453:         // if replacement value contains tags, populate those too?
 454:         'recursive' => false, 
 455:         // if a tag value resolves to a NULL, remove it? If false, tag will be left in tact.
 456:         'removeNullTags' => true, 
 457:         // entity encode values pulled from $vars?
 458:         'entityEncode' => false,    
 459:         // entity decode values pulled from $vars?
 460:         'entityDecode' => false, 
 461:     );
 462: 
 463:     $options = array_merge($defaults, $options); 
 464: 
 465:     // check if this string even needs anything populated
 466:     if(strpos($str, $options['tagOpen']) === false) return $str; 
 467:     if(strlen($options['tagClose']) && strpos($str, $options['tagClose']) === false) return $str; 
 468: 
 469:     // find all tags
 470:     $tagOpen = preg_quote($options['tagOpen']);
 471:     $tagClose = preg_quote($options['tagClose']); 
 472:     $numFound = preg_match_all('/' . $tagOpen . '([-_.|a-zA-Z0-9]+)' . $tagClose . '/', $str, $matches);
 473:     if(!$numFound) return $str; 
 474:     $replacements = array();
 475: 
 476:     // create a list of replacements by finding replacement values in $vars
 477:     foreach($matches[1] as $key => $fieldName) {
 478: 
 479:         $tag = $matches[0][$key];
 480:         if(isset($replacements[$tag])) continue; // if already found, don't continue
 481:         $fieldValue = null;
 482:         
 483:         if(is_object($vars)) {
 484:             if($vars instanceof Page) {
 485:                 $fieldValue = $vars->getMarkup($fieldName);
 486:                 
 487:             } else if($vars instanceof WireData) {
 488:                 $fieldValue = $vars->get($fieldName);
 489:                 
 490:             } else {
 491:                 $fieldValue = $vars->$fieldName;
 492:             }
 493:         } else if(is_array($vars)) {
 494:             $fieldValue = isset($vars[$fieldName]) ? $vars[$fieldName] : null;
 495:         }
 496: 
 497:         if($options['entityEncode']) $fieldValue = htmlentities($fieldValue, ENT_QUOTES, 'UTF-8', false); 
 498:         if($options['entityDecode']) $fieldValue = html_entity_decode($fieldValue, ENT_QUOTES, 'UTF-8'); 
 499: 
 500:         $replacements[$tag] = $fieldValue; 
 501:     }
 502: 
 503:     // replace the tags 
 504:     foreach($replacements as $tag => $value) {
 505: 
 506:         // populate tags recursively, if asked to do so
 507:         if($options['recursive'] && strpos($value, $options['tagOpen'])) {
 508:             $opt = array_merge($options, array('recursive' => false)); // don't go recursive beyond 1 level
 509:             $value = wirePopulateStringTags($value, $vars, $opt); 
 510:         }
 511: 
 512:         // replace tags with replacement values
 513:         if($value !== null || $options['removeNullTags']) {
 514:             $str = str_replace($tag, (string) $value, $str);
 515:         }
 516:     }
 517: 
 518:     return $str; 
 519: }

1136:  * $regions = region('*');
1137:  *

1139:  * region('content', '');
1140:  *
1141:  * // clear all regions
1142:  * region('*', '');
1143:  * ~~~~~
1144:  * 
1145:  * #pw-internal
1146:  *
1147:  * @param string $key Name of region to get or set.
1148:  *  - Specify "*" to retrieve all defined regions in an array.
1149:  *  - Prepend a "+" to the region name to have it prepend your given value to any existing value.
1150:  *  - Append a "+" to the region name to have it append your given value to any existing value.
1151:  *  - Prepend a "++" to region name to make future calls without "+" automatically prepend. 
1152:  *  - Append a "++" to region name to make future calls without "+" to automatically append. 
1153:  * @param null|string $value If setting a region, the text that you want to set.
1154:  * @return string|null|bool|array Returns string of text when getting a region, NULL if region not set, or TRUE if setting region.
1155:  *
1156:  */

1158:     
1159:     static $regions = array();
1160:     static $locked = array();
1161: 
1162:     if(empty($key) || $key === '*') {
1163:         // all regions
1164:         if($value === '') $regions = array(); // clear
1165:         return $regions;
1166:     }
1167: 
1168:     if(is_null($value)) {
1169:         // get region
1170:         $result = isset($regions[$key]) ? $regions[$key] : null;
1171: 
1172:     } else {
1173:         // set region
1174:         $pos = strpos($key, '+');
1175:         if($pos !== false) {
1176:             $lock = strpos($key, '++') !== false;
1177:             $key = trim($key, '+');
1178:             if($lock !== false && !isset($locked[$key])) {
1179:                 $locked[$key] = $lock === 0 ? '^' : '$'; // prepend : append
1180:             }
1181:         }
1182:         $lock = isset($locked[$key]) ? $locked[$key] : '';
1183:         if(!isset($regions[$key])) $regions[$key] = '';
1184:         if($pos === 0 || ($pos === false && $lock == '^')) {
1185:             // prepend

1188:             // append
1189:             $regions[$key] .= $value;
1190:         } else if($value === '') {
1191:             // clear region
1192:             if(!$lock) unset($regions[$key]);
1193:         } else {
1194:             // insert/replace
1195:             $regions[$key] = $value;
1196:         }

1198:     }
