105:     /**
106:      * Render individual RSS item
107:      *
108:      */
109:     protected function renderItem(Page $page) {
110: 
111:         $title = strip_tags($page->get($this->itemTitleField));
112:         if(empty($title)) return '';
113:         $title = html_entity_decode($title, ENT_QUOTES, 'UTF-8'); 
114:         $title = htmlspecialchars($title, ENT_QUOTES, 'UTF-8'); 
115:         $title = str_replace('&#039;', '&apos;', $title); 
116: 
117:         if($this->itemDateField && ($ts = $page->getUnformatted($this->itemDateField))) {
118:             $pubDate = "\t\t<pubDate>" . date(DATE_RSS, $ts) . "</pubDate>\n";
119:         } else {
120:             $pubDate = '';
121:         }
122:     
123:         $author = '';
124:         if($this->itemAuthorField) {
125:             $author = $page->getUnformatted($this->itemAuthorField); 
126:             if(strlen($author)) {
127:                 $author = $this->wire('sanitizer')->entities($author); 
128:                 $author = "\t\t<$this->itemAuthorElement>$author</$this->itemAuthorElement>\n"; 
129:             } else $author = '';
130:         }
131: 
132:         $description = $page->get($this->itemDescriptionField); 
133:         if(is_null($description)) $description = '';
134:         $description = $this->truncateDescription(trim($description)); 
135: 
136:         $out =  "\t<item>\n" .
137:                 "\t\t<title>$title</title>\n" .
138:                 "\t\t<description><![CDATA[$description]]></description>\n" .
139:                 $pubDate . $author . 
140:                 "\t\t<link>{$page->httpUrl}</link>\n" .
141:                 "\n\t<guid>{$page->httpUrl}</guid>\n" .

177:     /**
178:      * Truncate the description to a specific length and then truncate to avoid splitting any words.
179:      *
180:      */
181:     protected function truncateDescription($str) {
182: 
183:         $maxlen = $this->itemDescriptionLength;
184:         if(!$maxlen) return $str;
185: 
186:         // note: tags are not stripped if itemDescriptionLength == 0 and stripTags == true
187:         if($this->stripTags) $str = strip_tags($str); 
188: 
189:         if(strlen($str) < $maxlen) return $str; 
190: 
191:         $str = trim(substr($str, 0, $maxlen)); 
192: 
193:         // boundaries that we can end the summary with
194:         $boundaries = array('. ', '? ', '! ', ', ', '; ', '-');
195:         $bestPos = 0;
196: 
197:         foreach($boundaries as $boundary) {
198:             if(($pos = strrpos($str, $boundary)) !== false) {
199:                 // find the boundary that is furthest in string
200:                 if($pos > $bestPos) $bestPos = $pos;
201:             }
202:         }
203: 
204:         // determine if we should truncate to last punctuation or last space.
205:         // if the last punctuation is further away then 1/4th the total length, then we'll 
206:         // truncate to the last space. Otherwise, we'll truncate to the last punctuation.
207:         $spacePos = strrpos($str, ' '); 
208:         if($spacePos > $bestPos && (($spacePos - ($maxlen / 4)) > $bestPos)) $bestPos = $spacePos; 
209: 
210:         if(!$bestPos) $bestPos = $maxlen;
211: 
212:         return trim(substr($str, 0, $bestPos+1)); 
213:     }
