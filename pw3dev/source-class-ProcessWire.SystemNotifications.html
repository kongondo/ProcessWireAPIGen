755:     /**
756:      * Given a processKey, check for conflicts with other active processKeys
757:      * 
758:      * @param $processKey
759:      * 
760:      */
761:     protected function checkProcessKey($processKey) {
762:         
763:         list($prefix, $className, $pageID, $userID, $windowName) = explode('.', $processKey);
764:         
765:         // locate all currently active processKeys editing $page
766:         $processKeys = $this->wire('cache')->get("$prefix.$className.$pageID.*");
767:         
768:         $notified = $this->wire('session')->getFor($this, 'notifiedProcessKeys'); 
769:         if(!is_array($notified)) $notified = array();
770:         
771:         foreach($processKeys as $_processKey => $times) {
772:             
773:             if(isset($notified[$processKey]) && in_array($_processKey, $notified[$processKey])) continue;
774:             
775:             list($created, $modified) = explode(":", $times);   
776:             list($_prefix, $_className, $_pageID, $_userID, $_windowName) = explode('.', $_processKey);
777:             $recordNotify = false;
778:             
779:             if($_userID == $this->wire('user')->id) {
780:                 // same user
781:                 if($_windowName == $windowName) {
782:                     // this is the window we are already in and this is OK to skip
783:                 } else {
784:                     // editing in different window
785:                     $this->wire('user')->notifications()->error(
786:                         sprintf(
787:                             $this->_('Warning: you are editing this page in another window (editing started %s)'), 
788:                                 wireDate('%X', (int) $created)), 
789:                             Notification::flagAnnoy | Notification::flagSession);
790:                     $recordNotify = true; 
791:                 }
792:                 
793:             } else {
794:                 // different user
795:                 $editingUser = $this->wire('users')->get((int) $_userID);
796:                 if($editingUser->id) {
797:                     $this->wire('user')->notifications()->error(
798:                         sprintf($this->_('Warning: user "%s" is currently editing this page (editing started %s)'),
799:                             $editingUser->name, wireDate('%X', (int) $created)),
800:                                 Notification::flagAnnoy | Notification::flagSession);
801:                     $recordNotify = true; 
802:                 }
803:             }
804:         
805:             if($recordNotify) {
806:                 if(isset($notified[$processKey])) $notified[$processKey][] = $_processKey;
807:                     else $notified[$processKey] = array($_processKey);
808:             }
809:         }
810:         
811:         if(count($notified)) {
812:             $this->wire('session')->setFor($this, 'notifiedProcessKeys', $notified); 
813:         }
814:     }
