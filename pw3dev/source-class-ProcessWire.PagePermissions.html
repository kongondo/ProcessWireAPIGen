   3: /**
   4:  * ProcessWire Page Permissions Module
   5:  *
   6:  * Adds convenience methods to all Page objects for checking permissions, i.e. 
   7:  * 
   8:  * if($page->editable()) { do something }
   9:  * if(!$page->viewable()) { echo "sorry you can't view this"; }
  10:  * ...and so on...
  11:  * 
  12:  * ProcessWire 3.x, Copyright 2016 by Ryan Cramer
  13:  * https://processwire.com
  14:  * 
  15:  * Optional special permissions that are optional (by default, not installed):
  16:  * 
  17:  * 1. page-publish: when installed, editable() returns false, when it would otherwise return true, 
  18:  *    on published pages, if user doesn't have page-publish permission in their roles. 
  19:  * 
  20:  * 2. page-edit-created: when installed, editable() returns false, when it would otherwise 
  21:  *    return true, if user's role has this permission and they are not the $page->createdUser. 
  22:  *    This is a permission that reduces access rather than increasing it. Note that
  23:  *    page-edit-created does nothing if the user doesn't have page-edit permission. 
  24:  * 
  25:  * 3. page-rename: when installed, user must have this permission in their role before they
  26:  *    can change the name of a published page. They can still change the name of an unpublished
  27:  *    page. When not installed, this permission falls back to page-edit. 
  28:  * 
  29:  * 4. page-edit-lang-default: when installed on a multi-language site, user must have this 
  30:  *    permission in order edit multi-language fields in "default" language. This permission
  31:  *    is also required to create or delete pages (if user already has other permissions that enable
  32:  *    them to create or delete pages). 
  33:  * 
  34:  * 5. page-edit-lang-[language_name]: when installed on a multi-language site, user must have
  35:  *    this permission to edit multi-language fields in the [language_name] language.
  36:  * 
  37:  * 6. page-edit-lang-none: when installed on a multi-language site, user must have this 
  38:  *    permission to edit non-multi-language fields. They must also have it to create or delete
  39:  *    pages (if user already has other permissions that enable that). 
  40:  * 
  41:  * 7. user-admin-all: when installed, a user must have this permission in order to edit other
  42:  *    users of all roles (except superuser of course). When installed, the regular user-admin
  43:  *    permission only acts as a pre-requisite for editing users, and enables only listing users, 
  44:  *    and editing users that have only the 'guest' role. The user-admin-all essentially grants 
  45:  *    the same thing as user-admin permission does on a system that has no user-admin-all 
  46:  *    permission installed. That's what it does, but why is it here? The entire purpose is to 
  47:  *    support user-admin-[role] permissions, as described in the next item below: 
  48:  * 
  49:  * 8. user-admin-[role_name]: when installed on a site that also has user-admin-all permission
  50:  *    installed, a user must have this permission (along with user-admin permission) in order
  51:  *    to edit users in role [role_name], or to grant that role to other guest users, or to 
  52:  *    revoke it from them. Think of this permission as granting a user administrative privileges
  53:  *    to just a specific group of users. Note that there would be no reason to combine this
  54:  *    permission with the user-admin-all permission, as user-admin-all permission grants admin
  55:  *    privileges to all roles. 
  56:  * 
  57:  */

 166:      * Hookable implmentation for: Is the given page editable?
 167:      * 
 168:      * @param Page $page
 169:      * @return bool
 170:      *
 171:      */
 172:     protected function ___pageEditable(Page $page) {
 173: 
 174:         $user = $this->wire('user');
 175: 
 176:         // superuser can always do whatever they want
 177:         if($user->isSuperuser()) return true; 
 178: 
 179:         // note there is an exception in the case of system pages, which require superuser to edit
 180:         if(($page->status & Page::statusSystem) && $page->template != 'language') return false;
 181: 
 182:         // If page is locked and user doesn't have permission to unlock, don't let them edit
 183:         if($page->status & Page::statusLocked) {
 184:             if(!$user->hasPermission("page-lock", $page)) return false; 
 185:         }
 186: 
 187:         // special conditions apply if the page is a Language
 188:         if($page->template == 'language' && $user->hasPermission('page-edit') && $user->hasPermission('lang-edit')) {
 189:             return $this->hasLangPermission("page-edit-lang-$page->name", $user);
 190:         }
 191:             
 192:         // special conditions apply if the page is a User
 193:         if($page instanceof User || in_array($page->template->id, $this->wire('config')->userTemplateIDs)) {
 194:             return $this->userEditable($page);
 195:         }
 196: 
 197:         // if the user doesn't have page-edit permission, don't let them go further
 198:         if(!$user->hasPermission("page-edit", $page)) return false;
 199: 
 200:         // check if the system has a page-edit-created permission installed
 201:         if(is_null($this->hasPageEditCreated)) {
 202:             $this->hasPageEditCreated = $this->wire('permissions')->has('page-edit-created');
 203:         }
 204:         
 205:         if($this->hasPageEditCreated && $user->hasPermission('page-edit-created', $page)) {
 206:             // page-edit-created permission is installed, so we have to account for it 
 207:             // if user is not the one that created this page, don't allow them to edit it
 208:             if($page->created_users_id != $user->id) return false;
 209:         }
 210: 
 211:         // now check if there is a page-publish permission in the system, and use it if so
 212:         if(is_null($this->hasPagePublish)) {
 213:             $this->hasPagePublish = $this->wire('permissions')->get('page-publish')->id > 0;
 214:         }
 215:         
 216:         if($this->hasPagePublish) {
 217: 
 218:             // if user has the page-publish permission here, then we're good
 219:             if($user->hasPermission('page-publish', $page)) return true;
 220: 
 221:             // if the page is unpublished then we're fine too
 222:             if($page->hasStatus(Page::statusUnpublished)) return true;
 223: 
 224:             // otherwise user cannot edit this page
 225:             return false;
 226:         }
 227: 
 228:         return true; 
 229:     }

 231:     /**
 232:      * Returns whether the given user ($page) is editable by the current user 
 233:      * 
 234:      * @param User|Page $page
 235:      * @param array $options
 236:      *  - `viewable` (bool): Specify true if only a viewable check is needed (default=false)
 237:      * @return bool
 238:      * 
 239:      */
 240:     public function userEditable(Page $page, array $options = array()) {
 241:         
 242:         /** @var User $user */
 243:         $user = $this->wire('user');
 244:         
 245:         /** @var Process|ProcessProfile|ProcessPageView|ProcessUser|ProcessPageList|ProcessPageLister $process */
 246:         $process = $this->wire('process');
 247:         
 248:         /** @var Config $config */
 249:         $config = $this->wire('config');
 250:         
 251:         $defaults = array(
 252:             'viewable' => false, // specify true if method is being used to determine viewable state
 253:         );
 254:         
 255:         $options = count($options) ? array_merge($defaults, $options) : $defaults;
 256:     
 257:         if(!$page->id) return false;
 258:         if($page->className() != 'User') $page = $this->wire('users')->get($page->id);
 259:         if(!$page || $page instanceof NullPage) return false;
 260:     
 261:         if($user->id === $page->id && !$user->isGuest() && $user->hasPermission('profile-edit')) {
 262:             // user is the same as the page being edited or viewed
 263:             if($process == 'ProcessProfile') {
 264:                 // user editing themself in ProcssProfile
 265:                 return true;
 266:             } else if($this->wire('page') && $this->wire('page')->process == 'ProcessProfile') {
 267:                 // user editing themself in ProcessProfile, when process not yet established
 268:                 return true;
 269:             } else if($process == 'ProcessPageView' && $config->pagefileSecure && $options['viewable']) {
 270:                 // user is viewing a file that is part of their User page when pagefileSecure mode active
 271:                 return $process->getResponseType() == ProcessPageView::responseTypeFile;
 272:             }
 273:         }
 274: 
 275:         // if the current process is something other than ProcessUser, they don't have permission
 276:         if(!$options['viewable']) {
 277:             if($process != 'ProcessUser' && (!$process instanceof ProcessPageList) && (!$process instanceof ProcessPageLister)) {
 278:                 return false;
 279:             }
 280:         }
 281:     
 282:         // if user doesn't have user-admin permission, they have no edit access
 283:         if(!$user->hasPermission('user-admin')) return false;
 284: 
 285:         // if the user page being edited has a superuser role, and the current user doesn't,
 286:         // never let them edit regardless of any other permissions
 287:         $superuserRole = $this->wire('roles')->get($config->superUserRolePageID);
 288:         if($page->roles->has($superuserRole) && !$user->roles->has($superuserRole)) return false;
 289: 
 290:         // if we reach this point then check if there are more granular user-admin permissions available
 291:         // special permissions: user-admin-all, and user-admin-[role]
 292:         $userAdminAll = $this->wire('permissions')->get('user-admin-all');
 293:         
 294:         // if there are no special permissions, then let them through
 295:         if(!$userAdminAll->id) return true;
 296:         
 297:         // if user has user-admin-all permission, they are good to edit
 298:         if($user->hasPermission($userAdminAll)) return true;
 299: 
 300:         // there are role-specific permissions in the system, and user must have appropriate one to edit
 301:         $userEditable = false;
 302:         $guestRoleID = $config->guestUserRolePageID;
 303:         $n = 0;
 304:         foreach($page->roles as $role) {
 305:             $n++;
 306:             if($role->id == $guestRoleID) continue;
 307:             if($user->hasPermission("user-admin-$role->name")) {
 308:                 // found a matching permission for role, so it is editable
 309:                 $userEditable = true;
 310:                 break;
 311:             }
 312:         }
 313:         if($userEditable) return true;
 314:         // if there is only role (guest), then no specific permission needed for that
 315:         if($n == 0 || ($n == 1 && $page->roles->first()->id == $guestRoleID)) return true;
 316:         
 317:         return false;
 318:     }

 414:     /**
 415:      * Is the given field name editable?
 416:      * 
 417:      * This provides the implementation for the Page::fieldEditable($field) method. 
 418:      * 
 419:      * @param Page $page
 420:      * @param string|Field $name Field name
 421:      * @param bool $checkPageEditable Check if the page is editable? Default=true. 
 422:      *  Specify false here as an optimization if you've already confirmed $page is editable.
 423:      * @return bool
 424:      *
 425:      */
 426:     public function fieldEditable(Page $page, $name, $checkPageEditable = true) {
 427:     
 428:         if(empty($name)) return false;
 429:         if($checkPageEditable && !$this->pageEditable($page)) return false;
 430:         if(is_object($name) && $name instanceof Field) $name = $name->name;
 431:         if(!is_string($name)) return false; 
 432:         if(!strlen($name)) return true; 
 433: 
 434:         if($name == 'id' && ($page->status & Page::statusSystemID)) return false; 
 435: 
 436:         $user = $this->wire('user'); 
 437: 
 438:         if($page->status & Page::statusSystem) {
 439:             if(in_array($name, array('id', 'name', 'template', 'templates_id', 'parent', 'parent_id'))) {
 440:                 return false;
 441:             }
 442:         }
 443:         
 444:         if($name == 'template' || $name == 'templates_id') {
 445:             if($page->template->noChangeTemplate) return false;
 446:             if(!$user->hasPermission('page-template', $page)) return false; 
 447:         }
 448:         
 449:         if($name == 'name') {
 450:             // if page has no name (and not homepage), then it needs one, so it is allowed
 451:             if($page->id > 1 && !strlen($page->name)) return true; 
 452:             // if page is not yet published, user with page-edit can still change name
 453:             if($page->isUnpublished()) return true;
 454:             // otherwise verify page-rename permission
 455:             return $user->hasPermission('page-rename', $page);
 456:         }
 457: 
 458:         if($name == 'parent' || $name == 'parent_id') {
 459:             if($page->template->noMove) return false; 
 460:             if(!$user->hasPermission('page-move', $page)) return false; 
 461:         }
 462: 
 463:         if($name == 'sortfield') {
 464:             if(!$user->hasPermission('page-sort', $page)) return false; 
 465:         }
 466: 
 467:         if($name == 'roles') {
 468:             if(!$user->hasPermission('user-admin')) return false; 
 469:         }
 470:         
 471:         if($user->id === $page->id && !$user->isSuperuser() && !$user->hasPermission('user-admin')) {
 472:             return $this->userFieldEditable($name, $user);
 473:         }
 474:         
 475:         // check per-field edit access
 476:         $field = $this->wire('fields')->get($name); 
 477:         if($field && $field->useRoles) {
 478:             return $field->editable($page);
 479:         }
 480: 
 481:         return true; 
 482:     }

 504:     /**
 505:      * Hook for Page::viewable() or Page::viewable($user) method
 506:      * 
 507:      * Is the page viewable by the current user? (or specified user)
 508:      * 
 509:      * - Optionally specify User object to hook as first argument to check for a specific User.
 510:      * - Optionally specify a field name (or Field object) as first argument to check for specific field.
 511:      * - Optionally specify Language object or language name as first argument to check if viewable 
 512:      *   in that language (requires LanguageSupportPageNames module).
 513:      * - Optionally specify boolean false as first or second argument to bypass template filename check.
 514:      *
 515:      * @param HookEvent $event
 516:      *
 517:      */
 518:     public function viewable($event) {
 519: 
 520:         /** @var Page $page */
 521:         $page = $event->object; 
 522:         $viewable = true; 
 523:         $user = $this->wire('user'); 
 524:         $arg0 = $event->arguments(0); 
 525:         $arg1 = $event->arguments(1);
 526:         $field = null; // field name or Field object, if specified as arg0
 527:         $checkFile = true; // return false if template filename doesn't exist
 528:         $status = $page->status;
 529: 
 530:         // allow specifying User instance as argument 0 
 531:         // this gives you a "viewable to user" capability
 532:         if($arg0) {
 533:             if($arg0 instanceof User) {
 534:                 // user specified
 535:                 $user = $arg0;
 536:             } else if($arg0 instanceof Field || is_string($arg0)) {
 537:                 // field name, Field object or language name specified
 538:                 // @todo: prevent possible collision of field name and language name
 539:                 $field = $arg0;
 540:                 $checkFile = false;
 541:             } 
 542:         } 
 543:         if($arg0 === false || $arg1 === false) {
 544:             // bypass template filename check
 545:             $checkFile = false;
 546:         }
 547: 
 548:         // if page has corrupted status, this need not affect viewable access
 549:         if($status & Page::statusCorrupted) $status = $status & ~Page::statusCorrupted;
 550: 
 551:         // perform several viewable checks, in order
 552:         if($status >= Page::statusUnpublished) {
 553:             // unpublished pages are not viewable, but see override below this if/else statement
 554:             $viewable = false;
 555:         } else if(!$page->template || ($checkFile && !$page->template->filenameExists())) {
 556:             // template file does not exist
 557:             $viewable = false;
 558:         } else if($user->isSuperuser()) {
 559:             // superuser always allowed
 560:             $viewable = true;
 561:         } else if($page->process) {
 562:             // delegate access to permissions defined with Process module
 563:             $viewable = $this->processViewable($page);
 564:         } else if($page instanceof User && !$user->isGuest() && ($user->hasPermission('user-admin') || $page->id === $user->id)) {
 565:             // user administrator or user viewing themself
 566:             $viewable = $this->userViewable($page); 
 567:         } else if(!$user->hasPermission("page-view", $page)) {
 568:             // user lacks basic view permission to page
 569:             $viewable = false;
 570:         } else if($page->isTrash()) {
 571:             // pages in trash are not viewable, except to superuser
 572:             $viewable = false;
 573:         }
 574: 
 575:         // if the page is editable by the current user, force it to be viewable (if not viewable due to being unpublished)
 576:         if(!$viewable && !$user->isGuest() && ($status & Page::statusUnpublished)) {
 577:             if($page->editable() && (!$checkFile || $page->template->filenameExists())) $viewable = true; 
 578:         }
 579:         
 580:         if($field && $viewable) {
 581:             $viewable = $this->fieldViewable($page, $field, false);
 582:         }
 583: 
 584:         $event->return = $viewable;
 585:     }
